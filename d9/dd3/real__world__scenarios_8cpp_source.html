69class IoTDataCollectionSystem {
70private:
71 struct SensorReading {
72 std::string device_id;
73 std::string sensor_type;
74 double value;
75 std::chrono::system_clock::time_point timestamp;
76 };
77
78 std::atomic<int> readings_collected_{0};
79 std::atomic<int> batches_sent_{0};
80
81public:
82 void simulate_iot_scenario() {
83 std::cout << "\n=== IoT Data Collection Scenario ===" << std::endl;
84
85 const int num_devices = 5;
86 const int readings_per_device = 20;
87 const int batch_size = 10;
88
89 std::vector<std::string> device_types = {"temperature", "humidity", "pressure"};
90 std::vector<std::thread> device_threads;
91
92 std::queue<SensorReading> sensor_queue;
93 std::mutex queue_mutex;
94 std::condition_variable queue_cv;
95 std::atomic<bool> collection_active{true};
96
97 // Data aggregator thread
98 std::thread aggregator_thread([&]() {
99 std::vector<SensorReading> batch;
100 batch.reserve(batch_size);
101
102 while (collection_active || !sensor_queue.empty()) {
103 std::unique_lock<std::mutex> lock(queue_mutex);
104 queue_cv.wait(lock, [&]() { return !sensor_queue.empty() || !collection_active; });
105
106 while (!sensor_queue.empty() && batch.size() < static_cast<size_t>(batch_size)) {
107 batch.push_back(sensor_queue.front());
108 sensor_queue.pop();
109 }
110
111 if (!batch.empty()) {
112 lock.unlock();
113 send_iot_batch(batch);
114 batch.clear();
115 batches_sent_++;
116 }
117 }
118 });
119
120 // Device simulation threads
121 std::random_device rd;
122 for (int device_id = 0; device_id < num_devices; ++device_id) {
123 device_threads.emplace_back([&, device_id]() {
124 std::mt19937 gen(rd());
125 std::uniform_real_distribution<> temp_dist(18.0, 35.0);
126 std::uniform_real_distribution<> humidity_dist(30.0, 80.0);
127 std::uniform_real_distribution<> pressure_dist(990.0, 1030.0);
128
129 for (int reading = 0; reading < readings_per_device; ++reading) {
130 for (const auto& sensor_type : device_types) {
131 SensorReading sensor_reading;
132 sensor_reading.device_id = "device_" + std::to_string(device_id);
133 sensor_reading.sensor_type = sensor_type;
134 sensor_reading.timestamp = std::chrono::system_clock::now();
135
136 if (sensor_type == "temperature") {
137 sensor_reading.value = temp_dist(gen);
138 } else if (sensor_type == "humidity") {
139 sensor_reading.value = humidity_dist(gen);
140 } else {
141 sensor_reading.value = pressure_dist(gen);
142 }
143
144 {
145 std::lock_guard<std::mutex> lock(queue_mutex);
146 sensor_queue.push(sensor_reading);
147 }
148 queue_cv.notify_one();
149 readings_collected_++;
150
151 std::this_thread::sleep_for(std::chrono::milliseconds(5));
152 }
153 }
154 });
155 }
156
157 for (auto& thread : device_threads) {
158 thread.join();
159 }
160
161 collection_active = false;
162 queue_cv.notify_all();
163 aggregator_thread.join();
164
165 std::cout << "IoT simulation completed:" << std::endl;
166 std::cout << " Readings collected: " << readings_collected_.load() << std::endl;
167 std::cout << " Batches sent: " << batches_sent_.load() << std::endl;
168 }
169
170private:
171 void send_iot_batch(const std::vector<SensorReading>& batch) {
172 auto container = std::make_shared<value_container>();
173 container->set_source("iot_aggregator", "batch_processor");
174 container->set_target("iot_analytics_service", "data_processor");
175 container->set_message_type("sensor_data_batch");
176
177 container->set("batch_size", static_cast<int32_t>(batch.size()));
178 container->set("batch_timestamp", static_cast<int64_t>(
179 std::chrono::duration_cast<std::chrono::milliseconds>(
180 std::chrono::system_clock::now().time_since_epoch()).count()));
181
182 for (size_t i = 0; i < batch.size(); ++i) {
183 const auto& reading = batch[i];
184 std::string prefix = "reading_" + std::to_string(i) + "_";
185
186 container->set(prefix + "device_id", reading.device_id);
187 container->set(prefix + "sensor_type", reading.sensor_type);
188 container->set(prefix + "value", reading.value);
189 container->set(prefix + "timestamp", static_cast<int64_t>(
190 std::chrono::duration_cast<std::chrono::milliseconds>(
191 reading.timestamp.time_since_epoch()).count()));
192 }
193
194 std::string serialized = container->serialize_string(value_container::serialization_format::binary).value();
195 std::cout << " Sent IoT batch: " << batch.size() << " readings, "
196 << serialized.size() << " bytes" << std::endl;
197 }

203class FinancialTransactionSystem {
204private:
205 struct Transaction {
206 std::string transaction_id;
207 std::string account_from;
208 std::string account_to;
209 double amount;
210 std::string currency;
211 std::string transaction_type;
212 std::chrono::system_clock::time_point timestamp;
213 };
214
215 std::atomic<int> transactions_processed_{0};
216 std::atomic<int> fraud_alerts_{0};
217 double total_amount_{0.0};
218 std::mutex amount_mutex_;
219
220public:
221 void simulate_financial_scenario() {
222 std::cout << "\n=== Financial Transaction Processing Scenario ===" << std::endl;
223
224 const int num_transactions = 100;
225
226 std::queue<Transaction> transaction_queue;
227 std::mutex transaction_mutex;
228 std::condition_variable transaction_cv;
229 std::atomic<bool> processing_active{true};
230
231 std::thread fraud_detector([&]() {
232 while (processing_active || !transaction_queue.empty()) {
233 std::unique_lock<std::mutex> lock(transaction_mutex);
234 transaction_cv.wait(lock, [&]() { return !transaction_queue.empty() || !processing_active; });
235
236 if (!transaction_queue.empty()) {
237 Transaction transaction = transaction_queue.front();
238 transaction_queue.pop();
239 lock.unlock();
240
241 process_transaction(transaction);
242 transactions_processed_++;
243 }
244 }
245 });
246
247 std::random_device rd;
248 std::mt19937 gen(rd());
249 std::uniform_real_distribution<> amount_dist(10.0, 10000.0);
250 std::uniform_int_distribution<> account_dist(1000, 9999);
251 std::vector<std::string> currencies = {"USD", "EUR", "GBP"};
252 std::vector<std::string> types = {"transfer", "payment", "withdrawal"};
253
254 for (int i = 0; i < num_transactions; ++i) {
255 Transaction transaction;
256 transaction.transaction_id = "TXN" + std::to_string(1000000 + i);
257 transaction.account_from = "ACC" + std::to_string(account_dist(gen));
258 transaction.account_to = "ACC" + std::to_string(account_dist(gen));
259 transaction.amount = amount_dist(gen);
260 transaction.currency = currencies[i % currencies.size()];
261 transaction.transaction_type = types[i % types.size()];
262 transaction.timestamp = std::chrono::system_clock::now();
263
264 {
265 std::lock_guard<std::mutex> lock(transaction_mutex);
266 transaction_queue.push(transaction);
267 }
268 transaction_cv.notify_one();
269
270 std::this_thread::sleep_for(std::chrono::milliseconds(5));
271 }
272
273 processing_active = false;
274 transaction_cv.notify_all();
275 fraud_detector.join();
276
277 std::cout << "Financial processing completed:" << std::endl;
278 std::cout << " Transactions processed: " << transactions_processed_.load() << std::endl;
279 std::cout << " Fraud alerts generated: " << fraud_alerts_.load() << std::endl;
280 double total;
281 {
282 std::lock_guard<std::mutex> lock(amount_mutex_);
283 total = total_amount_;
284 }
285 std::cout << " Total amount processed: $" << std::fixed << std::setprecision(2) << total << std::endl;
286 }
287
288private:
289 void process_transaction(const Transaction& transaction) {
290 bool is_suspicious = (transaction.amount > 5000.0) ||
291 (transaction.account_from == transaction.account_to);
292
293 {
294 std::lock_guard<std::mutex> lock(amount_mutex_);
295 total_amount_ += transaction.amount;
296 }
297
298 auto container = std::make_shared<value_container>();
299 container->set_source("transaction_processor", "fraud_detection");
300 container->set_target("compliance_service", "transaction_monitor");
301 container->set_message_type(is_suspicious ? "suspicious_transaction" : "normal_transaction");
302
303 container->set("transaction_id", transaction.transaction_id);
304 container->set("account_from", transaction.account_from);
305 container->set("account_to", transaction.account_to);
306 container->set("amount", transaction.amount);
307 container->set("currency", transaction.currency);
308 container->set("transaction_type", transaction.transaction_type);
309 container->set("timestamp", static_cast<int64_t>(
310 std::chrono::duration_cast<std::chrono::milliseconds>(
311 transaction.timestamp.time_since_epoch()).count()));
312 container->set("risk_score", is_suspicious ? 85.0 : 15.0);
313
314 if (is_suspicious) {
315 fraud_alerts_++;
316 container->set("alert_reason",
317 std::string(transaction.amount > 5000.0 ? "high_amount" : "same_account"));
318 container->set("requires_manual_review", true);
319
320 std::cout << " FRAUD ALERT: " << transaction.transaction_id
321 << " Amount: $" << transaction.amount << std::endl;
322 }
323
324 container->serialize_string(value_container::serialization_format::binary).value();
325 }

331class GamingEventSystem {
332private:
333 struct GameEvent {
334 std::string player_id;
335 std::string event_type;
336 std::map<std::string, std::string> event_data;
337 std::chrono::system_clock::time_point timestamp;
338 };
339
340 std::atomic<int> events_processed_{0};
341 std::atomic<int> achievements_unlocked_{0};
342 std::map<std::string, int> player_scores_;
343 std::mutex scores_mutex_;
344
345public:
346 void simulate_gaming_scenario() {
347 std::cout << "\n=== Gaming Event System Scenario ===" << std::endl;
348
349 const int num_players = 10;
350 const int events_per_player = 50;
351 std::vector<std::thread> player_threads;
352
353 std::queue<GameEvent> event_queue;
354 std::mutex event_mutex;
355 std::condition_variable event_cv;
356 std::atomic<bool> game_active{true};
357
358 std::thread event_processor([&]() {
359 while (game_active || !event_queue.empty()) {
360 std::unique_lock<std::mutex> lock(event_mutex);
361 event_cv.wait(lock, [&]() { return !event_queue.empty() || !game_active; });
362
363 if (!event_queue.empty()) {
364 GameEvent event = event_queue.front();
365 event_queue.pop();
366 lock.unlock();
367
368 process_game_event(event);
369 events_processed_++;
370 }
371 }
372 });
373
374 std::random_device rd;
375 for (int player_id = 0; player_id < num_players; ++player_id) {
376 player_threads.emplace_back([&, player_id]() {
377 std::mt19937 gen(rd());
378 std::uniform_int_distribution<> action_dist(0, 4);
379 std::uniform_int_distribution<> score_dist(10, 500);
380 std::uniform_int_distribution<> level_dist(1, 50);
381
382 std::vector<std::string> actions = {"kill", "death", "level_up", "item_collected", "quest_completed"};
383
384 for (int event_count = 0; event_count < events_per_player; ++event_count) {
385 GameEvent event;
386 event.player_id = "player_" + std::to_string(player_id);
387 event.event_type = actions[action_dist(gen)];
388 event.timestamp = std::chrono::system_clock::now();
389
390 if (event.event_type == "kill") {
391 event.event_data["target"] = "player_" + std::to_string((player_id + 1) % num_players);
392 event.event_data["score"] = std::to_string(score_dist(gen));
393 } else if (event.event_type == "level_up") {
394 event.event_data["new_level"] = std::to_string(level_dist(gen));
395 }
396
397 {
398 std::lock_guard<std::mutex> lock(event_mutex);
399 event_queue.push(event);
400 }
401 event_cv.notify_one();
402
403 std::this_thread::sleep_for(std::chrono::milliseconds(20));
404 }
405 });
406 }
407
408 for (auto& thread : player_threads) {
409 thread.join();
410 }
411
412 game_active = false;
413 event_cv.notify_all();
414 event_processor.join();
415
416 print_leaderboard();
417
418 std::cout << "Gaming simulation completed:" << std::endl;
419 std::cout << " Events processed: " << events_processed_.load() << std::endl;
420 std::cout << " Achievements unlocked: " << achievements_unlocked_.load() << std::endl;
421 }
422
423private:
424 void process_game_event(const GameEvent& event) {
425 auto container = std::make_shared<value_container>();
426 container->set_source("game_client", event.player_id);
427 container->set_target("game_server", "event_processor");
428 container->set_message_type("game_event");
429
430 container->set("player_id", event.player_id);
431 container->set("event_type", event.event_type);
432 container->set("timestamp", static_cast<int64_t>(
433 std::chrono::duration_cast<std::chrono::milliseconds>(
434 event.timestamp.time_since_epoch()).count()));
435
436 for (const auto& data_pair : event.event_data) {
437 container->set(data_pair.first, data_pair.second);
438 }
439
440 if (event.event_data.find("score") != event.event_data.end()) {
441 int score = std::stoi(event.event_data.at("score"));
442 {
443 std::lock_guard<std::mutex> lock(scores_mutex_);
444 player_scores_[event.player_id] += score;
445 }
446 }
447
448 if (event.event_type == "level_up" && event.event_data.find("new_level") != event.event_data.end()) {
449 int level = std::stoi(event.event_data.at("new_level"));
450 if (level >= 25) {
451 achievements_unlocked_++;
452 send_achievement_notification(event.player_id, "High Level Achiever");
453 }
454 }
455
456 container->serialize_string(value_container::serialization_format::binary).value();
457 }

460 auto notification = std::make_shared<value_container>();
461 notification->set_source("achievement_system", "unlock_processor");
462 notification->set_target("notification_service", "player_notifier");
463 notification->set_message_type("achievement_unlocked");
464
465 notification->set("player_id", player_id);
466 notification->set("achievement_name", achievement);
467 notification->set("timestamp", static_cast<int64_t>(
468 std::chrono::duration_cast<std::chrono::milliseconds>(
469 std::chrono::system_clock::now().time_since_epoch()).count()));
470
471 std::cout << " ACHIEVEMENT: " << player_id << " unlocked '" << achievement << "'" << std::endl;
472 }
473
474 void print_leaderboard() {
475 std::cout << "\n === Leaderboard ===" << std::endl;
476
477 std::vector<std::pair<std::string, int>> leaderboard;
478 {
479 std::lock_guard<std::mutex> lock(scores_mutex_);
480 for (const auto& entry : player_scores_) {
481 leaderboard.emplace_back(entry.first, entry.second);
482 }
483 }
484
485 std::sort(leaderboard.begin(), leaderboard.end(),
486 [](const auto& a, const auto& b) { return a.second > b.second; });
487
488 for (size_t i = 0; i < std::min(size_t(5), leaderboard.size()); ++i) {
489 std::cout << " " << (i + 1) << ". " << leaderboard[i].first
490 << ": " << leaderboard[i].second << " points" << std::endl;
491 }
492 std::cout << " ===================" << std::endl;
493 }
