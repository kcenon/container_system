90 struct optimized_value
91 {
92 std::string name;
93 value_types type;
94 value_variant data;
95
96 optimized_value()
97 : name("")
98 , type(value_types::null_value)
99 , data(std::monostate{})
100 {}
101
102 optimized_value(const std::string& n, value_types t)
103 : name(n)
104 , type(t)
105 , data(std::monostate{})
106 {}
107
108 size_t memory_footprint() const
109 {
110 size_t base = sizeof(optimized_value);
111 base += name.capacity();
112 if (std::holds_alternative<std::string>(data)) {
113 base += std::get<std::string>(data).capacity();
114 } else if (std::holds_alternative<std::vector<uint8_t>>(data)) {
115 base += std::get<std::vector<uint8_t>>(data).capacity();
116 }
117 return base;
118 }
119
120 bool is_stack_allocated() const
121 {
122 return type != value_types::string_value &&
123 type != value_types::bytes_value &&
124 type != value_types::container_value;
125 }
126 };

131 struct pool_stats
132 {
133 size_t hits{0}; // Pool allocations satisfied
134 size_t misses{0}; // Heap allocations (pool bypassed)
135 size_t small_pool_allocs{0}; // Small pool (<=64 bytes) allocations
136 size_t medium_pool_allocs{0}; // Medium pool (<=256 bytes) allocations
137 size_t deallocations{0}; // Total deallocations
138 size_t available{0}; // Free blocks available
139
140 pool_stats() = default;
141
142 pool_stats(size_t h, size_t m, size_t a)
143 : hits(h), misses(m), available(a)
144 {}
145
146 pool_stats(size_t h, size_t m, size_t sm, size_t med, size_t deallocs, size_t avail)
147 : hits(h), misses(m), small_pool_allocs(sm), medium_pool_allocs(med)
148 , deallocations(deallocs), available(avail)
149 {}
150
151 [[nodiscard]] double hit_rate() const noexcept {
152 auto total = hits + misses;
153 return total > 0 ? static_cast<double>(hits) / static_cast<double>(total) : 0.0;
154 }
155 };
