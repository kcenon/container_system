144 template<typename T>
145 [[nodiscard]] std::optional<T> as() const noexcept
146 {
147 if constexpr (std::is_same_v<T, std::string_view>) {
148 if (type_ == value_types::string_value ||
149 type_ == value_types::bytes_value) {
150 return as_string_view();
151 }
152 return std::nullopt;
153 }
154 else if constexpr (std::is_same_v<T, std::string>) {
155 if (type_ == value_types::string_value ||
156 type_ == value_types::bytes_value) {
157 return as_string();
158 }
159 return std::nullopt;
160 }
161 else if constexpr (std::is_same_v<T, bool>) {
162 if (type_ == value_types::bool_value && value_length_ > 0) {
163 return (value_data_[0] == 't' || value_data_[0] == 'T' ||
164 value_data_[0] == '1');
165 }
166 return std::nullopt;
167 }
168 else if constexpr (std::is_integral_v<T>) {
169 return parse_integral<T>();
170 }
171 else if constexpr (std::is_floating_point_v<T>) {
172 return parse_floating<T>();
173 }
174 else {
175 return std::nullopt;
176 }
177 }

207 template<typename T>
208 [[nodiscard]] std::optional<T> parse_integral() const noexcept
209 {
210 if (!is_integral_type(type_)) {
211 return std::nullopt;
212 }
213
214 if (value_length_ == 0) {
215 return std::nullopt;
216 }
217
218 // Use strtoll/strtoull for parsing
219 char* end = nullptr;
220 std::string temp(value_data_, value_length_);
221
222 if constexpr (std::is_signed_v<T>) {
223 long long val = std::strtoll(temp.c_str(), &end, 10);
224 if (end != temp.c_str() + temp.size()) {
225 return std::nullopt;
226 }
227 return static_cast<T>(val);
228 } else {
229 unsigned long long val = std::strtoull(temp.c_str(), &end, 10);
230 if (end != temp.c_str() + temp.size()) {
231 return std::nullopt;
232 }
233 return static_cast<T>(val);
234 }
235 }
236
240 template<typename T>
241 [[nodiscard]] std::optional<T> parse_floating() const noexcept
242 {
243 if (type_ != value_types::float_value &&
244 type_ != value_types::double_value) {
245 return std::nullopt;
246 }
247
248 if (value_length_ == 0) {
249 return std::nullopt;
250 }
251
252 char* end = nullptr;
253 std::string temp(value_data_, value_length_);
254
255 if constexpr (std::is_same_v<T, float>) {
256 float val = std::strtof(temp.c_str(), &end);
257 if (end != temp.c_str() + temp.size()) {
258 return std::nullopt;
259 }
260 return val;
261 } else {
262 double val = std::strtod(temp.c_str(), &end);
263 if (end != temp.c_str() + temp.size()) {
264 return std::nullopt;
265 }
266 return static_cast<T>(val);
267 }
268 }
269
273 [[nodiscard]] static bool is_integral_type(value_types t) noexcept
274 {
275 switch (t) {
276 case value_types::short_value:
277 case value_types::ushort_value:
278 case value_types::int_value:
279 case value_types::uint_value:
280 case value_types::long_value:
281 case value_types::ulong_value:
282 case value_types::llong_value:
283 case value_types::ullong_value:
284 return true;
285 default:
286 return false;
287 }
288 }
