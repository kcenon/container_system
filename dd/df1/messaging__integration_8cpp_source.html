157std::string messaging_integration::get_metrics_summary() {
158 std::ostringstream oss;
159
160 auto containers = metrics_.containers_created.load(std::memory_order_relaxed);
161 auto serializations = metrics_.serializations_performed.load(std::memory_order_relaxed);
162 auto deserializations = metrics_.deserializations_performed.load(std::memory_order_relaxed);
163 auto serialize_time = metrics_.total_serialize_time_us.load(std::memory_order_relaxed);
164 auto deserialize_time = metrics_.total_deserialize_time_us.load(std::memory_order_relaxed);
165
166 oss << "Container System Metrics:\n";
167 oss << " Containers created: " << containers << "\n";
168 oss << " Serializations: " << serializations << "\n";
169 oss << " Deserializations: " << deserializations << "\n";
170
171 if (serializations > 0) {
172 oss << " Avg serialize time: " << std::fixed << std::setprecision(2)
173 << (static_cast<double>(serialize_time) / serializations) << " Œºs\n";
174 }
175
176 if (deserializations > 0) {
177 oss << " Avg deserialize time: " << std::fixed << std::setprecision(2)
178 << (static_cast<double>(deserialize_time) / deserializations) << " Œºs\n";
179 }
180
181 return oss.str();
182}
