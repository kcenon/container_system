158std::string messaging_integration::get_metrics_summary() {
159 std::ostringstream oss;
160
161 auto containers = metrics_.containers_created.load(std::memory_order_relaxed);
162 auto serializations = metrics_.serializations_performed.load(std::memory_order_relaxed);
163 auto deserializations = metrics_.deserializations_performed.load(std::memory_order_relaxed);
164 auto serialize_time = metrics_.total_serialize_time_us.load(std::memory_order_relaxed);
165 auto deserialize_time = metrics_.total_deserialize_time_us.load(std::memory_order_relaxed);
166
167 oss << "Container System Metrics:\n";
168 oss << " Containers created: " << containers << "\n";
169 oss << " Serializations: " << serializations << "\n";
170 oss << " Deserializations: " << deserializations << "\n";
171
172 if (serializations > 0) {
173 oss << " Avg serialize time: " << std::fixed << std::setprecision(2)
174 << (static_cast<double>(serialize_time) / serializations) << " Œºs\n";
175 }
176
177 if (deserializations > 0) {
178 oss << " Avg deserialize time: " << std::fixed << std::setprecision(2)
179 << (static_cast<double>(deserialize_time) / deserializations) << " Œºs\n";
180 }
181
182 return oss.str();
183}
