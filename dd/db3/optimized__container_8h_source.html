120 [[nodiscard]] std::shared_ptr<value> get_value_fast(
121 std::string_view target_name,
122 unsigned int index = 0
123 ) {
124 // Build index on first access
125 if (value_index_.empty() && !units_.empty()) {
126 build_value_index();
127 }
128
129 auto it = value_index_.find(std::string(target_name));
130 if (it == value_index_.end() || it->second.empty()) {
131 // Fallback to parent implementation
132 return value_container::get_value(std::string(target_name), index);
133 }
134
135 if (index < it->second.size()) {
136 return it->second[index];
137 }
138
139 return nullptr;
140 }

222 [[nodiscard]] size_t estimate_memory_usage() const {
223 size_t total = 0;
224
225 // Header strings
226 total += source_id_.capacity();
227 total += source_sub_id_.capacity();
228 total += target_id_.capacity();
229 total += target_sub_id_.capacity();
230 total += message_type_.capacity();
231 total += version_.capacity();
232
233 // Data string
234 total += data_string_.capacity();
235
236 // Values (rough estimate)
237 total += units_.size() * sizeof(std::shared_ptr<value>);
238 total += units_.size() * 128; // Average value size
239
240 // Index
241 total += value_index_.size() * 64; // HashMap overhead
242
243 return total;
244 }

250 void build_value_index() {

253
254 for (const auto& unit : units_) {
255 if (unit) {
256 auto name = unit->name();
257 value_index_[name].push_back(unit);
258 }
259 }
260 }
