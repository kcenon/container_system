139 [[nodiscard]] std::shared_ptr<value> get_value_fast(
140 std::string_view target_name,
141 unsigned int index = 0
142 ) {
143 // Build index on first access
144 if (value_index_.empty() && !units_.empty()) {
145 build_value_index();
146 }
147
148 auto it = value_index_.find(std::string(target_name));
149 if (it == value_index_.end() || it->second.empty()) {
150 // Fallback to parent implementation
151 return value_container::get_value(std::string(target_name), index);
152 }
153
154 if (index < it->second.size()) {

156 }
157
158 return nullptr;
159 }

241 [[nodiscard]] size_t estimate_memory_usage() const {
242 size_t total = 0;
243
244 // Header strings
245 total += source_id_.capacity();
246 total += source_sub_id_.capacity();
247 total += target_id_.capacity();
248 total += target_sub_id_.capacity();
249 total += message_type_.capacity();
250 total += version_.capacity();
251
252 // Data string
253 total += data_string_.capacity();
254
255 // Values (rough estimate)
256 total += units_.size() * sizeof(std::shared_ptr<value>);
257 total += units_.size() * 128; // Average value size
258
259 // Index
260 total += value_index_.size() * 64; // HashMap overhead
261
262 return total;
