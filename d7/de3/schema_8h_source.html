122 constexpr std::string_view get_message(int code) noexcept
123 {
124 switch (code)
125 {
126 case missing_required: return "Required field is missing";
127 case type_mismatch: return "Field type mismatch";
128 case out_of_range: return "Value is out of range";
129 case invalid_length: return "Invalid length";
130 case pattern_mismatch: return "Pattern mismatch";
131 case not_in_allowed_values: return "Value not in allowed list";
132 case custom_validation_failed: return "Custom validation failed";
133 case nested_validation_failed: return "Nested schema validation failed";
134 default: return error_codes::get_message(code);
135 }

142 struct validation_error
143 {
144 std::string field;
145 std::string message;
146 int code;
147
148 validation_error() : code(0) {}
149
150 validation_error(std::string_view f, std::string_view msg, int c)
151 : field(f), message(msg), code(c) {}
152
156 static validation_error missing_required(std::string_view field_name)
157 {
158 return validation_error(
159 field_name,
160 std::string("Required field '") + std::string(field_name) + "' is missing",
161 validation_codes::missing_required);
162 }
163
167 static validation_error type_mismatch(std::string_view field_name,
168 value_types expected,
169 value_types actual)
170 {
171 return validation_error(
172 field_name,
173 std::string("Field '") + std::string(field_name) + "' type mismatch: expected " +
174 std::string(get_string_from_type(expected)) + ", got " +
175 std::string(get_string_from_type(actual)),
176 validation_codes::type_mismatch);
177 }
178
182 template<typename T>
183 static validation_error out_of_range(std::string_view field_name,
184 T value, T min_val, T max_val)
185 {
186 return validation_error(
187 field_name,
188 std::string("Field '") + std::string(field_name) + "' value " +
189 std::to_string(value) + " is out of range [" +
190 std::to_string(min_val) + ", " + std::to_string(max_val) + "]",
191 validation_codes::out_of_range);
192 }
193
197 static validation_error invalid_length(std::string_view field_name,
198 size_t actual, size_t min_len, size_t max_len)
199 {
200 return validation_error(
201 field_name,
202 std::string("Field '") + std::string(field_name) + "' length " +
203 std::to_string(actual) + " is outside bounds [" +
204 std::to_string(min_len) + ", " + std::to_string(max_len) + "]",
205 validation_codes::invalid_length);
206 }
207
211 static validation_error pattern_mismatch(std::string_view field_name,
212 std::string_view pattern)
213 {
214 return validation_error(
215 field_name,
216 std::string("Field '") + std::string(field_name) +
217 "' does not match pattern '" + std::string(pattern) + "'",
218 validation_codes::pattern_mismatch);
219 }
220
224 static validation_error not_allowed(std::string_view field_name,
225 std::string_view value)
226 {
227 return validation_error(
228 field_name,
229 std::string("Field '") + std::string(field_name) + "' value '" +
230 std::string(value) + "' is not in allowed values",
231 validation_codes::not_in_allowed_values);
232 }
233
237 static validation_error custom_failed(std::string_view field_name,
238 std::string_view reason)
239 {
240 return validation_error(
241 field_name,
242 std::string("Field '") + std::string(field_name) +
243 "' custom validation failed: " + std::string(reason),
244 validation_codes::custom_validation_failed);
245 }
246
250 static validation_error nested_failed(std::string_view field_name,
251 const std::vector<validation_error>& nested_errors)
252 {
253 std::string msg = std::string("Field '") + std::string(field_name) +
254 "' nested validation failed with " +
255 std::to_string(nested_errors.size()) + " error(s)";
256 return validation_error(field_name, msg, validation_codes::nested_validation_failed);
257 }
