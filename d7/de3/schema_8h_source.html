137 constexpr std::string_view get_message(int code) noexcept
138 {
139 switch (code)
140 {
141 case missing_required: return "Required field is missing";
142 case type_mismatch: return "Field type mismatch";
143 case out_of_range: return "Value is out of range";
144 case invalid_length: return "Invalid length";
145 case pattern_mismatch: return "Pattern mismatch";
146 case not_in_allowed_values: return "Value not in allowed list";
147 case custom_validation_failed: return "Custom validation failed";
148 case nested_validation_failed: return "Nested schema validation failed";
149 default: return error_codes::get_message(code);
150 }
151 }

157 struct validation_error
158 {
159 std::string field;
160 std::string message;
161 int code;
162
163 validation_error() : code(0) {}
164
165 validation_error(std::string_view f, std::string_view msg, int c)
166 : field(f), message(msg), code(c) {}
167
171 static validation_error missing_required(std::string_view field_name)
172 {
173 return validation_error(
174 field_name,
175 std::string("Required field '") + std::string(field_name) + "' is missing",
176 validation_codes::missing_required);
177 }
178
182 static validation_error type_mismatch(std::string_view field_name,
183 value_types expected,
184 value_types actual)
185 {
186 return validation_error(
187 field_name,
188 std::string("Field '") + std::string(field_name) + "' type mismatch: expected " +
189 std::string(get_string_from_type(expected)) + ", got " +
190 std::string(get_string_from_type(actual)),
191 validation_codes::type_mismatch);
192 }
193
197 template<typename T>
198 static validation_error out_of_range(std::string_view field_name,
199 T value, T min_val, T max_val)
200 {
201 return validation_error(
202 field_name,
203 std::string("Field '") + std::string(field_name) + "' value " +
204 std::to_string(value) + " is out of range [" +
205 std::to_string(min_val) + ", " + std::to_string(max_val) + "]",
206 validation_codes::out_of_range);
207 }
208
212 static validation_error invalid_length(std::string_view field_name,
213 size_t actual, size_t min_len, size_t max_len)
214 {
215 return validation_error(
216 field_name,
217 std::string("Field '") + std::string(field_name) + "' length " +
218 std::to_string(actual) + " is outside bounds [" +
219 std::to_string(min_len) + ", " + std::to_string(max_len) + "]",
220 validation_codes::invalid_length);
221 }
222
226 static validation_error pattern_mismatch(std::string_view field_name,
227 std::string_view pattern)
228 {
229 return validation_error(
230 field_name,
231 std::string("Field '") + std::string(field_name) +
232 "' does not match pattern '" + std::string(pattern) + "'",
233 validation_codes::pattern_mismatch);
234 }
235
239 static validation_error not_allowed(std::string_view field_name,
240 std::string_view value)
241 {
242 return validation_error(
243 field_name,
244 std::string("Field '") + std::string(field_name) + "' value '" +
245 std::string(value) + "' is not in allowed values",
246 validation_codes::not_in_allowed_values);
247 }
248
252 static validation_error custom_failed(std::string_view field_name,
253 std::string_view reason)
254 {
255 return validation_error(
256 field_name,
257 std::string("Field '") + std::string(field_name) +
258 "' custom validation failed: " + std::string(reason),
259 validation_codes::custom_validation_failed);
260 }
261
265 static validation_error nested_failed(std::string_view field_name,
266 const std::vector<validation_error>& nested_errors)
267 {
268 std::string msg = std::string("Field '") + std::string(field_name) +
269 "' nested validation failed with " +
270 std::to_string(nested_errors.size()) + " error(s)";
271 return validation_error(field_name, msg, validation_codes::nested_validation_failed);
272 }
