159 void write_nil()
160 {
161 buffer_.push_back(msgpack_format::NIL);
162 }
163
167 void write_bool(bool value)
168 {
169 buffer_.push_back(value ? msgpack_format::TRUE : msgpack_format::FALSE);
170 }
171
175 void write_int(int64_t value)
176 {
177 if (value >= 0)
178 {
179 write_uint(static_cast<uint64_t>(value));
180 }
181 else if (value >= -32)
182 {
183 // Negative fixint
184 buffer_.push_back(static_cast<uint8_t>(value));
185 }
186 else if (value >= INT8_MIN)
187 {
188 buffer_.push_back(msgpack_format::INT8);
189 buffer_.push_back(static_cast<uint8_t>(value));
190 }
191 else if (value >= INT16_MIN)
192 {
193 buffer_.push_back(msgpack_format::INT16);
194 write_be16(static_cast<uint16_t>(value));
195 }
196 else if (value >= INT32_MIN)
197 {
198 buffer_.push_back(msgpack_format::INT32);
199 write_be32(static_cast<uint32_t>(value));
200 }
201 else
202 {
203 buffer_.push_back(msgpack_format::INT64);
204 write_be64(static_cast<uint64_t>(value));
205 }
206 }
207
211 void write_uint(uint64_t value)
212 {
213 if (value <= msgpack_format::POSITIVE_FIXINT_MAX)
214 {
215 // Positive fixint
216 buffer_.push_back(static_cast<uint8_t>(value));
217 }
218 else if (value <= UINT8_MAX)
219 {
220 buffer_.push_back(msgpack_format::UINT8);
221 buffer_.push_back(static_cast<uint8_t>(value));
222 }
223 else if (value <= UINT16_MAX)
224 {
225 buffer_.push_back(msgpack_format::UINT16);
226 write_be16(static_cast<uint16_t>(value));
227 }
228 else if (value <= UINT32_MAX)
229 {
230 buffer_.push_back(msgpack_format::UINT32);
231 write_be32(static_cast<uint32_t>(value));
232 }
233 else
234 {
235 buffer_.push_back(msgpack_format::UINT64);
236 write_be64(value);
237 }
238 }
239
243 void write_float(float value)
244 {
245 buffer_.push_back(msgpack_format::FLOAT32);
246 uint32_t bits = 0;
247 std::memcpy(&bits, &value, sizeof(bits));
248 write_be32(bits);
249 }
250
254 void write_double(double value)
255 {
256 buffer_.push_back(msgpack_format::FLOAT64);
257 uint64_t bits = 0;
258 std::memcpy(&bits, &value, sizeof(bits));
259 write_be64(bits);
260 }
261
265 void write_string(std::string_view value)
266 {
267 size_t len = value.size();
268 if (len <= msgpack_format::FIXSTR_MAX_SIZE)
269 {
270 buffer_.push_back(static_cast<uint8_t>(msgpack_format::FIXSTR_PREFIX | len));
271 }
272 else if (len <= UINT8_MAX)
273 {
274 buffer_.push_back(msgpack_format::STR8);
275 buffer_.push_back(static_cast<uint8_t>(len));
276 }
277 else if (len <= UINT16_MAX)
278 {
279 buffer_.push_back(msgpack_format::STR16);
280 write_be16(static_cast<uint16_t>(len));
281 }
282 else
283 {
284 buffer_.push_back(msgpack_format::STR32);
285 write_be32(static_cast<uint32_t>(len));
286 }
287 buffer_.insert(buffer_.end(), value.begin(), value.end());
288 }

293 void write_binary(const std::vector<uint8_t>& value)
294 {

298 buffer_.push_back(msgpack_format::BIN8);
299 buffer_.push_back(static_cast<uint8_t>(len));
300 }
301 else if (len <= UINT16_MAX)
302 {
303 buffer_.push_back(msgpack_format::BIN16);
304 write_be16(static_cast<uint16_t>(len));
305 }
306 else
307 {
308 buffer_.push_back(msgpack_format::BIN32);
309 write_be32(static_cast<uint32_t>(len));
310 }
311 buffer_.insert(buffer_.end(), value.begin(), value.end());
312 }

374 private:
375 void write_be16(uint16_t value)
376 {
377 buffer_.push_back(static_cast<uint8_t>(value >> 8));
378 buffer_.push_back(static_cast<uint8_t>(value));
379 }
380
381 void write_be32(uint32_t value)
382 {
383 buffer_.push_back(static_cast<uint8_t>(value >> 24));
384 buffer_.push_back(static_cast<uint8_t>(value >> 16));
385 buffer_.push_back(static_cast<uint8_t>(value >> 8));
386 buffer_.push_back(static_cast<uint8_t>(value));
387 }
388
389 void write_be64(uint64_t value)
390 {
391 buffer_.push_back(static_cast<uint8_t>(value >> 56));
392 buffer_.push_back(static_cast<uint8_t>(value >> 48));
393 buffer_.push_back(static_cast<uint8_t>(value >> 40));
394 buffer_.push_back(static_cast<uint8_t>(value >> 32));
395 buffer_.push_back(static_cast<uint8_t>(value >> 24));
396 buffer_.push_back(static_cast<uint8_t>(value >> 16));
397 buffer_.push_back(static_cast<uint8_t>(value >> 8));
398 buffer_.push_back(static_cast<uint8_t>(value));
399 }

423 msgpack_type peek_type() const
424 {
425 if (offset_ >= size_)
426 {
427 return msgpack_type::unknown;
428 }
429
430 uint8_t byte = data_[offset_];
431
432 // Positive fixint
433 if (byte <= msgpack_format::POSITIVE_FIXINT_MAX)
434 {
435 return msgpack_type::positive_int;
436 }
437
438 // Fixmap
439 if ((byte & 0xf0) == msgpack_format::FIXMAP_PREFIX)
440 {
441 return msgpack_type::map;
442 }
443
444 // Fixarray
445 if ((byte & 0xf0) == msgpack_format::FIXARRAY_PREFIX)
446 {
447 return msgpack_type::array;
448 }
449
450 // Fixstr
451 if ((byte & 0xe0) == msgpack_format::FIXSTR_PREFIX)
452 {
453 return msgpack_type::str;
454 }
455
456 // Negative fixint
457 if (byte >= msgpack_format::NEGATIVE_FIXINT_PREFIX)
458 {
459 return msgpack_type::negative_int;
460 }
461
462 switch (byte)
463 {
464 case msgpack_format::NIL:
465 return msgpack_type::nil;
466 case msgpack_format::FALSE:
467 case msgpack_format::TRUE:
468 return msgpack_type::boolean;
469 case msgpack_format::BIN8:
470 case msgpack_format::BIN16:
471 case msgpack_format::BIN32:
472 return msgpack_type::bin;
473 case msgpack_format::FLOAT32:
474 return msgpack_type::float32;
475 case msgpack_format::FLOAT64:
476 return msgpack_type::float64;
477 case msgpack_format::UINT8:
478 case msgpack_format::UINT16:
479 case msgpack_format::UINT32:
480 case msgpack_format::UINT64:
481 return msgpack_type::positive_int;
482 case msgpack_format::INT8:
483 case msgpack_format::INT16:
484 case msgpack_format::INT32:
485 case msgpack_format::INT64:
486 return msgpack_type::negative_int;
487 case msgpack_format::STR8:
488 case msgpack_format::STR16:
489 case msgpack_format::STR32:
490 return msgpack_type::str;
491 case msgpack_format::ARRAY16:
492 case msgpack_format::ARRAY32:
493 return msgpack_type::array;
494 case msgpack_format::MAP16:
495 case msgpack_format::MAP32:
496 return msgpack_type::map;
497 default:
498 return msgpack_type::unknown;
499 }

543 std::optional<int64_t> read_int()
544 {
545 if (offset_ >= size_)
546 {
547 return std::nullopt;
548 }
549
550 uint8_t byte = data_[offset_];
551
552 // Positive fixint
553 if (byte <= msgpack_format::POSITIVE_FIXINT_MAX)
554 {
555 ++offset_;
556 return static_cast<int64_t>(byte);
557 }
558
559 // Negative fixint
560 if (byte >= msgpack_format::NEGATIVE_FIXINT_PREFIX)
561 {
562 ++offset_;
563 return static_cast<int64_t>(static_cast<int8_t>(byte));
564 }
565
566 switch (byte)
567 {
568 case msgpack_format::INT8:
569 if (offset_ + 2 > size_)
570 {
571 return std::nullopt;
572 }
573 ++offset_;
574 return static_cast<int64_t>(static_cast<int8_t>(data_[offset_++]));
575
576 case msgpack_format::INT16:
577 if (offset_ + 3 > size_)
578 {
579 return std::nullopt;
580 }
581 ++offset_;
582 return static_cast<int64_t>(static_cast<int16_t>(read_be16()));
583
584 case msgpack_format::INT32:
585 if (offset_ + 5 > size_)
586 {
587 return std::nullopt;
588 }
589 ++offset_;
590 return static_cast<int64_t>(static_cast<int32_t>(read_be32()));
591
592 case msgpack_format::INT64:
593 if (offset_ + 9 > size_)
594 {
595 return std::nullopt;
596 }
597 ++offset_;
598 return static_cast<int64_t>(read_be64());
599
600 case msgpack_format::UINT8:
601 if (offset_ + 2 > size_)
602 {
603 return std::nullopt;
604 }
605 ++offset_;
606 return static_cast<int64_t>(data_[offset_++]);
607
608 case msgpack_format::UINT16:
609 if (offset_ + 3 > size_)
610 {
611 return std::nullopt;
612 }
613 ++offset_;
614 return static_cast<int64_t>(read_be16());
615
616 case msgpack_format::UINT32:
617 if (offset_ + 5 > size_)
618 {
619 return std::nullopt;
620 }
621 ++offset_;
622 return static_cast<int64_t>(read_be32());
623
624 case msgpack_format::UINT64:
625 if (offset_ + 9 > size_)
626 {
627 return std::nullopt;
628 }
629 ++offset_;
630 {
631 uint64_t val = read_be64();
632 if (val > static_cast<uint64_t>(INT64_MAX))
633 {
634 return std::nullopt; // Overflow
635 }
636 return static_cast<int64_t>(val);
637 }
638
639 default:
640 return std::nullopt;
641 }
642 }

763 std::optional<std::string> read_string()
764 {
765 if (offset_ >= size_)
766 {
767 return std::nullopt;
768 }
769
770 uint8_t byte = data_[offset_];
771 size_t len = 0;
772
773 // Fixstr
774 if ((byte & 0xe0) == msgpack_format::FIXSTR_PREFIX)
775 {
776 len = byte & 0x1f;
777 ++offset_;
778 }
779 else
780 {
781 switch (byte)
782 {
783 case msgpack_format::STR8:
784 if (offset_ + 2 > size_) return std::nullopt;
785 ++offset_;
786 len = data_[offset_++];
787 break;
788
789 case msgpack_format::STR16:
790 if (offset_ + 3 > size_) return std::nullopt;
791 ++offset_;
792 len = read_be16();
793 break;
794
795 case msgpack_format::STR32:
796 if (offset_ + 5 > size_) return std::nullopt;
797 ++offset_;
798 len = read_be32();
799 break;
800
801 default:
802 return std::nullopt;
803 }
804 }
805
806 if (offset_ + len > size_)
807 {
808 return std::nullopt;
809 }
810
811 std::string result(reinterpret_cast<const char*>(data_ + offset_), len);
812 offset_ += len;
813 return result;

950 private:
951 uint16_t read_be16()
952 {
953 uint16_t result = (static_cast<uint16_t>(data_[offset_]) << 8)
954 | static_cast<uint16_t>(data_[offset_ + 1]);
955 offset_ += 2;
956 return result;
957 }
958
959 uint32_t read_be32()
960 {
961 uint32_t result = (static_cast<uint32_t>(data_[offset_]) << 24)
962 | (static_cast<uint32_t>(data_[offset_ + 1]) << 16)
963 | (static_cast<uint32_t>(data_[offset_ + 2]) << 8)
964 | static_cast<uint32_t>(data_[offset_ + 3]);
965 offset_ += 4;
966 return result;
967 }
968
969 uint64_t read_be64()
