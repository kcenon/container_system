88template<typename P>
89concept StoragePolicy = requires(P p, const P cp, std::string_view key) {
90 // Type requirements
91 typename P::value_type;
92 typename P::iterator;
93 typename P::const_iterator;
94
95 // CRUD operations
96 { p.set(key, std::declval<const typename P::value_type&>()) } -> std::same_as<void>;
97 { cp.get(key) } -> std::same_as<std::optional<typename P::value_type>>;
98 { cp.contains(key) } -> std::convertible_to<bool>;
99 { p.remove(key) } -> std::convertible_to<bool>;
100 { p.clear() } -> std::same_as<void>;
101
102 // Iteration support
103 { p.begin() } -> std::same_as<typename P::iterator>;
104 { p.end() } -> std::same_as<typename P::iterator>;
105 { cp.begin() } -> std::same_as<typename P::const_iterator>;
106 { cp.end() } -> std::same_as<typename P::const_iterator>;
107 { cp.size() } -> std::convertible_to<std::size_t>;
108 { cp.empty() } -> std::convertible_to<bool>;
109};

335 void set(std::string_view key, const value_type& value) {
336 std::string key_str(key);
337 auto it = index_.find(key_str);
338 if (it != index_.end()) {
339 values_[it->second] = value;
340 } else {
341 index_[key_str] = values_.size();
342 values_.push_back(value);
343 }

351 void set(std::string_view key, value_type&& value) {
352 std::string key_str(key);
353 auto it = index_.find(key_str);
354 if (it != index_.end()) {
355 values_[it->second] = std::move(value);
356 } else {
357 index_[key_str] = values_.size();
358 values_.push_back(std::move(value));
359 }
360 }

367 [[nodiscard]] std::optional<value_type> get(std::string_view key) const {
368 auto it = index_.find(std::string(key));
369 if (it != index_.end()) {
370 return values_[it->second];
371 }
372 return std::nullopt;
373 }
374
380 [[nodiscard]] bool contains(std::string_view key) const {
381 return index_.find(std::string(key)) != index_.end();
382 }
383
391 bool remove(std::string_view key) {
392 std::string key_str(key);
393 auto it = index_.find(key_str);
394 if (it == index_.end()) {
395 return false;
396 }
397
398 std::size_t idx = it->second;
399 values_.erase(values_.begin() + static_cast<std::ptrdiff_t>(idx));
400 index_.erase(it);
401
402 // Rebuild index for elements after the removed one
403 for (auto& [k, v] : index_) {
404 if (v > idx) {
405 --v;
406 }
407 }
408 return true;
409 }
410
414 void clear() {
415 values_.clear();
416 index_.clear();
417 }
418
419 iterator begin() { return values_.begin(); }
420 iterator end() { return values_.end(); }
421 [[nodiscard]] const_iterator begin() const { return values_.begin(); }
422 [[nodiscard]] const_iterator end() const { return values_.end(); }
423 [[nodiscard]] std::size_t size() const { return values_.size(); }
424 [[nodiscard]] bool empty() const { return values_.empty(); }
425
430 void reserve(std::size_t capacity) {
431 values_.reserve(capacity);
432 index_.reserve(capacity);
433 }
434
440 void rebuild_index() {
441 index_.clear();
442 index_.reserve(values_.size());
443 for (std::size_t i = 0; i < values_.size(); ++i) {
444 index_[values_[i].name] = i;
445 }
446 }
447
448 container_type& data() { return values_; }
449 [[nodiscard]] const container_type& data() const { return values_; }

451private:
452 container_type values_;
453 index_type index_;

504 template<typename T>

506 (std::is_same_v<std::decay_t<T>, AllowedTypes> || ...);

516 template<typename T>
517 void set_typed(std::string_view key, T&& typed_value) {
518 static_assert(is_allowed_type_v<T>,
519 "Type not in allowed types list for static_storage_policy");
520
521 value_type val;
522 val.name = std::string(key);
523 val.data = std::forward<T>(typed_value);
524 val.type = static_cast<value_types>(val.data.index());
525
526 auto it = find_by_key(key);
527 if (it != values_.end()) {
528 *it = std::move(val);
529 } else {
530 values_.push_back(std::move(val));
531 }
532 }
