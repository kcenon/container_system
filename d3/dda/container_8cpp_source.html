398 void value_container::set_unit_impl(const optimized_value& val)
399 {
400 // Record metrics if enabled
401 auto timer = metrics_manager::make_timer(
402 metrics_manager::get().write_latency,
403 &metrics_manager::get().timing.total_write_ns);
404 if (metrics_manager::is_enabled())
405 {
406 metrics_manager::get().operations.writes.fetch_add(1, std::memory_order_relaxed);
407 }
408
409 write_lock_guard lock(this);
410
411 // Check if key already exists
412 for (auto& existing : optimized_units_) {
413 if (existing.name == val.name) {
414 existing = val;
415 changed_data_ = true;
416 return;
417 }
418 }
419
420 // Key doesn't exist, add new value
421 optimized_units_.push_back(val);
422 changed_data_ = true;
423
424 if (use_soo_ && val.is_stack_allocated()) {
425 stack_allocations_.fetch_add(1, std::memory_order_relaxed);
426 } else {
427 heap_allocations_.fetch_add(1, std::memory_order_relaxed);
428 }
429 }

582 size_t value_container::remove_batch(std::span<const std::string_view> keys)
583 {
584 if (keys.empty()) {
585 return 0;
586 }
587
588 write_lock_guard lock(this);
589
590 if (!parsed_data_) {
591 deserialize_values(data_string_, false);
592 }
593
594 // Build a set of keys for O(1) lookup
595 std::unordered_set<std::string_view> key_set(keys.begin(), keys.end());
596
597 // Count and remove matching elements
598 size_t original_size = optimized_units_.size();
599 optimized_units_.erase(
600 std::remove_if(optimized_units_.begin(), optimized_units_.end(),
601 [&key_set](const optimized_value& ov) {
602 return key_set.find(ov.name) != key_set.end();
603 }),
604 optimized_units_.end());
605
606 size_t removed = original_size - optimized_units_.size();
607 if (removed > 0) {
608 changed_data_ = true;
609 }
610
611 return removed;
612 }

683 std::optional<value_view> value_container::get_view(std::string_view key) const noexcept
684 {
685 read_lock_guard lock(this);
686
687 // Zero-copy mode requires raw_data_ptr_ to be valid
688 if (!zero_copy_mode_ || !raw_data_ptr_) {
689 return std::nullopt;
690 }
691
692 // Build index if not already built
693 if (!index_built_) {
694 build_index();
695 }
696
697 // Search in the index
698 if (value_index_) {
699 for (const auto& entry : *value_index_) {
700 if (entry.name == key) {
701 const char* data_ptr = raw_data_ptr_->data();
702 return value_view(
703 entry.name.data(), entry.name.size(),
704 data_ptr + entry.value_offset, entry.value_length,
705 entry.type
706 );
707 }
708 }
709 }
710
711 return std::nullopt;

721
722 void value_container::build_index() const
723 {
724 if (index_built_ || !raw_data_ptr_) {
725 return;
726 }
727
728 value_index_ = std::vector<value_index_entry>();
729 const std::string& data = *raw_data_ptr_;
730
731 // Find the @data section
732 // Match both single and double braces for @data section
733 std::regex reData("@data=\\s*\\{\\{?\\s*(.*?)\\s*\\}\\}?;");
734 std::smatch dataMatch;
735 if (!std::regex_search(data, dataMatch, reData)) {
736 index_built_ = true;
737 return;
738 }
739
740 // Get the position of the data section in the original string
741 size_t dataStart = static_cast<size_t>(dataMatch.position(1));
742 std::string dataInside = dataMatch[1].str();
743
744 // Parse items: [name,type,data];
745 std::regex reItems("\\[(\\w+),\\s*(\\w+),\\s*(.*?)\\];");
746 auto it = std::sregex_iterator(dataInside.begin(), dataInside.end(), reItems);
747 auto end = std::sregex_iterator();
748
749 for (; it != end; ++it) {
750 value_index_entry entry;
751
752 // Calculate offsets into the original raw data
753 size_t matchPos = static_cast<size_t>(it->position());
754 size_t nameStart = dataStart + matchPos + 1; // +1 for '['
755 size_t nameLen = (*it)[1].length();
756
757 // Store name as string_view into raw data
758 entry.name = std::string_view(data.data() + nameStart, nameLen);
759
760 // Parse type
761 std::string typeStr = (*it)[2].str();
762 entry.type = convert_value_type(typeStr);
763
764 // Calculate value offset and length
765 // Value starts after [name,type,
766 size_t valueStart = dataStart + matchPos + 1 + nameLen + 1 + (*it)[2].length() + 1;
767 // +1 for '[', +1 for first ',', +1 for second ','
768 entry.value_offset = valueStart;
769 entry.value_length = (*it)[3].length();
770
771 value_index_->push_back(entry);
772 }
773
774 index_built_ = true;
775 }

804 metrics_manager::get().operations.deserializations.fetch_add(1, std::memory_order_relaxed);
805 }
806
807 initialize();
808 if (data_str.empty())
809 return false;
810
811 // Remove newlines
812 std::regex newlineRe("\\r\\n?|\\n");
813 std::string clean = std::regex_replace(data_str, newlineRe, "");
814
815 // parse header portion - matches both single and double braces
816 std::regex fullRe("@header=\\s*\\{\\{?\\s*(.*?)\\s*\\}\\}?;");
817 std::smatch match;
818 if (!std::regex_search(clean, match, fullRe))
819 {
820 // No header => parse as data only
821 return deserialize_values(clean, parse_only_header);
822 }
823 // match[1] => inside the header
824 std::string headerInside = match[1].str();
825 std::regex pairRe("\\[(\\w+),(.*?)\\];");
826 auto it = std::sregex_iterator(headerInside.begin(), headerInside.end(),
827 pairRe);
828 auto end = std::sregex_iterator();
829 for (; it != end; ++it)
830 {
831 parsing((*it)[1].str(), std::to_string(TARGET_ID), (*it)[2].str(), target_id_);
832 parsing((*it)[1].str(), std::to_string(TARGET_SUB_ID), (*it)[2].str(), target_sub_id_);
833 parsing((*it)[1].str(), std::to_string(SOURCE_ID), (*it)[2].str(), source_id_);
834 parsing((*it)[1].str(), std::to_string(SOURCE_SUB_ID), (*it)[2].str(), source_sub_id_);
835 parsing((*it)[1].str(), std::to_string(MESSAGE_TYPE), (*it)[2].str(), message_type_);
836 parsing((*it)[1].str(), std::to_string(MESSAGE_VERSION), (*it)[2].str(), version_);
837 }
838
839 return deserialize_values(clean, parse_only_header);
840 }

1286 bool value_container::from_msgpack_impl(const std::vector<uint8_t>& data)
1287 {
1288 // Record metrics if enabled
1289 auto timer = metrics_manager::make_timer(
1290 metrics_manager::get().deserialize_latency,
1291 &metrics_manager::get().timing.total_deserialize_ns);
1292 if (metrics_manager::is_enabled())
1293 {
1294 metrics_manager::get().operations.deserializations.fetch_add(1, std::memory_order_relaxed);
1295 }
1296
1297 if (data.empty())
1298 {
1299 return false;
1300 }
1301
1302 initialize();
1303
1304 msgpack_decoder decoder(data);
1305
1306 // Expect outer map
1307 auto outer_count = decoder.read_map_header();
1308 if (!outer_count)
1309 {
1310 return false;
1311 }
1312
1313 std::unique_lock<std::shared_mutex> lock(mutex_);
1314
1315 for (size_t i = 0; i < *outer_count; ++i)
1316 {
1317 auto key = decoder.read_string();
1318 if (!key)
1319 {
1320 return false;
1321 }
1322
1323 if (*key == "header")
1324 {
1325 auto header_count = decoder.read_map_header();
1326 if (!header_count)
1327 {
1328 return false;
1329 }
1330
1331 for (size_t j = 0; j < *header_count; ++j)
1332 {
1333 auto hkey = decoder.read_string();
1334 auto hval = decoder.read_string();
1335 if (!hkey || !hval)
1336 {
1337 return false;
1338 }
1339
1340 if (*hkey == "target_id")
1341 {
1342 target_id_ = *hval;
1343 }
1344 else if (*hkey == "target_sub_id")
1345 {
1346 target_sub_id_ = *hval;
1347 }
1348 else if (*hkey == "source_id")
1349 {
1350 source_id_ = *hval;
1351 }
1352 else if (*hkey == "source_sub_id")
1353 {
1354 source_sub_id_ = *hval;
1355 }
1356 else if (*hkey == "message_type")
1357 {
1358 message_type_ = *hval;
1359 }
1360 else if (*hkey == "version")
1361 {
1362 version_ = *hval;
1363 }
1364 }
1365 }
1366 else if (*key == "values")
1367 {
1368 auto values_count = decoder.read_map_header();
1369 if (!values_count)
1370 {
1371 return false;
1372 }
1373
1374 for (size_t j = 0; j < *values_count; ++j)
1375 {
1376 auto vkey = decoder.read_string();
1377 if (!vkey)
1378 {
1379 return false;
1380 }
1381
1382 optimized_value val;
1383 val.name = *vkey;
1384
1385 // Read value based on type
1386 auto type = decoder.peek_type();
1387 switch (type)
1388 {
1389 case msgpack_type::nil:
1390 decoder.read_nil();
1391 val.type = value_types::null_value;
1392 val.data = std::monostate{};
1393 break;
1394
1395 case msgpack_type::boolean:
1396 {
1397 auto b = decoder.read_bool();
1398 if (!b)
1399 {
1400 return false;
1401 }
1402 val.type = value_types::bool_value;
1403 val.data = *b;
1404 }
1405 break;
1406
1407 case msgpack_type::positive_int:
1408 case msgpack_type::negative_int:
1409 {
1410 auto n = decoder.read_int();
1411 if (!n)
1412 {
1413 return false;
1414 }
1415 // Determine best type based on value
1416 if (*n >= 0 && *n <= INT32_MAX)
1417 {
1418 val.type = value_types::int_value;
1419 val.data = static_cast<int>(*n);
1420 }
1421 else if (*n >= INT32_MIN && *n < 0)
1422 {
1423 val.type = value_types::int_value;
1424 val.data = static_cast<int>(*n);
1425 }
1426 else
1427 {
1428 val.type = value_types::llong_value;
1429 val.data = static_cast<long long>(*n);
1430 }
1431 }
1432 break;
1433
1434 case msgpack_type::float32:
1435 {
1436 auto f = decoder.read_float();
1437 if (!f)
1438 {
1439 return false;
1440 }
1441 val.type = value_types::float_value;
1442 val.data = *f;
1443 }
1444 break;
1445
1446 case msgpack_type::float64:
1447 {
1448 auto d = decoder.read_double();
1449 if (!d)
1450 {
1451 return false;
1452 }
1453 val.type = value_types::double_value;
1454 val.data = *d;
1455 }
1456 break;
1457
1458 case msgpack_type::str:
1459 {
1460 auto s = decoder.read_string();
1461 if (!s)
1462 {
1463 return false;
1464 }
1465 val.type = value_types::string_value;
1466 val.data = *s;
1467 }
1468 break;
1469
1470 case msgpack_type::bin:
1471 {
1472 auto b = decoder.read_binary();
1473 if (!b)
1474 {
1475 return false;
1476 }
1477 val.type = value_types::bytes_value;
1478 val.data = *b;
1479 }
1480 break;
1481
1482 default:
1483 // Skip unknown types
1484 val.type = value_types::null_value;
1485 val.data = std::monostate{};
1486 break;
1487 }
1488
1489 optimized_units_.push_back(std::move(val));
1490 }
1491 }
1492 }
1493
1494 parsed_data_ = true;
1495 changed_data_ = false;
1496
1497 return true;
1498 }

1500 value_container::serialization_format value_container::detect_format(
1501 const std::vector<uint8_t>& data)
1502 {
1503 if (data.empty())
1504 {
1505 return serialization_format::unknown;
1506 }
1507
1508 // Check for MessagePack format
1509 // MessagePack maps start with 0x80-0x8f (fixmap) or 0xde (map16) or 0xdf (map32)
1510 uint8_t first_byte = data[0];
1511 if ((first_byte >= 0x80 && first_byte <= 0x8f) || // fixmap
1512 first_byte == 0xde || // map16
1513 first_byte == 0xdf) // map32
1514 {
1515 return serialization_format::msgpack;
1516 }
1517
1518 // Check for text-based formats (convert to string for analysis)
1519 std::string_view str_view(reinterpret_cast<const char*>(data.data()), data.size());
1520 return detect_format(str_view);
1521 }
1522
1523 value_container::serialization_format value_container::detect_format(
1524 std::string_view data)
1525 {
1526 if (data.empty())
1527 {
1528 return serialization_format::unknown;
1529 }
1530
1531 // Skip leading whitespace
1532 size_t pos = 0;
1533 while (pos < data.size() && std::isspace(static_cast<unsigned char>(data[pos])))
1534 {
1535 ++pos;
1536 }
1537
1538 if (pos >= data.size())
1539 {
1540 return serialization_format::unknown;
1541 }
1542
1543 // Check first non-whitespace character
1544 char first_char = data[pos];
1545
1546 // JSON starts with '{' or '['
1547 if (first_char == '{' || first_char == '[')
1548 {
1549 return serialization_format::json;
1550 }
1551
1552 // XML starts with '<'
1553 if (first_char == '<')
1554 {
1555 return serialization_format::xml;
1556 }
1557
1558 // Binary format starts with "@header"
1559 if (data.substr(pos, 7) == "@header")
1560 {
1561 return serialization_format::binary;
1562 }
1563
1564 // Binary format may also start with "@data"
1565 if (data.substr(pos, 5) == "@data")
1566 {
1567 return serialization_format::binary;
1568 }
1569
1570 return serialization_format::unknown;
1571 }
