285 void value_container::set_unit_impl(const optimized_value& val)
286 {
287 // Record metrics if enabled
288 auto timer = metrics_manager::make_timer(
289 metrics_manager::get().write_latency,
290 &metrics_manager::get().timing.total_write_ns);
291 if (metrics_manager::is_enabled())
292 {
293 metrics_manager::get().operations.writes.fetch_add(1, std::memory_order_relaxed);
294 }
295
296 write_lock_guard lock(this);
297
298 // Check if key already exists
299 for (auto& existing : optimized_units_) {
300 if (existing.name == val.name) {
301 existing = val;
302 changed_data_ = true;
303 return;
304 }
305 }
306
307 // Key doesn't exist, add new value
308 optimized_units_.push_back(val);
309 changed_data_ = true;
310
311 if (use_soo_ && val.is_stack_allocated()) {
312 stack_allocations_.fetch_add(1, std::memory_order_relaxed);
313 } else {
314 heap_allocations_.fetch_add(1, std::memory_order_relaxed);
315 }
316 }

352 value_container& value_container::bulk_insert(std::vector<optimized_value>&& values)
353 {
354 if (values.empty()) {
355 return *this;
356 }
357
358 write_lock_guard lock(this);
359
360 // Pre-allocate for efficiency
361 optimized_units_.reserve(optimized_units_.size() + values.size());
362
363 // Move all values
364 for (auto& val : values) {
365 if (use_soo_ && val.is_stack_allocated()) {
366 stack_allocations_.fetch_add(1, std::memory_order_relaxed);
367 } else {
368 heap_allocations_.fetch_add(1, std::memory_order_relaxed);
369 }
370 optimized_units_.push_back(std::move(val));
371 }
372
373 changed_data_ = true;
374 return *this;
375 }

469 size_t value_container::remove_batch(std::span<const std::string_view> keys)
470 {
471 if (keys.empty()) {
472 return 0;
473 }
474
475 write_lock_guard lock(this);
476
477 if (!parsed_data_) {
478 deserialize_values(data_string_, false);
479 }
480
481 // Build a set of keys for O(1) lookup
482 std::unordered_set<std::string_view> key_set(keys.begin(), keys.end());
483
484 // Count and remove matching elements
485 size_t original_size = optimized_units_.size();
486 optimized_units_.erase(
487 std::remove_if(optimized_units_.begin(), optimized_units_.end(),
488 [&key_set](const optimized_value& ov) {
489 return key_set.find(ov.name) != key_set.end();
490 }),
491 optimized_units_.end());
492
493 size_t removed = original_size - optimized_units_.size();
494 if (removed > 0) {
495 changed_data_ = true;
496 }
497
498 return removed;
499 }

570 std::optional<value_view> value_container::get_view(std::string_view key) const noexcept
571 {
572 read_lock_guard lock(this);
573
574 // Zero-copy mode requires raw_data_ptr_ to be valid
575 if (!zero_copy_mode_ || !raw_data_ptr_) {
576 return std::nullopt;
577 }
578
579 // Build index if not already built
580 if (!index_built_) {
581 build_index();
582 }
583
584 // Search in the index
585 if (value_index_) {
586 for (const auto& entry : *value_index_) {
587 if (entry.name == key) {
588 const char* data_ptr = raw_data_ptr_->data();
589 return value_view(
590 entry.name.data(), entry.name.size(),
591 data_ptr + entry.value_offset, entry.value_length,
592 entry.type
593 );
594 }
595 }
596 }
597
598 return std::nullopt;
599 }

608
609 void value_container::build_index() const
610 {
611 if (index_built_ || !raw_data_ptr_) {
612 return;
613 }
614
615 value_index_ = std::vector<value_index_entry>();
616 const std::string& data = *raw_data_ptr_;
617
618 // Find the @data section
619 // Match both single and double braces for @data section
620 std::regex reData("@data=\\s*\\{\\{?\\s*(.*?)\\s*\\}\\}?;");
621 std::smatch dataMatch;
622 if (!std::regex_search(data, dataMatch, reData)) {
623 index_built_ = true;
624 return;
625 }
626
627 // Get the position of the data section in the original string
628 size_t dataStart = static_cast<size_t>(dataMatch.position(1));
629 std::string dataInside = dataMatch[1].str();
630
631 // Parse items: [name,type,data];
632 std::regex reItems("\\[(\\w+),\\s*(\\w+),\\s*(.*?)\\];");
633 auto it = std::sregex_iterator(dataInside.begin(), dataInside.end(), reItems);
634 auto end = std::sregex_iterator();
635
636 for (; it != end; ++it) {
637 value_index_entry entry;
638
639 // Calculate offsets into the original raw data
640 size_t matchPos = static_cast<size_t>(it->position());
641 size_t nameStart = dataStart + matchPos + 1; // +1 for '['
642 size_t nameLen = (*it)[1].length();
643
644 // Store name as string_view into raw data
645 entry.name = std::string_view(data.data() + nameStart, nameLen);
646
647 // Parse type
648 std::string typeStr = (*it)[2].str();
649 entry.type = convert_value_type(typeStr);
650
651 // Calculate value offset and length
652 // Value starts after [name,type,
653 size_t valueStart = dataStart + matchPos + 1 + nameLen + 1 + (*it)[2].length() + 1;
654 // +1 for '[', +1 for first ',', +1 for second ','
655 entry.value_offset = valueStart;
656 entry.value_length = (*it)[3].length();
657
658 value_index_->push_back(entry);
659 }
660
661 index_built_ = true;
662 }

682 std::string value_container::serialize_impl() const
683 {
684 // Record metrics if enabled
685 auto timer = metrics_manager::make_timer(
686 metrics_manager::get().serialize_latency,
687 &metrics_manager::get().timing.total_serialize_ns);
688 if (metrics_manager::is_enabled())
689 {
690 metrics_manager::get().operations.serializations.fetch_add(1, std::memory_order_relaxed);
691 }
692
693 std::shared_lock<std::shared_mutex> lock(mutex_);
694
695 // If everything parsed, just rebuild data
696 std::string ds = (parsed_data_ ? datas() : data_string_);
697
698 // Compose header with pre-allocated buffer to reduce reallocations
699 // Estimate: header is typically ~150 bytes, reserve more to be safe
700 std::string result;
701 result.reserve(200 + ds.size());
702
703 // Note: In fmt library, {{}} produces single {}, so {{{{}}}} produces {{}}
704 formatter::format_to(std::back_inserter(result), "@header={{{{");
705
706 if (message_type_ != "data_container")
707 {
708 formatter::format_to(std::back_inserter(result), "[{},{}];",
709 TARGET_ID, target_id_);
710 formatter::format_to(std::back_inserter(result), "[{},{}];",
711 TARGET_SUB_ID, target_sub_id_);
712 formatter::format_to(std::back_inserter(result), "[{},{}];",
713 SOURCE_ID, source_id_);
714 formatter::format_to(std::back_inserter(result), "[{},{}];",
715 SOURCE_SUB_ID, source_sub_id_);
716 }
717 formatter::format_to(std::back_inserter(result), "[{},{}];",
718 MESSAGE_TYPE, message_type_);
719 formatter::format_to(std::back_inserter(result), "[{},{}];",
720 MESSAGE_VERSION, version_);
721 formatter::format_to(std::back_inserter(result), "}}}};");
722
723 // Append data section directly (avoids string concatenation copy)
724 result.append(ds);
725
726 return result;
727 }

744 bool value_container::deserialize(const std::string& data_str,
745 bool parse_only_header)
746 {
747 // Record metrics if enabled
748 auto timer = metrics_manager::make_timer(
749 metrics_manager::get().deserialize_latency,
750 &metrics_manager::get().timing.total_deserialize_ns);
751 if (metrics_manager::is_enabled())
752 {
753 metrics_manager::get().operations.deserializations.fetch_add(1, std::memory_order_relaxed);
754 }
755
756 initialize();
757 if (data_str.empty())
758 return false;
759
760 // Remove newlines
761 std::regex newlineRe("\\r\\n?|\\n");
762 std::string clean = std::regex_replace(data_str, newlineRe, "");
763
764 // parse header portion - matches both single and double braces
765 std::regex fullRe("@header=\\s*\\{\\{?\\s*(.*?)\\s*\\}\\}?;");
766 std::smatch match;
767 if (!std::regex_search(clean, match, fullRe))
768 {
769 // No header => parse as data only
770 return deserialize_values(clean, parse_only_header);
771 }
772 // match[1] => inside the header
773 std::string headerInside = match[1].str();
774 std::regex pairRe("\\[(\\w+),(.*?)\\];");
775 auto it = std::sregex_iterator(headerInside.begin(), headerInside.end(),
776 pairRe);
777 auto end = std::sregex_iterator();
778 for (; it != end; ++it)
779 {
780 parsing((*it)[1].str(), std::to_string(TARGET_ID), (*it)[2].str(), target_id_);
781 parsing((*it)[1].str(), std::to_string(TARGET_SUB_ID), (*it)[2].str(), target_sub_id_);

1346
1347 std::vector<uint8_t> value_container::to_msgpack_impl() const
1348 {
1349 // Record metrics if enabled
1350 auto timer = metrics_manager::make_timer(
1351 metrics_manager::get().serialize_latency,
1352 &metrics_manager::get().timing.total_serialize_ns);
1353 if (metrics_manager::is_enabled())
1354 {
1355 metrics_manager::get().operations.serializations.fetch_add(1, std::memory_order_relaxed);
1356 }
1357
1358 std::shared_lock<std::shared_mutex> lock(mutex_);
1359
1360 msgpack_encoder encoder;
1361
1362 // Estimate buffer size: header (~100 bytes) + values
1363 encoder.reserve(200 + optimized_units_.size() * 32);
1364
1365 // MessagePack structure:
1366 // {
1367 // "header": { ... },
1368 // "values": { ... }
1369 // }
1370
1371 // Write outer map with 2 entries (header + values)
1372 encoder.write_map_header(2);
1373
1374 // Write "header" key
1375 encoder.write_string("header");
1376
1377 // Calculate header entries count
1378 size_t header_count = 2; // message_type and version are always present
1379 if (message_type_ != "data_container")
1380 {
1381 header_count += 4; // target_id, target_sub_id, source_id, source_sub_id
1382 }
1383
1384 // Write header map
1385 encoder.write_map_header(header_count);
1386
1387 if (message_type_ != "data_container")
1388 {
1389 encoder.write_string("target_id");
1390 encoder.write_string(target_id_);
1391
1392 encoder.write_string("target_sub_id");
1393 encoder.write_string(target_sub_id_);
1394
1395 encoder.write_string("source_id");
1396 encoder.write_string(source_id_);
1397
1398 encoder.write_string("source_sub_id");
1399 encoder.write_string(source_sub_id_);
1400 }
1401
1402 encoder.write_string("message_type");
1403 encoder.write_string(message_type_);
1404
1405 encoder.write_string("version");
1406 encoder.write_string(version_);
1407
1408 // Write "values" key
1409 encoder.write_string("values");
1410
1411 // Write values map
1412 encoder.write_map_header(optimized_units_.size());
1413
1414 for (const auto& unit : optimized_units_)
1415 {
1416 // Write key
1417 encoder.write_string(unit.name);
1418
1419 // Write value based on type
1420 switch (unit.type)
1421 {
1422 case value_types::null_value:
1423 encoder.write_nil();
1424 break;
1425
1426 case value_types::bool_value:
1427 encoder.write_bool(std::get<bool>(unit.data));
1428 break;
1429
1430 case value_types::short_value:
1431 encoder.write_int(std::get<short>(unit.data));
1432 break;
1433
1434 case value_types::ushort_value:
1435 encoder.write_uint(std::get<unsigned short>(unit.data));
1436 break;
1437
1438 case value_types::int_value:
1439 encoder.write_int(std::get<int>(unit.data));
1440 break;
1441
1442 case value_types::uint_value:
1443 encoder.write_uint(std::get<unsigned int>(unit.data));
1444 break;
1445
1446 case value_types::long_value:
1447 encoder.write_int(std::get<long>(unit.data));
1448 break;
1449
1450 case value_types::ulong_value:
1451 encoder.write_uint(std::get<unsigned long>(unit.data));
1452 break;
1453
1454 case value_types::llong_value:
1455 encoder.write_int(std::get<long long>(unit.data));
1456 break;
1457
1458 case value_types::ullong_value:
1459 encoder.write_uint(std::get<unsigned long long>(unit.data));
1460 break;
1461
1462 case value_types::float_value:
1463 encoder.write_float(std::get<float>(unit.data));
1464 break;
1465
1466 case value_types::double_value:
1467 encoder.write_double(std::get<double>(unit.data));
1468 break;
1469
1470 case value_types::string_value:
1471 encoder.write_string(std::get<std::string>(unit.data));
1472 break;
1473
1474 case value_types::bytes_value:
1475 encoder.write_binary(std::get<std::vector<uint8_t>>(unit.data));
1476 break;
1477
1478 case value_types::container_value:
1479 {
1480 // Nested container: serialize recursively
1481 auto nested = std::get<std::shared_ptr<value_container>>(unit.data);
1482 if (nested)
1483 {

1485 encoder.write_binary(nested_data);
1486 }
1487 else

1491 }
1492 break;

1494 case value_types::array_value:
1495 // Array values are stored as containers
1496 encoder.write_nil();
1497 break;
1498
1499 default:
1500 encoder.write_nil();
1501 break;
1502 }
1503 }
1504

1508 bool value_container::from_msgpack_impl(const std::vector<uint8_t>& data)
1509 {
1510 // Record metrics if enabled
1511 auto timer = metrics_manager::make_timer(
1512 metrics_manager::get().deserialize_latency,
1513 &metrics_manager::get().timing.total_deserialize_ns);
1514 if (metrics_manager::is_enabled())
1515 {
1516 metrics_manager::get().operations.deserializations.fetch_add(1, std::memory_order_relaxed);
1517 }
1518
1519 if (data.empty())
1520 {
1521 return false;
1522 }
1523
1524 initialize();
1525
1526 msgpack_decoder decoder(data);
1527
1528 // Expect outer map
1529 auto outer_count = decoder.read_map_header();
1530 if (!outer_count)
1531 {
1532 return false;
1533 }
1534
1535 std::unique_lock<std::shared_mutex> lock(mutex_);
1536
1537 for (size_t i = 0; i < *outer_count; ++i)
1538 {
1539 auto key = decoder.read_string();
1540 if (!key)
1541 {
1542 return false;
1543 }
1544
1545 if (*key == "header")
1546 {
1547 auto header_count = decoder.read_map_header();
1548 if (!header_count)
1549 {
1550 return false;
1551 }
1552
1553 for (size_t j = 0; j < *header_count; ++j)
1554 {
1555 auto hkey = decoder.read_string();
1556 auto hval = decoder.read_string();
1557 if (!hkey || !hval)
1558 {
1559 return false;
1560 }
1561
1562 if (*hkey == "target_id")
1563 {
1564 target_id_ = *hval;
1565 }
1566 else if (*hkey == "target_sub_id")
1567 {
1568 target_sub_id_ = *hval;
1569 }
1570 else if (*hkey == "source_id")
1571 {
1572 source_id_ = *hval;
1573 }
1574 else if (*hkey == "source_sub_id")
1575 {
1576 source_sub_id_ = *hval;
1577 }
1578 else if (*hkey == "message_type")
1579 {
1580 message_type_ = *hval;
1581 }
1582 else if (*hkey == "version")
1583 {
1584 version_ = *hval;
1585 }
1586 }
1587 }
1588 else if (*key == "values")
1589 {
1590 auto values_count = decoder.read_map_header();
1591 if (!values_count)
1592 {
1593 return false;
1594 }
1595
1596 for (size_t j = 0; j < *values_count; ++j)
1597 {
1598 auto vkey = decoder.read_string();
1599 if (!vkey)
1600 {
1601 return false;
1602 }
1603
1604 optimized_value val;
1605 val.name = *vkey;
1606
1607 // Read value based on type
1608 auto type = decoder.peek_type();
1609 switch (type)
1610 {
1611 case msgpack_type::nil:
1612 decoder.read_nil();
1613 val.type = value_types::null_value;
1614 val.data = std::monostate{};
1615 break;
1616
1617 case msgpack_type::boolean:
1618 {
1619 auto b = decoder.read_bool();
1620 if (!b)
1621 {
1622 return false;
1623 }
1624 val.type = value_types::bool_value;
1625 val.data = *b;
1626 }
1627 break;
1628
1629 case msgpack_type::positive_int:
1630 case msgpack_type::negative_int:
1631 {
1632 auto n = decoder.read_int();
1633 if (!n)
1634 {
1635 return false;
1636 }
1637 // Determine best type based on value
1638 if (*n >= 0 && *n <= INT32_MAX)
1639 {
1640 val.type = value_types::int_value;
1641 val.data = static_cast<int>(*n);
1642 }
1643 else if (*n >= INT32_MIN && *n < 0)
1644 {
1645 val.type = value_types::int_value;
1646 val.data = static_cast<int>(*n);
1647 }
1648 else
1649 {
1650 val.type = value_types::llong_value;
1651 val.data = static_cast<long long>(*n);
1652 }
1653 }
1654 break;
1655
1656 case msgpack_type::float32:
1657 {
1658 auto f = decoder.read_float();
1659 if (!f)
1660 {
1661 return false;
1662 }
1663 val.type = value_types::float_value;
1664 val.data = *f;
1665 }
1666 break;
1667
1668 case msgpack_type::float64:
1669 {
1670 auto d = decoder.read_double();
1671 if (!d)
1672 {
1673 return false;
1674 }
1675 val.type = value_types::double_value;
1676 val.data = *d;
1677 }
1678 break;
1679
1680 case msgpack_type::str:
1681 {
1682 auto s = decoder.read_string();
1683 if (!s)
1684 {
1685 return false;
1686 }
1687 val.type = value_types::string_value;
1688 val.data = *s;
1689 }
1690 break;
1691
1692 case msgpack_type::bin:
1693 {
1694 auto b = decoder.read_binary();
1695 if (!b)
1696 {
1697 return false;
1698 }
1699 val.type = value_types::bytes_value;
1700 val.data = *b;
1701 }
1702 break;
1703
1704 default:
1705 // Skip unknown types
1706 val.type = value_types::null_value;
1707 val.data = std::monostate{};
1708 break;
1709 }
1710
1711 optimized_units_.push_back(std::move(val));
1712 }
1713 }
1714 }
1715
1716 parsed_data_ = true;
1717 changed_data_ = false;
1718
1719 return true;
1720 }

1722 value_container::serialization_format value_container::detect_format(
1723 const std::vector<uint8_t>& data)
1724 {
1725 if (data.empty())
1726 {
1727 return serialization_format::unknown;
1728 }
1729
1730 // Check for MessagePack format
1731 // MessagePack maps start with 0x80-0x8f (fixmap) or 0xde (map16) or 0xdf (map32)
1732 uint8_t first_byte = data[0];
1733 if ((first_byte >= 0x80 && first_byte <= 0x8f) || // fixmap
1734 first_byte == 0xde || // map16
1735 first_byte == 0xdf) // map32
1736 {
1737 return serialization_format::msgpack;
1738 }
1739
1740 // Check for text-based formats (convert to string for analysis)
1741 std::string_view str_view(reinterpret_cast<const char*>(data.data()), data.size());
1742 return detect_format(str_view);
1743 }
1744
1745 value_container::serialization_format value_container::detect_format(
1746 std::string_view data)
1747 {
1748 if (data.empty())
1749 {
1750 return serialization_format::unknown;
1751 }
1752
1753 // Skip leading whitespace
1754 size_t pos = 0;
1755 while (pos < data.size() && std::isspace(static_cast<unsigned char>(data[pos])))
1756 {
1757 ++pos;
1758 }
1759
1760 if (pos >= data.size())
1761 {
1762 return serialization_format::unknown;
1763 }
1764
1765 // Check first non-whitespace character
1766 char first_char = data[pos];
1767
1768 // JSON starts with '{' or '['
1769 if (first_char == '{' || first_char == '[')
1770 {
1771 return serialization_format::json;
1772 }
1773
1774 // XML starts with '<'
1775 if (first_char == '<')
1776 {
1777 return serialization_format::xml;
1778 }
1779
1780 // Binary format starts with "@header"
1781 if (data.substr(pos, 7) == "@header")
1782 {
1783 return serialization_format::binary;
1784 }
1785
1786 // Binary format may also start with "@data"
1787 if (data.substr(pos, 5) == "@data")
1788 {
1789 return serialization_format::binary;
1790 }
1791
1792 return serialization_format::unknown;
1793 }

2360 bool value_container::deserialize_values(const std::string& data,
2361 bool parse_only_header)
2362 {
2363 if (!optimized_units_.empty())
2364 {
2365 optimized_units_.clear();
2366 }
2367 changed_data_ = false;
2368
2369 // Match both single and double braces for @data section
2370 std::regex reData("@data=\\s*\\{\\{?\\s*(.*?)\\s*\\}\\}?;");
2371 std::smatch match;
2372 if (!std::regex_search(data, match, reData))
2373 {
2374 data_string_ = "@data={{}}";
2375 parsed_data_ = true;
2376 return false;
2377 }
2378 data_string_ = match[0];
2379
2380 if (parse_only_header)
2381 {
2382 parsed_data_ = false;
2383 return true;
2384 }
2385 parsed_data_ = true;
2386
2387 // Parse items: [name,type,data];
2388 std::regex reItems("\\[(\\w+),\\s*(\\w+),\\s*(.*?)\\];");
2389 auto it = std::sregex_iterator(data_string_.begin(), data_string_.end(),
2390 reItems);
2391 auto end = std::sregex_iterator();
2392
2393 for (; it != end; ++it)
2394 {
2395 auto nameStr = (*it)[1].str();
2396 auto typeStr = (*it)[2].str();
2397 auto dataStr = (*it)[3].str();
2398
2399 // Convert string -> value_types
2400 auto vt = convert_value_type(typeStr);
2401
2402 // Create optimized_value and populate variant based on type
2403 optimized_value ov;
2404 ov.name = nameStr;
2405 ov.type = vt;
2406
2407 // Parse data string into appropriate variant type
2408 switch (vt)
2409 {
2410 case value_types::null_value:
2411 ov.data = std::monostate{};
2412 break;
2413 case value_types::bool_value:
2414 ov.data = (dataStr == "true" || dataStr == "1");
2415 break;
2416 case value_types::short_value:
2417 ov.data = static_cast<short>(std::stoi(dataStr));
2418 break;
2419 case value_types::ushort_value:
2420 ov.data = static_cast<unsigned short>(std::stoul(dataStr));
2421 break;
2422 case value_types::int_value:
2423 ov.data = std::stoi(dataStr);
2424 break;
2425 case value_types::uint_value:
2426 ov.data = static_cast<unsigned int>(std::stoul(dataStr));
2427 break;
2428 case value_types::long_value:
2429 ov.data = std::stol(dataStr);
2430 break;
2431 case value_types::ulong_value:
2432 ov.data = std::stoul(dataStr);
2433 break;
2434 case value_types::llong_value:
2435 ov.data = std::stoll(dataStr);
2436 break;
2437 case value_types::ullong_value:
2438 ov.data = std::stoull(dataStr);
2439 break;
2440 case value_types::float_value:
2441 ov.data = std::stof(dataStr);
2442 break;
2443 case value_types::double_value:
2444 ov.data = std::stod(dataStr);
2445 break;
2446 case value_types::string_value:
2447 ov.data = dataStr;
2448 break;
2449 case value_types::bytes_value:
2450 {
2451 // For bytes, dataStr should be the string representation
