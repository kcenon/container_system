138 void defer_delete(void* ptr, std::function<void(void*)> deleter) {
139 if (ptr == nullptr) {
140 return;
141 }
142
143 uint64_t epoch = global_epoch_.load(std::memory_order_relaxed);
144 size_t bucket = epoch % NUM_EPOCHS;
145
146 std::lock_guard<std::mutex> lock(retire_mutex_[bucket]);
147 retired_[bucket].emplace_back(ptr, std::move(deleter));
148 }
149
158 template<typename T>
159 void defer_delete(T* ptr) {
160 defer_delete(static_cast<void*>(ptr), [](void* p) {
161 delete static_cast<T*>(p);
162 });
163 }

173 size_t try_gc() {
174 // Advance epoch
175 uint64_t current = global_epoch_.fetch_add(1, std::memory_order_acq_rel);
176
177 // Calculate which epoch is safe to collect
178 // We need 2 epoch advances before an epoch is safe
179 if (current < 2) {
180 return 0;
181 }
182
183 uint64_t safe_epoch = current - 2;
184 size_t bucket = safe_epoch % NUM_EPOCHS;
185
186 // Check if any thread is still in the safe epoch
187 // For simplicity, we just try to collect
188 // In production, you'd check all registered threads
189
190 std::vector<std::pair<void*, std::function<void(void*)>>> to_delete;
191 {
192 std::lock_guard<std::mutex> lock(retire_mutex_[bucket]);
193 to_delete = std::move(retired_[bucket]);
194 retired_[bucket].clear();
195 }
196
197 // Actually delete the objects
198 for (auto& [ptr, deleter] : to_delete) {
199 deleter(ptr);
200 }
201
202 gc_count_.fetch_add(1, std::memory_order_relaxed);
203 reclaimed_count_.fetch_add(to_delete.size(), std::memory_order_relaxed);
204
205 return to_delete.size();
206 }
207
216 size_t force_gc() {
217 size_t total = 0;
218 for (size_t i = 0; i < NUM_EPOCHS; ++i) {
219 std::vector<std::pair<void*, std::function<void(void*)>>> to_delete;
220 {
221 std::lock_guard<std::mutex> lock(retire_mutex_[i]);
222 to_delete = std::move(retired_[i]);
223 retired_[i].clear();
224 }
225
226 for (auto& [ptr, deleter] : to_delete) {
227 deleter(ptr);
228 }
229 total += to_delete.size();
230 }
231
232 if (total > 0) {
233 gc_count_.fetch_add(1, std::memory_order_relaxed);
234 reclaimed_count_.fetch_add(total, std::memory_order_relaxed);
235 }
236
237 return total;
238 }
