66std::vector<uint8_t> array_value::serialize(void)
67{
68 // Serialize all values first to calculate total size
69 std::vector<std::string> serialized_values;
70 size_t total_values_size = 0;
71
72 for (const auto& val : values_)
73 {
74 auto serialized = val->serialize();
75 total_values_size += serialized.size();
76 serialized_values.push_back(std::move(serialized));
77 }
78
79 // Calculate total size: count (4 bytes) + all serialized values
80 uint32_t value_size = sizeof(uint32_t) + static_cast<uint32_t>(total_values_size);
81
82 // Create result vector
83 std::vector<uint8_t> result;
84
85 // Add type (1 byte)
86 result.push_back(static_cast<uint8_t>(value_types::array_value));
87
88 // Add name length (4 bytes, little-endian)
89 uint32_t name_len = static_cast<uint32_t>(name().length());
90 result.push_back(static_cast<uint8_t>(name_len & 0xFF));
91 result.push_back(static_cast<uint8_t>((name_len >> 8) & 0xFF));
92 result.push_back(static_cast<uint8_t>((name_len >> 16) & 0xFF));
93 result.push_back(static_cast<uint8_t>((name_len >> 24) & 0xFF));
94
95 // Add name
96 const std::string& n = name();
97 result.insert(result.end(), n.begin(), n.end());
98
99 // Add value size (4 bytes, little-endian)
100 result.push_back(static_cast<uint8_t>(value_size & 0xFF));
101 result.push_back(static_cast<uint8_t>((value_size >> 8) & 0xFF));
102 result.push_back(static_cast<uint8_t>((value_size >> 16) & 0xFF));
103 result.push_back(static_cast<uint8_t>((value_size >> 24) & 0xFF));
104
105 // Add count (4 bytes, little-endian)
106 uint32_t count = static_cast<uint32_t>(values_.size());
107 result.push_back(static_cast<uint8_t>(count & 0xFF));
108 result.push_back(static_cast<uint8_t>((count >> 8) & 0xFF));
109 result.push_back(static_cast<uint8_t>((count >> 16) & 0xFF));
110 result.push_back(static_cast<uint8_t>((count >> 24) & 0xFF));
111
112 // Add all serialized values
113 for (const auto& serialized : serialized_values)
114 {
115 result.insert(result.end(),
116 reinterpret_cast<const uint8_t*>(serialized.data()),
117 reinterpret_cast<const uint8_t*>(serialized.data() + serialized.size()));
118 }
119
120 return result;
121}
