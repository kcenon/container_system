1057 // Thread-safe lock guard classes
1058 // Always acquire lock to eliminate TOCTOU vulnerability (see #190)
1059 // Modern mutex implementations have ~20ns overhead for uncontended locks
1060 // which is negligible compared to actual data access costs

1066 }

1131 // Template implementation for get<T>()
1132 template<typename T>
1133 [[nodiscard]] kcenon::common::Result<T> value_container::get(std::string_view key) const noexcept {
1134 read_lock_guard lock(this);
1135
1136 for (const auto& val : optimized_units_) {
1137 if (val.name == key) {
1138 // Try to extract the value of type T from the variant
1139 if (auto* ptr = std::get_if<T>(&val.data)) {
1140 return kcenon::common::ok(*ptr);
1141 }
1142 // Type mismatch
1143 return kcenon::common::Result<T>(
1144 kcenon::common::error_info{
1145 error_codes::type_mismatch,
1146 error_codes::make_message(error_codes::type_mismatch, key),
1147 "container_system"});
1148 }
1149 }
1150 // Key not found
1151 return kcenon::common::Result<T>(
1152 kcenon::common::error_info{
1153 error_codes::key_not_found,
1154 error_codes::make_message(error_codes::key_not_found, key),
1155 "container_system"});
1156 }

1160 [[nodiscard]] kcenon::common::VoidResult value_container::set_result(
1161 std::string_view key, T&& data_val) noexcept {
1162 try {
1163 if (key.empty()) {
1164 return kcenon::common::VoidResult(
1165 kcenon::common::error_info{
1166 error_codes::empty_key,
1167 error_codes::make_message(error_codes::empty_key),
1168 "container_system"});
1169 }
1170
1171 optimized_value val;
1172 val.name = std::string(key);
1173 val.data = std::forward<T>(data_val);
1174 val.type = static_cast<value_types>(val.data.index());
1175 set_unit_impl(val);
1176 return kcenon::common::ok();
1177 } catch (const std::bad_alloc&) {
1178 return kcenon::common::VoidResult(
1179 kcenon::common::error_info{
1180 error_codes::memory_allocation_failed,

1185 kcenon::common::error_info{
