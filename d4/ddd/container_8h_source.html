350 // Thread-safe helper classes
351 // FIXED: Eliminated TOCTOU (Time-Of-Check-Time-Of-Use) vulnerability
352 // by always acquiring lock first, then checking thread_safe_enabled
353 // inside the critical section
354 class read_lock_guard {
355 std::shared_lock<std::shared_mutex> lock_;
356 bool is_active_;
357 public:
358 read_lock_guard(const value_container* c)
359 : lock_(c->mutex_) // Always acquire lock first (eliminates race)
360 , is_active_(c->thread_safe_enabled_.load(std::memory_order_acquire)) {
361 // Now safely check if we needed the lock
362 if (is_active_) {
363 c->read_count_.fetch_add(1, std::memory_order_relaxed);
364 }
365 }
366
367 // Query whether lock is actually being used
368 bool is_locked() const noexcept { return is_active_; }
369 };
