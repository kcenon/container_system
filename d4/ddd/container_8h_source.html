1049 // Thread-safe lock guard classes
1050 // Always acquire lock to eliminate TOCTOU vulnerability (see #190)
1051 // Modern mutex implementations have ~20ns overhead for uncontended locks
1052 // which is negligible compared to actual data access costs
1053 class read_lock_guard {
1054 std::shared_lock<std::shared_mutex> lock_;
1055 public:
1056 explicit read_lock_guard(const value_container* c)
1057 : lock_(c->mutex_) {
1058 }
1059

1062 };

1124 template<typename T>
1125 [[nodiscard]] kcenon::common::Result<T> value_container::get(std::string_view key) const noexcept {
1126 read_lock_guard lock(this);
1127
1128 for (const auto& val : optimized_units_) {
1129 if (val.name == key) {
1130 // Try to extract the value of type T from the variant
1131 if (auto* ptr = std::get_if<T>(&val.data)) {
1132 return kcenon::common::ok(*ptr);
1133 }
1134 // Type mismatch
1135 return kcenon::common::Result<T>(
1136 kcenon::common::error_info{
1137 error_codes::type_mismatch,
1138 error_codes::make_message(error_codes::type_mismatch, key),
1139 "container_system"});
1140 }
1141 }
1142 // Key not found
1143 return kcenon::common::Result<T>(
1144 kcenon::common::error_info{
1145 error_codes::key_not_found,
1146 error_codes::make_message(error_codes::key_not_found, key),
