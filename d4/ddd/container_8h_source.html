77 struct optimized_value
78 {
79 std::string name;
80 value_types type;
81 value_variant data;
82
83 optimized_value()
84 : name("")
85 , type(value_types::null_value)
86 , data(std::monostate{})
87 {}
88
89 optimized_value(const std::string& n, value_types t)
90 : name(n)
91 , type(t)
92 , data(std::monostate{})
93 {}
94
95 size_t memory_footprint() const
96 {
97 size_t base = sizeof(optimized_value);
98 base += name.capacity();
99 if (std::holds_alternative<std::string>(data)) {
100 base += std::get<std::string>(data).capacity();
101 } else if (std::holds_alternative<std::vector<uint8_t>>(data)) {
102 base += std::get<std::vector<uint8_t>>(data).capacity();
103 }
104 return base;
105 }
106
107 bool is_stack_allocated() const
108 {
109 return type != value_types::string_value &&
110 type != value_types::bytes_value &&
111 type != value_types::container_value;
112 }
113 };

140 inline std::string to_string(const value_variant& var, value_types type)
141 {
142 switch (type)
143 {
144 case value_types::null_value:
145 return "";
146 case value_types::bool_value:
147 return std::get<bool>(var) ? "true" : "false";
148 case value_types::short_value:
149 return std::to_string(std::get<short>(var));
150 case value_types::ushort_value:
151 return std::to_string(std::get<unsigned short>(var));
152 case value_types::int_value:
153 return std::to_string(std::get<int>(var));
154 case value_types::uint_value:
155 return std::to_string(std::get<unsigned int>(var));
156 case value_types::long_value:
157 return std::to_string(std::get<long>(var));
158 case value_types::ulong_value:
159 return std::to_string(std::get<unsigned long>(var));
160 case value_types::llong_value:
161 return std::to_string(std::get<long long>(var));
162 case value_types::ullong_value:
163 return std::to_string(std::get<unsigned long long>(var));
164 case value_types::float_value:
165 return std::to_string(std::get<float>(var));
166 case value_types::double_value:
167 return std::to_string(std::get<double>(var));
168 case value_types::string_value:
169 return std::get<std::string>(var);
170 default:
171 return "";
172 }
173 }

178 inline size_t data_size(const value_variant& var, value_types type)
179 {
180 switch (type)
181 {
182 case value_types::null_value:
183 return 0;
184 case value_types::bool_value:
185 return sizeof(bool);
186 case value_types::short_value:
187 return sizeof(short);
188 case value_types::ushort_value:
189 return sizeof(unsigned short);
190 case value_types::int_value:
191 return sizeof(int);
192 case value_types::uint_value:
193 return sizeof(unsigned int);
194 case value_types::long_value:
195 return sizeof(long);
196 case value_types::ulong_value:
197 return sizeof(unsigned long);
198 case value_types::llong_value:
199 return sizeof(long long);
200 case value_types::ullong_value:
201 return sizeof(unsigned long long);
202 case value_types::float_value:
203 return sizeof(float);
204 case value_types::double_value:
205 return sizeof(double);
206 case value_types::string_value:
207 return std::get<std::string>(var).size();
208 case value_types::bytes_value:
209 return std::get<std::vector<uint8_t>>(var).size();
210 default:
211 return 0;
212 }
213 }

395 template<typename T>
396 void add_value(const std::string& name, T&& data_val) {
397 write_lock_guard lock(this);
398
399 optimized_value val;
400 val.name = name;
401 val.data = std::forward<T>(data_val);
402 val.type = static_cast<value_types>(val.data.index());
403
404 optimized_units_.push_back(std::move(val));
405 changed_data_ = true;
406
407 if (use_soo_ && val.is_stack_allocated()) {
408 stack_allocations_.fetch_add(1, std::memory_order_relaxed);
409 } else {
410 heap_allocations_.fetch_add(1, std::memory_order_relaxed);
411 }
412 }

697 // Thread-safe helper classes with conditional lock acquisition
698 // Optimized: Only acquire lock when thread-safe mode is enabled
699 // This eliminates lock overhead in single-threaded scenarios
700 class read_lock_guard {
701 std::optional<std::shared_lock<std::shared_mutex>> lock_;
702 bool is_active_;
703 public:
704 read_lock_guard(const value_container* c)
705 : is_active_(c->thread_safe_enabled_.load(std::memory_order_acquire)) {
706 // Conditionally acquire lock only when thread-safe mode is enabled
707 if (is_active_) {
708 lock_.emplace(c->mutex_);
709 c->read_count_.fetch_add(1, std::memory_order_relaxed);
710 }
711 }
712
713 // Query whether lock is actually being used
714 bool is_locked() const noexcept { return is_active_; }
715 };
716
717 class write_lock_guard {
