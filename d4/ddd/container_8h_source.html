215 template<typename T>
216 void add_value(const std::string& name, T&& value) {
217 write_lock_guard lock(this);
218
219 optimized_value val;
220 val.name = name;
221 val.data = std::forward<T>(value);
222 val.type = static_cast<value_types>(val.data.index());
223
224 optimized_units_.push_back(std::move(val));
225 changed_data_ = true;
226
227 if (use_soo_ && val.is_stack_allocated()) {
228 stack_allocations_.fetch_add(1, std::memory_order_relaxed);
229 } else {
230 heap_allocations_.fetch_add(1, std::memory_order_relaxed);
231 }
232 }

448 // Thread-safe helper classes with conditional lock acquisition
449 // Optimized: Only acquire lock when thread-safe mode is enabled
450 // This eliminates lock overhead in single-threaded scenarios
451 class read_lock_guard {
452 std::optional<std::shared_lock<std::shared_mutex>> lock_;
453 bool is_active_;
454 public:
455 read_lock_guard(const value_container* c)
456 : is_active_(c->thread_safe_enabled_.load(std::memory_order_acquire)) {
457 // Conditionally acquire lock only when thread-safe mode is enabled
458 if (is_active_) {
459 lock_.emplace(c->mutex_);
460 c->read_count_.fetch_add(1, std::memory_order_relaxed);
461 }
462 }
463
464 // Query whether lock is actually being used
465 bool is_locked() const noexcept { return is_active_; }
466 };
467
468 class write_lock_guard {
469 std::optional<std::unique_lock<std::shared_mutex>> lock_;
470 bool is_active_;
471 public:
472 write_lock_guard(value_container* c)
473 : is_active_(c->thread_safe_enabled_.load(std::memory_order_acquire)) {
474 // Conditionally acquire lock only when thread-safe mode is enabled
475 if (is_active_) {
476 lock_.emplace(c->mutex_);
477 c->write_count_.fetch_add(1, std::memory_order_relaxed);
478 }
479 }
480
481 // Query whether lock is actually being used
482 bool is_locked() const noexcept { return is_active_; }
