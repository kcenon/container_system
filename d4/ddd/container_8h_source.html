247 [[deprecated("Use set() instead")]]
248 void add_value(const std::string& name, T&& data_val) {
249 write_lock_guard lock(this);
250
251 optimized_value val;
252 val.name = name;
253 val.data = std::forward<T>(data_val);
254 val.type = static_cast<value_types>(val.data.index());
255
256 optimized_units_.push_back(std::move(val));
257 changed_data_ = true;
258
259 if (use_soo_ && val.is_stack_allocated()) {
260 stack_allocations_.fetch_add(1, std::memory_order_relaxed);
261 } else {
262 heap_allocations_.fetch_add(1, std::memory_order_relaxed);
263 }
264 }

621 // Thread-safe lock guard classes
622 // Always acquire lock to eliminate TOCTOU vulnerability (see #190)
623 // Modern mutex implementations have ~20ns overhead for uncontended locks
624 // which is negligible compared to actual data access costs
625 class read_lock_guard {
626 std::shared_lock<std::shared_mutex> lock_;
627 public:
628 explicit read_lock_guard(const value_container* c)
629 : lock_(c->mutex_) {
630 }
631
632 // Always locked - no TOCTOU race condition possible
633 bool is_locked() const noexcept { return true; }
634 };
635
636 class write_lock_guard {
637 std::unique_lock<std::shared_mutex> lock_;
638 public:
639 explicit write_lock_guard(value_container* c)
640 : lock_(c->mutex_) {
641 }
642
643 // Always locked - no TOCTOU race condition possible
644 bool is_locked() const noexcept { return true; }
645 };

682#if CONTAINER_HAS_COMMON_RESULT
683 // Template implementation for get<T>()
684 template<typename T>
685 [[nodiscard]] kcenon::common::Result<T> value_container::get(std::string_view key) const noexcept {
686 read_lock_guard lock(this);
687
688 for (const auto& val : optimized_units_) {
689 if (val.name == key) {
690 // Try to extract the value of type T from the variant
691 if (auto* ptr = std::get_if<T>(&val.data)) {
692 return kcenon::common::ok(*ptr);
693 }
694 // Type mismatch
695 return kcenon::common::Result<T>(
696 kcenon::common::error_info{-2, "Type mismatch for key: " + std::string(key), "container_system"});
697 }
698 }
699 // Key not found
700 return kcenon::common::Result<T>(
701 kcenon::common::error_info{-1, "Key not found: " + std::string(key), "container_system"});
702 }
