394 // Thread-safe helper classes with conditional lock acquisition
395 // Optimized: Only acquire lock when thread-safe mode is enabled
396 // This eliminates lock overhead in single-threaded scenarios
397 class read_lock_guard {
398 std::optional<std::shared_lock<std::shared_mutex>> lock_;
399 bool is_active_;
400 public:
401 read_lock_guard(const value_container* c)
402 : is_active_(c->thread_safe_enabled_.load(std::memory_order_acquire)) {
403 // Conditionally acquire lock only when thread-safe mode is enabled
404 if (is_active_) {
405 lock_.emplace(c->mutex_);
406 c->read_count_.fetch_add(1, std::memory_order_relaxed);
407 }
408 }
409
410 // Query whether lock is actually being used
411 bool is_locked() const noexcept { return is_active_; }
412 };
413
414 class write_lock_guard {
415 std::optional<std::unique_lock<std::shared_mutex>> lock_;
416 bool is_active_;
417 public:
418 write_lock_guard(value_container* c)
419 : is_active_(c->thread_safe_enabled_.load(std::memory_order_acquire)) {
420 // Conditionally acquire lock only when thread-safe mode is enabled
421 if (is_active_) {
422 lock_.emplace(c->mutex_);
423 c->write_count_.fetch_add(1, std::memory_order_relaxed);
