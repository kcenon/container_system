46 // Convert to string
47 template<typename T>
48 static std::string to_string(const T& value) {
49 if constexpr (std::is_arithmetic_v<T>) {
50 return std::to_string(value);
51 } else {
52 std::ostringstream oss;
53 oss << value;
54 return oss.str();
55 }
56 }

58 // Specialization for bool
59 static std::string to_string(bool value) {
60 return value ? "true" : "false";
61 }
62
63 // Specialization for string (no conversion needed)
64 static const std::string& to_string(const std::string& value) {
65 return value;
66 }

68 // Convert from string
69 template<typename T>
70 static T from_string(const std::string& str) {
71 if constexpr (std::is_same_v<T, std::string>) {
72 return str;
73 } else if constexpr (std::is_same_v<T, int>) {
74 return std::stoi(str);
75 } else if constexpr (std::is_same_v<T, long>) {
76 return std::stol(str);
77 } else if constexpr (std::is_same_v<T, long long>) {
78 return std::stoll(str);
79 } else if constexpr (std::is_same_v<T, float>) {
80 return std::stof(str);
81 } else if constexpr (std::is_same_v<T, double>) {
82 return std::stod(str);
83 } else if constexpr (std::is_same_v<T, bool>) {
84 return str == "true" || str == "1" || str == "yes";
85 } else {
86 std::istringstream iss(str);
87 T value;
88 iss >> value;
89 return value;
90 }
91 }

93 // Convert bytes to hex string
94 static std::string bytes_to_hex(const std::vector<uint8_t>& bytes) {
95 std::ostringstream oss;
96 oss << std::hex;
97 for (uint8_t byte : bytes) {
98 oss << static_cast<int>(byte);
99 }
100 return oss.str();
101 }
102
103 // Convert hex string to bytes
104 static std::vector<uint8_t> hex_to_bytes(const std::string& hex) {
105 std::vector<uint8_t> bytes;
106 bytes.reserve(hex.length() / 2);
107
108 for (size_t i = 0; i < hex.length(); i += 2) {
109 std::string byte_str = hex.substr(i, 2);
110 uint8_t byte = static_cast<uint8_t>(std::stoul(byte_str, nullptr, 16));
111 bytes.push_back(byte);
112 }
113
114 return bytes;
115 }

117 // Trim whitespace
118 static std::string trim(const std::string& str) {
119 size_t start = str.find_first_not_of(" \t\r\n");
120 if (start == std::string::npos) return "";
121
122 size_t end = str.find_last_not_of(" \t\r\n");
123 return str.substr(start, end - start + 1);
124 }

126 // Split string by delimiter
127 static std::vector<std::string> split(const std::string& str, char delimiter) {
128 std::vector<std::string> tokens;
129 std::istringstream iss(str);
130 std::string token;
131
132 while (std::getline(iss, token, delimiter)) {
133 tokens.push_back(token);
134 }
135
136 return tokens;
137 }
138
139 // Join strings with delimiter
140 static std::string join(const std::vector<std::string>& strings, const std::string& delimiter) {
141 if (strings.empty()) return "";
142
143 std::ostringstream oss;
144 oss << strings[0];
145
146 for (size_t i = 1; i < strings.size(); ++i) {
147 oss << delimiter << strings[i];
148 }
149
150 return oss.str();
151 }
