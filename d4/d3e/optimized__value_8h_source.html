112 size_t memory_footprint() const
113 {
114 size_t base = sizeof(optimized_value);
115
116 // Add dynamic string size
117 base += name.capacity();
118
119 // Add variant content size for dynamic types
120 if (std::holds_alternative<std::string>(data))
121 {
122 base += std::get<std::string>(data).capacity();
123 }
124 else if (std::holds_alternative<std::vector<uint8_t>>(data))
125 {
126 base += std::get<std::vector<uint8_t>>(data).capacity();
127 }
128
129 return base;
130 }

136 bool is_stack_allocated() const
137 {
138 return type != value_types::string_value &&
139 type != value_types::bytes_value &&
140 type != value_types::container_value;
141 }
