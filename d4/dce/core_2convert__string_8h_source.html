73 static std::pair<std::string, std::string> to_base64(const std::vector<uint8_t>& data) {
74 try {
75 const std::string chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
76 std::string result;
77 int val = 0;
78 int valb = -6;
79
80 for (uint8_t c : data) {
81 val = (val << 8) + c;
82 valb += 8;
83 while (valb >= 0) {
84 result.push_back(chars[(val >> valb) & 0x3F]);
85 valb -= 6;
86 }
87 }
88
89 if (valb > -6) {
90 result.push_back(chars[((val << 8) >> (valb + 8)) & 0x3F]);
91 }
92
93 while (result.size() % 4) {
94 result.push_back('=');
95 }
96
97 return {result, ""};
98 } catch (const std::exception& e) {
99 return {"", e.what()};
100 }

106 static std::pair<std::vector<uint8_t>, std::string> from_base64(const std::string& str) {
107 try {
108 const std::string chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
109 std::vector<uint8_t> result;
110 int val = 0;
111 int valb = -8;
112
113 for (char c : str) {
114 if (c == '=') break;
115 size_t pos = chars.find(c);
116 if (pos == std::string::npos) continue;
117
118 val = (val << 6) + pos;
119 valb += 6;
120 if (valb >= 0) {
121 result.push_back((val >> valb) & 0xFF);
122 valb -= 8;
123 }
124 }
125
126 return {result, ""};
127 } catch (const std::exception& e) {
128 return {{}, e.what()};
129 }

135 static void replace(std::string& str, const std::string& from, const std::string& to) {
136 if (from.empty()) return;
137
138 size_t start_pos = 0;
139 while ((start_pos = str.find(from, start_pos)) != std::string::npos) {
140 str.replace(start_pos, from.length(), to);
141 start_pos += to.length();
142 }
143 }
