227 void record(uint64_t latency_ns) noexcept
228 {
229 // Update sample count
230 uint64_t count = sample_count.fetch_add(1, std::memory_order_relaxed) + 1;
231
232 // Update max
233 uint64_t current_max = max_ns.load(std::memory_order_relaxed);
234 while (latency_ns > current_max &&
235 !max_ns.compare_exchange_weak(current_max, latency_ns,
236 std::memory_order_relaxed))
237 {
238 }
239
240 // Update min
241 uint64_t current_min = min_ns.load(std::memory_order_relaxed);
242 while (latency_ns < current_min &&
243 !min_ns.compare_exchange_weak(current_min, latency_ns,
244 std::memory_order_relaxed))
245 {
246 }
247
248 // Update sum
249 sum_ns.fetch_add(latency_ns, std::memory_order_relaxed);
250
251 // Reservoir sampling
252 size_t idx = reservoir_count.load(std::memory_order_relaxed);
253 if (idx < kReservoirSize)
254 {
255 // Fill the reservoir initially
256 reservoir[idx].store(latency_ns, std::memory_order_relaxed);
257 reservoir_count.fetch_add(1, std::memory_order_relaxed);
258 }
259 else
260 {
261 // Probabilistic replacement for uniform sampling
262 // Use simple hash of count as pseudo-random
263 size_t replace_idx = (count * 2654435769ULL) % kReservoirSize;
264 if ((count * 2654435769ULL >> 32) % count < kReservoirSize)
265 {
266 reservoir[replace_idx].store(latency_ns, std::memory_order_relaxed);
267 }
268 }
269 }

276 [[nodiscard]] uint64_t get_percentile(double percentile) const noexcept
277 {
278 size_t count = reservoir_count.load(std::memory_order_relaxed);
279 if (count == 0)
280 {
281 return 0;
282 }
283
284 // Copy reservoir values to local array for sorting
285 std::array<uint64_t, kReservoirSize> values{};
286 for (size_t i = 0; i < count && i < kReservoirSize; ++i)
287 {
288 values[i] = reservoir[i].load(std::memory_order_relaxed);
289 }
290
291 // Sort the values
292 std::sort(values.begin(), values.begin() + count);
293
294 // Calculate percentile index
295 size_t idx = static_cast<size_t>(percentile * static_cast<double>(count - 1));
296 if (idx >= count)
297 {
298 idx = count - 1;
299 }
300
301 return values[idx];
302 }

327 [[nodiscard]] double avg() const noexcept
328 {
329 uint64_t count = sample_count.load(std::memory_order_relaxed);
330 if (count == 0)
331 {
332 return 0.0;
333 }
334 return static_cast<double>(sum_ns.load(std::memory_order_relaxed)) /

605 ~scoped_timer() noexcept
606 {
607 if (enabled_)
608 {

612 auto latency = static_cast<uint64_t>(duration);
613
614 histogram_.record(latency);

619 }
