164// Template implementation for add_value
165template<typename T>
166messaging_container_builder& messaging_container_builder::add_value(const std::string& key, T&& value) {
167 if constexpr (std::is_same_v<std::decay_t<T>, std::shared_ptr<value_container>>) {
168 // Handle nested containers by serializing and storing as bytes
169 std::string serialized_data = value->serialize();
170 std::vector<uint8_t> bytes(serialized_data.begin(), serialized_data.end());
171 container_->set_value(key, bytes);
172 } else if constexpr (std::is_same_v<std::decay_t<T>, bool>) {
173 container_->set_value(key, value);
174 } else if constexpr (std::is_integral_v<std::decay_t<T>>) {
175 if constexpr (sizeof(T) <= 4) {
176 container_->set_value(key, static_cast<int32_t>(value));
177 } else {
178 container_->set_value(key, static_cast<int64_t>(value));
179 }
180 } else if constexpr (std::is_floating_point_v<std::decay_t<T>>) {
181 if constexpr (std::is_same_v<std::decay_t<T>, float>) {
182 container_->set_value(key, value);
183 } else {
184 container_->set_value(key, value);
185 }
186 } else if constexpr (std::is_same_v<std::decay_t<T>, std::string>) {
187 container_->set_value(key, std::string(value));
188 }
189
190 return *this;
191}
