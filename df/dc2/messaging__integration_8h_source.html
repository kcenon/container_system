166// Template implementation for add_value
167template<typename T>
168messaging_container_builder& messaging_container_builder::add_value(const std::string& key, T&& value) {
169 if constexpr (std::is_same_v<std::decay_t<T>, std::shared_ptr<value_container>>) {
170 // Handle nested containers by serializing and storing as container_value
171 std::string serialized_data = value->serialize();
172 auto val = std::make_shared<container_module::value>(key, value_types::container_value, serialized_data);
173 container_->add(val);
174 } else if constexpr (std::is_same_v<std::decay_t<T>, bool>) {
175 auto val = std::make_shared<bool_value>(key, value);
176 container_->add(val);
177 } else if constexpr (std::is_integral_v<std::decay_t<T>>) {
178 if constexpr (sizeof(T) <= 4) {
179 auto val = std::make_shared<int_value>(key, static_cast<int32_t>(value));
180 container_->add(val);
181 } else {
182 auto val = std::make_shared<llong_value>(key, static_cast<int64_t>(value));
183 container_->add(val);
184 }
185 } else if constexpr (std::is_floating_point_v<std::decay_t<T>>) {
186 if constexpr (std::is_same_v<std::decay_t<T>, float>) {
187 auto val = std::make_shared<float_value>(key, value);
188 container_->add(val);
189 } else {
190 auto val = std::make_shared<double_value>(key, value);
191 container_->add(val);
192 }
193 } else if constexpr (std::is_same_v<std::decay_t<T>, std::string>) {
194 auto val = std::make_shared<string_value>(key, value);
195 container_->add(val);
196 }
197
198 return *this;
199}
