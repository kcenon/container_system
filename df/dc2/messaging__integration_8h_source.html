175// Template implementation for set (unified API)
176// Uses concepts to provide clearer error messages for unsupported types
177template<typename T>
178messaging_container_builder& messaging_container_builder::set(const std::string& key, T&& value) {
179 using DecayedT = std::decay_t<T>;
180
181 if constexpr (std::is_same_v<DecayedT, std::shared_ptr<value_container>>) {
182 // Handle nested containers by serializing and storing as bytes
183 auto result = value->serialize_string(value_container::serialization_format::binary);
184 if (result.is_ok()) {
185 std::string serialized_data = result.value();
186 std::vector<uint8_t> bytes(serialized_data.begin(), serialized_data.end());
187 container_->set(key, bytes);
188 }
189 } else if constexpr (std::is_same_v<DecayedT, bool>) {
190 container_->set(key, value);
191 } else if constexpr (concepts::IntegralType<DecayedT>) {
192 if constexpr (sizeof(DecayedT) <= 4) {
193 container_->set(key, static_cast<int32_t>(value));
194 } else {
195 container_->set(key, static_cast<int64_t>(value));
196 }
197 } else if constexpr (concepts::FloatingPointType<DecayedT>) {
198 if constexpr (std::is_same_v<DecayedT, float>) {
199 container_->set(key, value);
200 } else {
201 container_->set(key, value);
202 }
203 } else if constexpr (std::is_same_v<DecayedT, std::string>) {
204 container_->set(key, std::string(value));
205 } else if constexpr (concepts::StringLike<DecayedT>) {
206 container_->set(key, std::string(value));
207 }
208
209 return *this;
210}
