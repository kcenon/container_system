58struct pool_size_class {
59 static constexpr std::size_t small_threshold = 64; // <= 64 bytes
60 static constexpr std::size_t medium_threshold = 256; // <= 256 bytes
61 static constexpr std::size_t blocks_per_chunk = 1024;
62};

67struct pool_allocator_stats {
68 std::size_t pool_hits{0}; // Allocations satisfied from pool
69 std::size_t pool_misses{0}; // Allocations that went to heap
70 std::size_t small_pool_allocs{0}; // Small pool allocations
71 std::size_t medium_pool_allocs{0}; // Medium pool allocations
72 std::size_t deallocations{0}; // Total deallocations
73
74 double hit_rate() const noexcept {
75 auto total = pool_hits + pool_misses;
76 return total > 0 ? static_cast<double>(pool_hits) / total : 0.0;
77 }
78};

86class pool_allocator {
87public:
92 static pool_allocator& instance() noexcept {
93 thread_local pool_allocator allocator;
94 return allocator;
95 }
96
102 void* allocate(std::size_t size) noexcept {
103 if (size == 0) {
104 return nullptr;
105 }
106
107#if CONTAINER_USE_MEMORY_POOL
108 try {
109 if (size <= pool_size_class::small_threshold) {
110 void* ptr = small_pool_.allocate();
111 stats_.pool_hits++;
112 stats_.small_pool_allocs++;
113 return ptr;
114 } else if (size <= pool_size_class::medium_threshold) {
115 void* ptr = medium_pool_.allocate();
116 stats_.pool_hits++;
117 stats_.medium_pool_allocs++;
118 return ptr;
119 }
120 } catch (...) {
121 // Pool allocation failed, fall through to heap
122 }
123#endif
124 // Large allocation or pool disabled: use heap
125 stats_.pool_misses++;
126 return ::operator new(size, std::nothrow);
127 }
128
134 void deallocate(void* ptr, std::size_t size) noexcept {
135 if (!ptr) {
136 return;
137 }
138
139 stats_.deallocations++;
140
141#if CONTAINER_USE_MEMORY_POOL
142 if (size <= pool_size_class::small_threshold) {
143 small_pool_.deallocate(ptr);
144 return;
145 } else if (size <= pool_size_class::medium_threshold) {
146 medium_pool_.deallocate(ptr);
147 return;
148 }
149#endif
150 // Large allocation: use heap
151 ::operator delete(ptr);
152 }
153
158 pool_allocator_stats get_stats() const noexcept {
159 return stats_;
160 }
161
166 fixed_block_pool::statistics get_small_pool_stats() const {
167 return small_pool_.get_statistics();
168 }
169
174 fixed_block_pool::statistics get_medium_pool_stats() const {
175 return medium_pool_.get_statistics();
176 }
177
181 void reset_stats() noexcept {
182 stats_ = pool_allocator_stats{};
183 }
184
185private:
186 pool_allocator()
187 : small_pool_(pool_size_class::small_threshold, pool_size_class::blocks_per_chunk)
188 , medium_pool_(pool_size_class::medium_threshold, pool_size_class::blocks_per_chunk)
189 {}
190
191 // Non-copyable, non-movable
192 pool_allocator(const pool_allocator&) = delete;
193 pool_allocator& operator=(const pool_allocator&) = delete;
194 pool_allocator(pool_allocator&&) = delete;
195 pool_allocator& operator=(pool_allocator&&) = delete;
196
197 fixed_block_pool small_pool_;
198 fixed_block_pool medium_pool_;
199 pool_allocator_stats stats_;
200};

209template<typename T, typename... Args>
210T* pool_allocate(Args&&... args) noexcept {
211 static_assert(std::is_nothrow_destructible_v<T>,
212 "pool_allocate requires nothrow destructible types");
213
214 void* ptr = pool_allocator::instance().allocate(sizeof(T));
215 if (!ptr) {
216 return nullptr;
217 }
218
219 try {
220 return new (ptr) T(std::forward<Args>(args)...);
221 } catch (...) {
222 pool_allocator::instance().deallocate(ptr, sizeof(T));
223 return nullptr;
224 }
225}
226
232template<typename T>
233void pool_deallocate(T* ptr) noexcept {
234 if (!ptr) {
235 return;
236 }
237
238 ptr->~T();
239 pool_allocator::instance().deallocate(ptr, sizeof(T));
240}

247template<typename T>
248constexpr bool is_pool_allocatable() noexcept {
249 return sizeof(T) <= pool_size_class::medium_threshold;
250}

257inline constexpr int get_size_class(std::size_t size) noexcept {
258 if (size <= pool_size_class::small_threshold) {
259 return 0;
260 } else if (size <= pool_size_class::medium_threshold) {
261 return 1;
262 }
263 return 2;
