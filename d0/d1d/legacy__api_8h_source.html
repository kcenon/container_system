239template<typename T>
240void value_container::add_value(const std::string& name, T&& data_val) {
241 write_lock_guard lock(this);
242
243 optimized_value val;
244 val.name = name;
245 val.data = std::forward<T>(data_val);
246 val.type = static_cast<value_types>(val.data.index());
247
248 optimized_units_.push_back(std::move(val));
249 changed_data_ = true;
250
251 if (use_soo_ && val.is_stack_allocated()) {
252 stack_allocations_.fetch_add(1, std::memory_order_relaxed);
253 } else {
254 heap_allocations_.fetch_add(1, std::memory_order_relaxed);
255 }
256}
