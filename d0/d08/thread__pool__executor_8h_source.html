141 void await_suspend(std::coroutine_handle<> handle)
142 {
143 auto state = state_;
144 state->continuation_ = handle;
145
146#ifdef KCENON_HAS_COMMON_SYSTEM
147 if (executor_ && executor_->is_running())
148 {
149 // Create a job that executes the work
150 class work_job : public kcenon::common::interfaces::IJob
151 {
152 public:
153 work_job(std::shared_ptr<executor_state<T>> s)
154 : state_(std::move(s)) {}
155
156 kcenon::common::VoidResult execute() override
157 {
158 try {
159 state_->result_.emplace(state_->work_());
160 } catch (...) {
161 state_->exception_ = std::current_exception();
162 }
163 state_->ready_.store(true, std::memory_order_release);
164 if (state_->continuation_) {
165 state_->continuation_.resume();
166 }
167 return kcenon::common::ok();
168 }
169
170 std::string get_name() const override {
171 return "async_container_work";
172 }
173
174 private:
175 std::shared_ptr<executor_state<T>> state_;
176 };
177
178 auto job = std::make_unique<work_job>(state);
179 auto result = executor_->execute(std::move(job));
180 if (!result.is_ok()) {
181 // Fallback to thread if executor fails
182 run_in_thread(state, handle);
183 }
184 return;
185 }
186#endif
187 // Fallback: run in a detached thread
188 run_in_thread(state, handle);
189 }
190
191 T await_resume()
192 {
193 while (!state_->ready_.load(std::memory_order_acquire)) {
194 // Spin until ready
195 }
196 if (state_->exception_) {
197 std::rethrow_exception(state_->exception_);
198 }
199 return std::move(*state_->result_);
200 }
201
202 private:
203 static void run_in_thread(
204 std::shared_ptr<executor_state<T>> state,
205 std::coroutine_handle<> handle)
206 {
207 std::thread([state, handle]() mutable {
208 try {
209 state->result_.emplace(state->work_());
210 } catch (...) {
211 state->exception_ = std::current_exception();
212 }
213 state->ready_.store(true, std::memory_order_release);
214 handle.resume();
215 }).detach();
216 }
217 };
