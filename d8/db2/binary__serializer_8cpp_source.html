52#if CONTAINER_HAS_RESULT
53kcenon::common::Result<std::vector<uint8_t>>
54binary_serializer::serialize(const value_container& container) const noexcept
55{
56 try
57 {
58 // Get container data through public accessors
59 auto message_type = container.message_type();
60 auto target_id = container.target_id();
61 auto target_sub_id = container.target_sub_id();
62 auto source_id = container.source_id();
63 auto source_sub_id = container.source_sub_id();
64 auto version = container.version();
65 auto data_section = container.datas();
66
67 // Build header with pre-allocated buffer
68 std::string result;
69 result.reserve(200 + data_section.size());
70
71 // Note: In fmt library, {{}} produces single {}, so {{{{}}}} produces {{}}
72 utility_module::formatter::format_to(std::back_inserter(result), "@header={{{{");
73
74 if (message_type != "data_container")
75 {
76 utility_module::formatter::format_to(std::back_inserter(result), "[{},{}];",
77 TARGET_ID, target_id);
78 utility_module::formatter::format_to(std::back_inserter(result), "[{},{}];",
79 TARGET_SUB_ID, target_sub_id);
80 utility_module::formatter::format_to(std::back_inserter(result), "[{},{}];",
81 SOURCE_ID, source_id);
82 utility_module::formatter::format_to(std::back_inserter(result), "[{},{}];",
83 SOURCE_SUB_ID, source_sub_id);
84 }
85 utility_module::formatter::format_to(std::back_inserter(result), "[{},{}];",
86 MESSAGE_TYPE, message_type);
87 utility_module::formatter::format_to(std::back_inserter(result), "[{},{}];",
88 MESSAGE_VERSION, version);
89 utility_module::formatter::format_to(std::back_inserter(result), "}}}};");
90
91 // Append data section
92 result.append(data_section);
93
94 // Convert to byte array
95 auto [arr, err] = utility_module::convert_string::to_array(result);
96 if (!err.empty())
97 {
98 return kcenon::common::Result<std::vector<uint8_t>>(
99 kcenon::common::error_info{
100 -1,
101 "Failed to convert binary string to bytes: " + err,
102 "binary_serializer"});
103 }
104
105 return kcenon::common::ok(std::move(arr));
106 }
107 catch (const std::bad_alloc&)
108 {
109 return kcenon::common::Result<std::vector<uint8_t>>(
110 kcenon::common::error_info{
111 -2,
112 "Memory allocation failed during binary serialization",
113 "binary_serializer"});
114 }
115 catch (const std::exception& e)
116 {
117 return kcenon::common::Result<std::vector<uint8_t>>(
118 kcenon::common::error_info{
119 -1,
120 std::string("Binary serialization failed: ") + e.what(),
121 "binary_serializer"});
122 }
123}
