118template<typename T>
119concept SimdPolicy = requires(const T& policy, const float* data, size_t count) {
120 { T::name() } -> std::convertible_to<std::string_view>;
121 { T::simd_width } -> std::convertible_to<size_t>;
122 { policy.sum_floats(data, count) } -> std::same_as<float>;
123 { policy.min_float(data, count) } -> std::same_as<float>;
124 { policy.max_float(data, count) } -> std::same_as<float>;
125};

137struct scalar_simd_policy {
138 static constexpr std::string_view name() noexcept { return "scalar"; }
139 static constexpr size_t simd_width = 1;
140
141 [[nodiscard]] float sum_floats(const float* data, size_t count) const noexcept {
142 float sum = 0.0f;
143 for (size_t i = 0; i < count; ++i) {
144 sum += data[i];
145 }
146 return sum;
147 }
148
149 [[nodiscard]] float min_float(const float* data, size_t count) const noexcept {
150 if (count == 0) return std::numeric_limits<float>::max();
151 float min_val = data[0];
152 for (size_t i = 1; i < count; ++i) {
153 if (data[i] < min_val) {
154 min_val = data[i];
155 }
156 }
157 return min_val;
158 }
159
160 [[nodiscard]] float max_float(const float* data, size_t count) const noexcept {
161 if (count == 0) return std::numeric_limits<float>::lowest();
162 float max_val = data[0];
163 for (size_t i = 1; i < count; ++i) {
164 if (data[i] > max_val) {
165 max_val = data[i];
166 }
167 }
168 return max_val;
169 }
170
171 [[nodiscard]] double sum_doubles(const double* data, size_t count) const noexcept {
172 double sum = 0.0;
173 for (size_t i = 0; i < count; ++i) {
174 sum += data[i];
175 }
176 return sum;
177 }
178};

184#if defined(CONTAINER_HAS_SSE42) || defined(CONTAINER_HAS_SSE2)
188struct sse_simd_policy {
189 static constexpr std::string_view name() noexcept { return "sse"; }
190 static constexpr size_t simd_width = 4;
191
192 [[nodiscard]]
193#if defined(__GNUC__) || defined(__clang__)
194 __attribute__((target("sse3")))
195#endif
196 float sum_floats(const float* data, size_t count) const noexcept {
197 __m128 sum_vec = _mm_setzero_ps();
198 size_t simd_end = count - (count % 4);
199
200 for (size_t i = 0; i < simd_end; i += 4) {
201 __m128 vec = _mm_loadu_ps(&data[i]);
202 sum_vec = _mm_add_ps(sum_vec, vec);
203 }
204
205 // Horizontal sum using hadd
206 sum_vec = _mm_hadd_ps(sum_vec, sum_vec);
207 sum_vec = _mm_hadd_ps(sum_vec, sum_vec);
208
209 float sum = _mm_cvtss_f32(sum_vec);
210
211 // Handle remaining elements
212 for (size_t i = simd_end; i < count; ++i) {
213 sum += data[i];
214 }
215
216 return sum;
217 }
218
219 [[nodiscard]]
220#if defined(__GNUC__) || defined(__clang__)
221 __attribute__((target("sse2")))
222#endif
223 float min_float(const float* data, size_t count) const noexcept {
224 if (count == 0) return std::numeric_limits<float>::max();
225
226 __m128 min_vec = _mm_set1_ps(std::numeric_limits<float>::max());
227 size_t simd_end = count - (count % 4);
228
229 for (size_t i = 0; i < simd_end; i += 4) {
230 __m128 vec = _mm_loadu_ps(&data[i]);
231 min_vec = _mm_min_ps(min_vec, vec);
232 }
233
234 // Extract minimum from vector
235 alignas(16) float result[4];
236 _mm_store_ps(result, min_vec);
237 float min_val = result[0];
238 for (int i = 1; i < 4; ++i) {
239 if (result[i] < min_val) min_val = result[i];
240 }
241
242 // Handle remaining elements
243 for (size_t i = simd_end; i < count; ++i) {
244 if (data[i] < min_val) min_val = data[i];
245 }
246
247 return min_val;
248 }
249
250 [[nodiscard]]
251#if defined(__GNUC__) || defined(__clang__)
252 __attribute__((target("sse2")))
253#endif
254 float max_float(const float* data, size_t count) const noexcept {
255 if (count == 0) return std::numeric_limits<float>::lowest();
256
257 __m128 max_vec = _mm_set1_ps(std::numeric_limits<float>::lowest());
258 size_t simd_end = count - (count % 4);
259
260 for (size_t i = 0; i < simd_end; i += 4) {
261 __m128 vec = _mm_loadu_ps(&data[i]);
262 max_vec = _mm_max_ps(max_vec, vec);
263 }
264
265 // Extract maximum from vector
266 alignas(16) float result[4];
267 _mm_store_ps(result, max_vec);
268 float max_val = result[0];
269 for (int i = 1; i < 4; ++i) {
270 if (result[i] > max_val) max_val = result[i];
271 }
272
273 // Handle remaining elements
274 for (size_t i = simd_end; i < count; ++i) {
275 if (data[i] > max_val) max_val = data[i];
276 }
277
278 return max_val;
279 }

281 [[nodiscard]] double sum_doubles(const double* data, size_t count) const noexcept {
282 // Use scalar for doubles in SSE (simpler)
283 double sum = 0.0;
284 for (size_t i = 0; i < count; ++i) {
285 sum += data[i];
286 }
287 return sum;
288 }

516#if defined(CONTAINER_HAS_ARM_NEON)
520struct neon_simd_policy {
521 static constexpr std::string_view name() noexcept { return "neon"; }
522 static constexpr size_t simd_width = 4;
523
524 [[nodiscard]] float sum_floats(const float* data, size_t count) const noexcept {
525 float32x4_t sum_vec = vdupq_n_f32(0.0f);
526 size_t simd_end = count - (count % 4);
527
528 for (size_t i = 0; i < simd_end; i += 4) {
529 float32x4_t vec = vld1q_f32(&data[i]);
530 sum_vec = vaddq_f32(sum_vec, vec);
531 }
532
533 // Horizontal sum
534 float32x2_t sum_low = vget_low_f32(sum_vec);
535 float32x2_t sum_high = vget_high_f32(sum_vec);
536 float32x2_t sum_pair = vadd_f32(sum_low, sum_high);
537 float sum = vget_lane_f32(sum_pair, 0) + vget_lane_f32(sum_pair, 1);
538
539 for (size_t i = simd_end; i < count; ++i) {
540 sum += data[i];
541 }
542
543 return sum;
544 }
545
546 [[nodiscard]] float min_float(const float* data, size_t count) const noexcept {
547 if (count == 0) return std::numeric_limits<float>::max();
548
549 float32x4_t min_vec = vdupq_n_f32(std::numeric_limits<float>::max());
550 size_t simd_end = count - (count % 4);
551
552 for (size_t i = 0; i < simd_end; i += 4) {
553 float32x4_t vec = vld1q_f32(&data[i]);
554 min_vec = vminq_f32(min_vec, vec);
555 }
556
557 // Extract minimum from vector
558 float result[4];
559 vst1q_f32(result, min_vec);
