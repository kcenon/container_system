140 void demonstrate_multithreaded_operations() {
141 std::cout << "\n--- Multi-threaded Operations ---" << std::endl;
142
143 const int num_producers = 2;
144 const int num_consumers = 2;
145 const int items_per_producer = 500;
146
147 // Shared queue simulation using vector with mutex
148 std::vector<std::shared_ptr<value_container>> shared_queue;
149 std::mutex queue_mutex;
150 std::condition_variable cv;
151 std::atomic<bool> producers_done{false};
152
153 // Producer threads
154 std::vector<std::thread> producer_threads;
155 for (int p = 0; p < num_producers; ++p) {
156 producer_threads.emplace_back([&, p]() {
157 std::random_device rd;
158 std::mt19937 gen(rd());
159 std::uniform_int_distribution<> dis(1, 1000);
160
161 for (int i = 0; i < items_per_producer; ++i) {
162 // Create container using set_value API
163 auto container = std::make_shared<value_container>();
164 container->set_source("producer_" + std::to_string(p), "thread_" + std::to_string(p));
165 container->set_target("consumer_pool", "any_available");
166 container->set_message_type("work_item");
167
168 container->set("producer_id", static_cast<int32_t>(p));
169 container->set("item_id", static_cast<int32_t>(i));
170 container->set("random_value", static_cast<int32_t>(dis(gen)));
171 container->set("timestamp", static_cast<int64_t>(
172 std::chrono::duration_cast<std::chrono::milliseconds>(
173 std::chrono::system_clock::now().time_since_epoch()).count()));
174
175 // Add to queue
176 {
177 std::lock_guard<std::mutex> lock(queue_mutex);
178 shared_queue.push_back(container);
179 }
180 cv.notify_one();
181
182 stats_.created++;
183
184 // Small delay to simulate work
185 std::this_thread::sleep_for(std::chrono::microseconds(100));
186 }
187 });
188 }
189
190 // Consumer threads
191 std::vector<std::thread> consumer_threads;
192 for (int c = 0; c < num_consumers; ++c) {
193 consumer_threads.emplace_back([&, c]() {
194 int items_processed = 0;
195
196 while (running_) {
197 std::shared_ptr<value_container> container;
198
199 // Get item from queue
200 {
201 std::unique_lock<std::mutex> lock(queue_mutex);
202 cv.wait(lock, [&]() { return !shared_queue.empty() || producers_done.load(); });
203
204 if (!shared_queue.empty()) {
205 container = shared_queue.back();
206 shared_queue.pop_back();
207 } else if (producers_done.load()) {
208 break;
209 }
210 }
211
212 if (container) {
213 // Process container (serialize/deserialize simulation)
214 std::string serialized = container->serialize_string(value_container::serialization_format::binary).value();
215 processed_bytes_ += serialized.size();
216
217 auto processed = std::make_shared<value_container>(serialized);
218 items_processed++;
219 processed_containers_++;
220 stats_.serialized++;
221 stats_.deserialized++;
222
223 // Simulate processing time
224 std::this_thread::sleep_for(std::chrono::microseconds(50));
225 }
226 }
227
228 std::cout << " Consumer " << c << " processed " << items_processed << " items" << std::endl;
229 });
230 }
231
232 // Wait for producers to finish
233 for (auto& thread : producer_threads) {
234 thread.join();
235 }
236
237 producers_done = true;
238 cv.notify_all();
239
240 // Wait for consumers to finish
241 for (auto& thread : consumer_threads) {
242 thread.join();
243 }
244
245 std::cout << "Multi-threaded processing completed:" << std::endl;
246 std::cout << " Total items processed: " << processed_containers_.load() << std::endl;
247 std::cout << " Total bytes processed: " << processed_bytes_.load() << std::endl;
248 std::cout << " Remaining in queue: " << shared_queue.size() << std::endl;
249 }

294 // Scenario 1: High-frequency small messages
295 std::cout << "Scenario 1: High-frequency small messages" << std::endl;
296
297 const int small_message_count = 10000;
298 auto start_time = std::chrono::high_resolution_clock::now();
299
300 for (int i = 0; i < small_message_count; ++i) {
301 auto container = std::make_shared<value_container>();
302 container->set_source("high_freq_client", "session_" + std::to_string(i % 100));
303 container->set_target("high_freq_server", "handler");
304 container->set_message_type("ping");
305
306 container->set("sequence", static_cast<int32_t>(i));
307 container->set("timestamp", static_cast<int64_t>(
308 std::chrono::duration_cast<std::chrono::microseconds>(
309 std::chrono::system_clock::now().time_since_epoch()).count()));
310
311 // Quick serialization test
312 container->serialize_string(value_container::serialization_format::binary).value();
313 }
314
315 auto end_time = std::chrono::high_resolution_clock::now();
316 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
317
318 double rate = (small_message_count * 1000000.0) / duration.count();
319 std::cout << " Rate: " << std::fixed << std::setprecision(2) << rate << " messages/second" << std::endl;
320
321 // Scenario 2: Low-frequency large messages
322 std::cout << "Scenario 2: Low-frequency large messages" << std::endl;
323
324 const int large_message_count = 100;
325 start_time = std::chrono::high_resolution_clock::now();
326
327 for (int i = 0; i < large_message_count; ++i) {
328 auto container = std::make_shared<value_container>();
329 container->set_source("large_msg_client", "upload_session");
330 container->set_target("large_msg_server", "file_handler");
331 container->set_message_type("file_upload");

335 container->set("file_content", file_data);

341 processed_bytes_ += serialized.size();

345 duration = std::chrono::duration_cast<std::chrono::milliseconds>(end_time - start_time);
346
347 double large_rate = (large_message_count * 1000.0) / duration.count();
348 std::cout << " Rate: " << std::fixed << std::setprecision(2) << large_rate << " large messages/second" << std::endl;
349 std::cout << " Data processed: " << (processed_bytes_.load() / 1024 / 1024) << " MB" << std::endl;
350

361

364 std::cout << "Containers created: " << stats_.created.load() << std::endl;
365 std::cout << "Serializations: " << stats_.serialized.load() << std::endl;
