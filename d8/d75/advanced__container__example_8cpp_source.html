217
218 const int num_producers = 2;
219 const int num_consumers = 2;
220 const int items_per_producer = 500;

222 // Shared queue simulation using vector with mutex
223 std::vector<std::shared_ptr<value_container>> shared_queue;
224 std::mutex queue_mutex;
225 std::condition_variable cv;
226 std::atomic<bool> producers_done{false};
227
228 // Producer threads
229 std::vector<std::thread> producer_threads;
230 for (int p = 0; p < num_producers; ++p) {
231 producer_threads.emplace_back([&, p]() {
232 std::random_device rd;
233 std::mt19937 gen(rd());
234 std::uniform_int_distribution<> dis(1, 1000);
235
236 for (int i = 0; i < items_per_producer; ++i) {
237 // Create container
238#ifdef HAS_MESSAGING_FEATURES
239 auto container = integration::messaging_container_builder()
240 .source("producer_" + std::to_string(p), "thread_" + std::to_string(p))
241 .target("consumer_pool", "any_available")
242 .message_type("work_item")
243 .add_value("producer_id", p)
244 .add_value("item_id", i)
245 .add_value("random_value", dis(gen))
246 .add_value("timestamp", std::chrono::duration_cast<std::chrono::milliseconds>(
247 std::chrono::system_clock::now().time_since_epoch()).count())
248 .optimize_for_speed()
249 .build();
250#else
251 auto container = std::make_shared<value_container>();
252 container->set_source("producer_" + std::to_string(p), "thread_" + std::to_string(p));
253 container->set_target("consumer_pool", "any_available");
254 container->set_message_type("work_item");
255 std::string prod_id_key = "producer_id";
256 container->add(std::make_shared<int_value>(prod_id_key, p));
257 std::string item_id_key = "item_id";
258 container->add(std::make_shared<int_value>(item_id_key, i));
259 std::string rand_key = "random_value";
260 container->add(std::make_shared<int_value>(rand_key, dis(gen)));
261 std::string ts_key = "timestamp";
262 container->add(std::make_shared<long_value>(ts_key,
263 std::chrono::duration_cast<std::chrono::milliseconds>(
264 std::chrono::system_clock::now().time_since_epoch()).count()));
265#endif
266
267 // Add to queue
268 {
269 std::lock_guard<std::mutex> lock(queue_mutex);
270 shared_queue.push_back(container);
271 }
272 cv.notify_one();
273
274 stats_.created++;
275
276 // Small delay to simulate work
277 std::this_thread::sleep_for(std::chrono::microseconds(100));
278 }
279 });
280 }
281
282 // Consumer threads
283 std::vector<std::thread> consumer_threads;
284 for (int c = 0; c < num_consumers; ++c) {
285 consumer_threads.emplace_back([&, c]() {
286 int items_processed = 0;
287
288 while (running_) {
289 std::shared_ptr<value_container> container;
290
291 // Get item from queue
292 {
293 std::unique_lock<std::mutex> lock(queue_mutex);
294 cv.wait(lock, [&]() { return !shared_queue.empty() || producers_done.load(); });
295
296 if (!shared_queue.empty()) {
297 container = shared_queue.back();
298 shared_queue.pop_back();
299 } else if (producers_done.load()) {
300 break;
301 }
302 }
303
304 if (container) {
305 // Process container (serialize/deserialize simulation)
306 std::string serialized = container->serialize();
307 processed_bytes_ += serialized.size();
308
309 auto processed = std::make_shared<value_container>(serialized);
310 items_processed++;
311 processed_containers_++;
312 stats_.serialized++;
313 stats_.deserialized++;
314
315 // Simulate processing time
316 std::this_thread::sleep_for(std::chrono::microseconds(50));
317 }
318 }
319
320 std::cout << " Consumer " << c << " processed " << items_processed << " items" << std::endl;
321 });
322 }
323
324 // Wait for producers to finish
325 for (auto& thread : producer_threads) {
326 thread.join();
327 }
328
329 producers_done = true;
330 cv.notify_all();
331
332 // Wait for consumers to finish
333 for (auto& thread : consumer_threads) {
334 thread.join();
335 }
336
337 std::cout << "Multi-threaded processing completed:" << std::endl;
338 std::cout << " Total items processed: " << processed_containers_.load() << std::endl;
339 std::cout << " Total bytes processed: " << processed_bytes_.load() << std::endl;
