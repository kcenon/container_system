485 template<concepts::ValueVariantType T>
486 [[nodiscard]] std::optional<T> get(std::string_view key) const noexcept {
487 auto snap = std::atomic_load_explicit(&snapshot_, std::memory_order_acquire);
488 if (!snap) {
489 return std::nullopt;
490 }
491
492 auto it = snap->find(std::string(key));
493 if (it != snap->end()) {
494 return it->second.get<T>();

496 return std::nullopt;
497 }

507 [[nodiscard]] bool contains(std::string_view key) const noexcept {
508 auto snap = std::atomic_load_explicit(&snapshot_, std::memory_order_acquire);
509 if (!snap) {
510 return false;
511 }
512 return snap->find(std::string(key)) != snap->end();
513 }

522 [[nodiscard]] size_t size() const noexcept {

545 [[nodiscard]] std::vector<std::string> keys() const {
546 auto snap = std::atomic_load_explicit(&snapshot_, std::memory_order_acquire);
547 std::vector<std::string> result;
548 if (snap) {
549 result.reserve(snap->size());
550 for (const auto& [key, val] : *snap) {

588 void refresh() {
589 auto new_snapshot = std::make_shared<snapshot_type>();
590 container_->for_each([&new_snapshot](const std::string& key, const value& val) {
591 new_snapshot->emplace(key, val);
592 });
593 std::atomic_store_explicit(&snapshot_,
594 snapshot_ptr(new_snapshot),
595 std::memory_order_release);
596 refresh_count_.fetch_add(1, std::memory_order_relaxed);
597 }
598
603 [[nodiscard]] size_t refresh_count() const noexcept {
604 return refresh_count_.load(std::memory_order_relaxed);
605 }

612 return container_;
613 }

680 void stop() {
681 {
682 std::lock_guard<std::mutex> lock(mutex_);
683 if (!running_) {
684 return;
685 }
686 running_ = false;
687 }
688 cv_.notify_one();
689 if (refresh_thread_.joinable()) {
690 refresh_thread_.join();
691 }
692 }

801 private:
802 void refresh_loop() {
803 while (true) {
804 std::unique_lock<std::mutex> lock(mutex_);
805 if (cv_.wait_for(lock, refresh_interval_, [this] { return !running_; })) {
806 break; // Stopped
807 }
808 lock.unlock();
809 reader_->refresh();
810 }
811 }
