96std::string value_store::serialize_impl() const {
97 read_count_.fetch_add(1, std::memory_order_relaxed);
98
99 std::string result = "{";
100 bool first = true;
101
102 for (const auto& [key, val] : values_) {
103 if (!first) {
104 result += ",";
105 }
106 first = false;
107
108 // Escape key for JSON
109 result += "\"";
110 for (char c : key) {
111 switch (c) {
112 case '"': result += "\\\""; break;
113 case '\\': result += "\\\\"; break;
114 case '\n': result += "\\n"; break;
115 case '\r': result += "\\r"; break;
116 case '\t': result += "\\t"; break;
117 default: result += c;
118 }
119 }
120 result += "\":";
121 result += val.to_json();
122 }
123
124 result += "}";
125 return result;
126}

134std::vector<uint8_t> value_store::serialize_binary_impl() const {
135 read_count_.fetch_add(1, std::memory_order_relaxed);
136
137 std::vector<uint8_t> result;
138
139 // Version byte for future compatibility
140 constexpr uint8_t version = 1;
141 result.push_back(version);
142
143 // Header: number of entries (4 bytes)
144 uint32_t count = static_cast<uint32_t>(values_.size());
145 result.insert(result.end(),
146 reinterpret_cast<const uint8_t*>(&count),
147 reinterpret_cast<const uint8_t*>(&count) + sizeof(count));
148
149 // Serialize each key-value pair
150 for (const auto& [key, val] : values_) {
151 // Key length and key
152 uint32_t key_len = static_cast<uint32_t>(key.size());
153 result.insert(result.end(),
154 reinterpret_cast<const uint8_t*>(&key_len),
155 reinterpret_cast<const uint8_t*>(&key_len) + sizeof(key_len));
156 result.insert(result.end(), key.begin(), key.end());
157
158 // Value serialization
159 auto value_data = val.serialize();
160 uint32_t value_len = static_cast<uint32_t>(value_data.size());
161 result.insert(result.end(),
162 reinterpret_cast<const uint8_t*>(&value_len),
163 reinterpret_cast<const uint8_t*>(&value_len) + sizeof(value_len));
164 result.insert(result.end(), value_data.begin(), value_data.end());
165 }
166
167 return result;
168}

177std::unique_ptr<value_store> value_store::deserialize_binary(const std::vector<uint8_t>& binary_data) {
178 auto store = std::make_unique<value_store>();
179
180 if (binary_data.size() < 1 + sizeof(uint32_t)) {
181 throw std::runtime_error("value_store::deserialize_binary() - invalid data: too small");
182 }
183
184 size_t offset = 0;
185
186 // Read version byte
187 uint8_t version = binary_data[offset++];
188 if (version != 1) {
189 throw std::runtime_error("value_store::deserialize_binary() - unsupported version: "
190 + std::to_string(version));
191 }
192
193 // Read number of entries
194 uint32_t count;
195 std::memcpy(&count, binary_data.data() + offset, sizeof(count));
196 offset += sizeof(count);
197
198 // Read each key-value pair
199 for (uint32_t i = 0; i < count; ++i) {
200 if (offset + sizeof(uint32_t) > binary_data.size()) {
201 throw std::runtime_error("value_store::deserialize_binary() - truncated data at entry "
202 + std::to_string(i));
203 }
204
205 // Read key length
206 uint32_t key_len;
207 std::memcpy(&key_len, binary_data.data() + offset, sizeof(key_len));
208 offset += sizeof(key_len);
209
210 if (offset + key_len + sizeof(uint32_t) > binary_data.size()) {
211 throw std::runtime_error("value_store::deserialize_binary() - truncated key data");
212 }
213
214 // Read key
215 std::string key(binary_data.begin() + offset,
216 binary_data.begin() + offset + key_len);
217 offset += key_len;
218
219 // Read value length
220 uint32_t value_len;
221 std::memcpy(&value_len, binary_data.data() + offset, sizeof(value_len));
222 offset += sizeof(value_len);
223
224 if (offset + value_len > binary_data.size()) {
225 throw std::runtime_error("value_store::deserialize_binary() - truncated value data");
226 }
227
228 // Deserialize value
229 std::vector<uint8_t> value_data(binary_data.begin() + offset,
230 binary_data.begin() + offset + value_len);
231 offset += value_len;
232
233 auto value_opt = value::deserialize(value_data);
234 if (value_opt) {
235 store->values_[key] = std::move(*value_opt);
236 } else {
237 throw std::runtime_error("value_store::deserialize_binary() - failed to deserialize value for key: "
238 + key);
239 }
240 }
241
242 return store;
243}
