136 template<typename T>
137 inline value make_numeric_value(std::string_view name, T value) {
138 static_assert(std::is_arithmetic_v<T>, "T must be an arithmetic type");
139
140 if constexpr (std::is_same_v<T, bool>) {
141 return make_bool_value(name, value);
142 } else if constexpr (std::is_same_v<T, int16_t> || std::is_same_v<T, short>) {
143 return make_short_value(name, static_cast<int16_t>(value));
144 } else if constexpr (std::is_same_v<T, uint16_t> || std::is_same_v<T, unsigned short>) {
145 return make_ushort_value(name, static_cast<uint16_t>(value));
146 } else if constexpr (std::is_same_v<T, int32_t> || std::is_same_v<T, int>) {
147 return make_int_value(name, static_cast<int32_t>(value));
148 } else if constexpr (std::is_same_v<T, uint32_t> || std::is_same_v<T, unsigned int>) {
149 return make_uint_value(name, static_cast<uint32_t>(value));
150 } else if constexpr (std::is_same_v<T, int64_t> ||
151 std::is_same_v<T, long> ||
152 std::is_same_v<T, long long>) {
153 return make_long_value(name, static_cast<int64_t>(value));
154 } else if constexpr (std::is_same_v<T, uint64_t> ||
155 std::is_same_v<T, unsigned long> ||
156 std::is_same_v<T, unsigned long long>) {
157 return make_ulong_value(name, static_cast<uint64_t>(value));
158 } else if constexpr (std::is_same_v<T, float>) {
159 return make_float_value(name, value);
160 } else if constexpr (std::is_same_v<T, double>) {
161 return make_double_value(name, value);
162 } else {
163 // Fallback: use the closest integer type
164 if constexpr (sizeof(T) <= sizeof(int32_t)) {
165 return make_int_value(name, static_cast<int32_t>(value));
166 } else {
167 return make_long_value(name, static_cast<int64_t>(value));
168 }
169 }
170 }

305 inline value clone_with_name(const value& original,
306 std::string_view new_name) {
307 // Serialize and deserialize to create a deep copy, then change name
308 auto serialized = original.serialize();
309 auto cloned = value::deserialize(serialized);
310 if (!cloned) {
311 throw std::runtime_error("Failed to clone value");
312 }
313
314 // Create new value with same data but different name
315 return value(new_name, cloned->type(), serialized);
316 }
