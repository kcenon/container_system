380 inline value clone_with_name(const value& original,
381 std::string_view new_name) {
382 // Serialize and deserialize to create a deep copy, then change name
383 auto serialized = original.serialize();
384 auto cloned = value::deserialize(serialized);
385 if (!cloned) {
386 throw std::runtime_error("Failed to clone value");
387 }
388
389 // Create new value with same data but different name
390 return value(new_name, cloned->type(), serialized);
391 }

403 inline std::string_view type_name(const value& value) {
404 switch (value.type()) {
405 case value_types::null_value: return "null";
406 case value_types::bool_value: return "bool";
407 case value_types::short_value: return "short";
408 case value_types::ushort_value: return "ushort";
409 case value_types::int_value: return "int";
410 case value_types::uint_value: return "uint";
411 case value_types::long_value: return "long";
412 case value_types::ulong_value: return "ulong";
413 case value_types::llong_value: return "llong";
414 case value_types::ullong_value: return "ullong";
415 case value_types::float_value: return "float";
416 case value_types::double_value: return "double";
417 case value_types::bytes_value: return "bytes";
418 case value_types::string_value: return "string";
419 case value_types::container_value: return "container";
420 case value_types::array_value: return "array";
421 default: return "unknown";
