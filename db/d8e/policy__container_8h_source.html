359 [[nodiscard]] kcenon::common::Result<T> get(std::string_view key) const noexcept {
360 std::shared_lock lock(mutex_);
361 auto result = storage_.get(key);
362 if (!result) {
363 return kcenon::common::Result<T>(
364 kcenon::common::error_info{
365 error_codes::key_not_found,
366 error_codes::make_message(error_codes::key_not_found, key),
367 "container_system"});
368 }
369 if (auto* ptr = std::get_if<T>(&result->data)) {
370 return kcenon::common::ok(*ptr);
371 }
372 return kcenon::common::Result<T>(
373 kcenon::common::error_info{
374 error_codes::type_mismatch,
375 error_codes::make_message(error_codes::type_mismatch, key),
376 "container_system"});
377 }

386 template<typename T>
387 [[nodiscard]] kcenon::common::VoidResult set_result(std::string_view key, T&& data_val) noexcept {
388 try {
389 if (key.empty()) {
390 return kcenon::common::VoidResult(
391 kcenon::common::error_info{
392 error_codes::empty_key,
393 error_codes::make_message(error_codes::empty_key),
394 "container_system"});
395 }
396
397 optimized_value val;
398 val.name = std::string(key);
399 val.data = std::forward<T>(data_val);
400 val.type = static_cast<value_types>(val.data.index());
401
402 std::unique_lock lock(mutex_);
403 storage_.set(key, std::move(val));
404 return kcenon::common::ok();
405 } catch (const std::bad_alloc&) {
406 return kcenon::common::VoidResult(
407 kcenon::common::error_info{
408 error_codes::memory_allocation_failed,
409 error_codes::make_message(error_codes::memory_allocation_failed, key),
410 "container_system"});
411 } catch (const std::exception& e) {
412 return kcenon::common::VoidResult(
413 kcenon::common::error_info{
414 error_codes::invalid_value,
415 std::string("Failed to set value: ") + e.what(),
416 "container_system"});
417 }
418 }

444 basic_value_container& bulk_insert(std::span<const optimized_value> values,
445 size_t reserve_hint = 0) {
446 std::unique_lock lock(mutex_);
447 if (reserve_hint > 0) {

461 [[nodiscard]] std::vector<std::optional<optimized_value>> get_batch(
462 std::span<const std::string_view> keys) const noexcept {
463 std::shared_lock lock(mutex_);
464 std::vector<std::optional<optimized_value>> results;
465 results.reserve(keys.size());
466 for (const auto& key : keys) {
467 results.push_back(storage_.get(key));
468 }
469 return results;
470 }
471
477 [[nodiscard]] std::vector<bool> contains_batch(
478 std::span<const std::string_view> keys) const noexcept {
479 std::shared_lock lock(mutex_);
480 std::vector<bool> results;
481 results.reserve(keys.size());
482 for (const auto& key : keys) {
483 results.push_back(storage_.contains(key));
484 }
485 return results;
486 }
487
493 size_t remove_batch(std::span<const std::string_view> keys) {
494 std::unique_lock lock(mutex_);
495 size_t count = 0;
496 for (const auto& key : keys) {
497 if (storage_.remove(key)) {
498 ++count;
499 }
500 }
501 return count;
502 }
