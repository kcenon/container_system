64 inline std::string json_escape(std::string_view input)
65 {
66 std::string result;
67 result.reserve(input.size() + input.size() / 8);
68
69 for (char c : input)
70 {
71 switch (c)
72 {
73 case '"':
74 result += "\\\"";
75 break;
76 case '\\':
77 result += "\\\\";
78 break;
79 case '\b':
80 result += "\\b";
81 break;
82 case '\f':
83 result += "\\f";
84 break;
85 case '\n':
86 result += "\\n";
87 break;
88 case '\r':
89 result += "\\r";
90 break;
91 case '\t':
92 result += "\\t";
93 break;
94 default:
95 if (static_cast<unsigned char>(c) < 0x20)
96 {
97 // Control character - use \uXXXX format
98 char buf[8];
99 std::snprintf(buf, sizeof(buf), "\\u%04x",
100 static_cast<unsigned char>(c));
101 result += buf;
102 }
103 else
104 {
105 result += c;
106 }
107 break;
108 }
109 }
110
111 return result;
112 }

119 inline std::string xml_encode(std::string_view input)
120 {
121 std::string result;
122 result.reserve(input.size() + input.size() / 8);
123
124 for (char c : input)
125 {
126 switch (c)
127 {
128 case '&':
129 result += "&amp;";
130 break;
131 case '<':
132 result += "&lt;";
133 break;
134 case '>':
135 result += "&gt;";
136 break;
137 case '"':
138 result += "&quot;";
139 break;
140 case '\'':
141 result += "&apos;";
142 break;
143 default:
144 if (static_cast<unsigned char>(c) < 0x20
145 && c != '\t' && c != '\n' && c != '\r')
146 {
147 // Control character - use numeric character reference
148 char buf[16];
149 std::snprintf(buf, sizeof(buf), "&#x%02x;",
150 static_cast<unsigned char>(c));
151 result += buf;
152 }
153 else
154 {
155 result += c;
156 }
157 break;
158 }
159 }
160
161 return result;
162 }

167 inline std::string to_string(const value_variant& var, value_types type)
168 {
169 switch (type)
170 {
171 case value_types::null_value:
172 return "";
173 case value_types::bool_value:
174 return std::get<bool>(var) ? "true" : "false";
175 case value_types::short_value:
176 return std::to_string(std::get<short>(var));
177 case value_types::ushort_value:
178 return std::to_string(std::get<unsigned short>(var));
179 case value_types::int_value:
180 return std::to_string(std::get<int>(var));
181 case value_types::uint_value:
182 return std::to_string(std::get<unsigned int>(var));
183 case value_types::long_value:
184 return std::to_string(std::get<long>(var));
185 case value_types::ulong_value:
186 return std::to_string(std::get<unsigned long>(var));
187 case value_types::llong_value:
188 return std::to_string(std::get<long long>(var));
189 case value_types::ullong_value:
190 return std::to_string(std::get<unsigned long long>(var));
191 case value_types::float_value:
192 return std::to_string(std::get<float>(var));
193 case value_types::double_value:
194 return std::to_string(std::get<double>(var));
195 case value_types::string_value:
196 return std::get<std::string>(var);
197 default:
198 return "";
199 }
200 }
