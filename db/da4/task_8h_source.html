109 template<typename Promise>
110 std::coroutine_handle<> await_suspend(
111 std::coroutine_handle<Promise> handle) noexcept
112 {
113 auto& promise = handle.promise();
114 // Read continuation BEFORE marking as complete to avoid
115 // data race with destructor. Once completed_ is true,
116 // the task owner may destroy the coroutine frame.
117 auto continuation = promise.continuation_;
118 // Mark as completed with release semantics to ensure
119 // all prior writes are visible to threads checking done()
120 promise.completed_.store(true, std::memory_order_release);
121 if (continuation)
122 {
123 return continuation;
124 }
125 return std::noop_coroutine();
126 }
127
128 void await_resume() noexcept {}
129 };

134 [[nodiscard]] final_awaiter final_suspend() noexcept
135 {
136 return {};
137 }
