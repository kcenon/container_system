28 variant_value_v2 value_bridge::to_modern(const value& legacy) {
29 try {
30 // Track conversion attempt - increment at start, decrement on failure
31 {
32 std::lock_guard lock(stats_mutex_);
33 stats_.successful_conversions++;
34 }
35
36 value_types type = legacy.type();
37
38 switch (type) {
39 case value_types::null_value:
40 return variant_value_v2(legacy.name());
41
42 case value_types::bool_value:
43 return variant_value_v2(legacy.name(), legacy.to_boolean());
44
45 case value_types::short_value:
46 return variant_value_v2(legacy.name(), static_cast<int16_t>(legacy.to_short()));
47
48 case value_types::ushort_value:
49 return variant_value_v2(legacy.name(), static_cast<uint16_t>(legacy.to_ushort()));
50
51 case value_types::int_value:
52 return variant_value_v2(legacy.name(), static_cast<int32_t>(legacy.to_int()));
53
54 case value_types::uint_value:
55 return variant_value_v2(legacy.name(), static_cast<uint32_t>(legacy.to_uint()));
56
57 case value_types::long_value:
58 case value_types::llong_value:
59 return variant_value_v2(legacy.name(), static_cast<int64_t>(legacy.to_llong()));
60
61 case value_types::ulong_value:
62 case value_types::ullong_value:
63 return variant_value_v2(legacy.name(), static_cast<uint64_t>(legacy.to_ullong()));
64
65 case value_types::float_value:
66 return variant_value_v2(legacy.name(), legacy.to_float());
67
68 case value_types::double_value:
69 return variant_value_v2(legacy.name(), legacy.to_double());
70
71 case value_types::bytes_value:
72 return convert_bytes_value(legacy);
73
74 case value_types::string_value:
75 return convert_string_value(legacy);
76
77 case value_types::container_value:
78 return convert_container_value(legacy);
79
80 case value_types::array_value:
81 return convert_array_value(legacy);
82
83 default:
84 throw std::runtime_error("Unsupported legacy value type");
85 }
86 } catch (const std::exception& e) {
87 // Decrement success count and track failure
88 {
89 std::lock_guard lock(stats_mutex_);
90 stats_.successful_conversions--;
91 stats_.failed_conversions++;
92 }
93 record_error(std::string("to_modern failed: ") + e.what());
94 throw;
95 }
96 }

115 variant_value_v2 value_bridge::convert_container_value(const value& legacy) {
116 // Container value contains nested values
117 // For now, serialize and deserialize through binary format
118 // Note: serialize() is not const, but we're only reading
119 auto serialized = const_cast<value&>(legacy).serialize();
120 auto bytes = std::vector<uint8_t>(serialized.begin(), serialized.end());
121
122 // Deserialize into thread_safe_container
123 auto container = thread_safe_container::deserialize(bytes);
124 return variant_value_v2(legacy.name(), container);
125 }

127 variant_value_v2 value_bridge::convert_array_value(const value& legacy) {
128 // Cast to array_value to access array-specific methods
129 const auto* arr_value = dynamic_cast<const array_value*>(&legacy);
130 if (!arr_value) {
131 throw std::runtime_error("Expected array_value type");
132 }
133
134 array_variant arr;
135 arr.values.reserve(arr_value->size());
136
137 // Use size() and at() to access elements
138 for (size_t i = 0; i < arr_value->size(); ++i) {
139 auto child = arr_value->at(i);
140 if (child) {
141 auto modern_child = to_modern(child);
142 arr.values.push_back(std::make_shared<variant_value_v2>(std::move(modern_child)));
143 }
144 }
145
146 return variant_value_v2(legacy.name(), std::move(arr));
147 }

215 std::shared_ptr<value> value_bridge::create_legacy_numeric(const variant_value_v2& modern) {
216 value_types type = modern.type();
217
218 switch (type) {
219 case value_types::short_value: {
220 auto val = modern.get<int16_t>();
221 return std::make_shared<short_value>(std::string(modern.name()), *val);
222 }
223 case value_types::ushort_value: {
224 auto val = modern.get<uint16_t>();
225 return std::make_shared<ushort_value>(std::string(modern.name()), *val);
226 }
227 case value_types::int_value: {
228 auto val = modern.get<int32_t>();
229 return std::make_shared<int_value>(std::string(modern.name()), *val);
230 }
231 case value_types::uint_value: {
232 auto val = modern.get<uint32_t>();
233 return std::make_shared<uint_value>(std::string(modern.name()), *val);
234 }
235 case value_types::long_value:
236 case value_types::llong_value: {
237 auto val = modern.get<int64_t>();
238 // On macOS/Linux, long_value enforces 32-bit range despite long being 64-bit
239 // Use llong_value for values outside 32-bit range
240 constexpr int64_t kInt32Min = -2147483648LL;
241 constexpr int64_t kInt32Max = 2147483647LL;
242 bool is_32bit_range = (*val >= kInt32Min && *val <= kInt32Max);
243
244 if (type == value_types::long_value && is_32bit_range) {
245 return std::make_shared<long_value>(std::string(modern.name()), *val);
246 } else {
247 return std::make_shared<llong_value>(std::string(modern.name()), *val);
248 }
249 }
250 case value_types::ulong_value:
251 case value_types::ullong_value: {
252 auto val = modern.get<uint64_t>();
253 // On macOS/Linux, ulong_value enforces 32-bit range despite unsigned long being 64-bit
254 // Use ullong_value for values outside 32-bit range
255 constexpr uint64_t kUInt32Max = 4294967295ULL;
256 bool is_32bit_range = (*val <= kUInt32Max);
257
258 if (type == value_types::ulong_value && is_32bit_range) {
259 return std::make_shared<ulong_value>(std::string(modern.name()), *val);
260 } else {
261 return std::make_shared<ullong_value>(std::string(modern.name()), *val);
262 }
263 }
264 case value_types::float_value: {
265 auto val = modern.get<float>();
266 return std::make_shared<float_value>(std::string(modern.name()), *val);
267 }
268 case value_types::double_value: {
269 auto val = modern.get<double>();
270 return std::make_shared<double_value>(std::string(modern.name()), *val);
271 }
272 default:
273 throw std::runtime_error("Not a numeric type");
274 }
275 }

304 std::shared_ptr<value> value_bridge::create_legacy_array(const variant_value_v2& modern) {
305 auto arr = modern.get<array_variant>();
306 if (!arr) throw std::runtime_error("Type mismatch: expected array");
307
308 // Create empty array first to avoid shared_from_this() issues in constructor
309 auto legacy_array = std::make_shared<array_value>(std::string(modern.name()));
310
311 // Convert and add elements one by one
312 for (const auto& modern_elem : arr->values) {
313 if (modern_elem) {
314 auto legacy_elem = to_legacy(*modern_elem);
315 legacy_array->push_back(legacy_elem);
316 }
317 }
318
319 return legacy_array;
320 }

358 bool value_bridge::verify_round_trip(std::shared_ptr<value> original) {
359 try {
360 // Legacy ‚Üí Modern ‚Üí Legacy
361 auto modern = to_modern(original);
362 auto back_to_legacy = to_legacy(modern);
363
364 // Compare serialized forms (byte-for-byte equality)
365 auto original_serialized = original->serialize();
366 auto result_serialized = back_to_legacy->serialize();
367
368 if (original_serialized != result_serialized) {
369 stats_.round_trip_errors++;
370 record_error("Round-trip mismatch: serialized data differs");
371 return false;
372 }
373
374 return true;
375 } catch (const std::exception& e) {
376 stats_.round_trip_errors++;
377 record_error(std::string("Round-trip failed: ") + e.what());
378 return false;
379 }
380 }
