105 // Create container with various data

110

113 container->set("pi", 3.14159);
114 container->set("success", true);
115
116 // Serialize
117 std::cout << "Serializing container..." << std::endl;
118 std::string serialized_data = container->serialize_string(value_container::serialization_format::binary).value();
119 std::cout << "Serialized size: " << serialized_data.size() << " bytes" << std::endl;
120
121 // Deserialize
122 std::cout << "Deserializing container..." << std::endl;
123 auto new_container = std::make_shared<value_container>(serialized_data);
124
125 std::cout << "Deserialization successful!" << std::endl;
126 std::cout << "Deserialized container:" << std::endl;
127 std::cout << " Source: " << new_container->source_id() << "/" << new_container->source_sub_id() << std::endl;
128 std::cout << " Target: " << new_container->target_id() << "/" << new_container->target_sub_id() << std::endl;
129 std::cout << " Type: " << new_container->message_type() << std::endl;
130
131 // Verify specific values using new get_value API
132 if (auto message_value = new_container->get_value("message")) {
133 if (auto* str = std::get_if<std::string>(&message_value->data)) {
134 std::cout << " Message: " << *str << std::endl;
135 }
136 }
137
138 if (auto count_value = new_container->get_value("count")) {
139 if (auto* val = std::get_if<int32_t>(&count_value->data)) {
140 std::cout << " Count: " << *val << std::endl;
141 }
142 }

207void demonstrate_performance_basics() {
208 std::cout << "\n=== Basic Performance Demonstration ===" << std::endl;
209
210 const int num_operations = 1000;
211
212 // Container creation performance
213 auto start_time = std::chrono::high_resolution_clock::now();
214
215 std::vector<std::shared_ptr<value_container>> containers;
216 containers.reserve(num_operations);
217
218 for (int i = 0; i < num_operations; ++i) {
219 auto container = std::make_shared<value_container>();
220 container->set_source("perf_client", "session_" + std::to_string(i));
221 container->set_target("perf_server", "handler");
222 container->set_message_type("performance_test");
223
224 container->set("index", static_cast<int32_t>(i));
225 container->set("data", std::string("test_data_" + std::to_string(i)));
226
227 containers.push_back(container);
228 }
229
230 auto end_time = std::chrono::high_resolution_clock::now();
231 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
232
233 double containers_per_second = (num_operations * 1000000.0) / duration.count();
234
235 std::cout << "Performance results:" << std::endl;
236 std::cout << " Created " << num_operations << " containers in "
237 << duration.count() << " microseconds" << std::endl;
238 std::cout << " Rate: " << std::fixed << std::setprecision(2)
239 << containers_per_second << " containers/second" << std::endl;
240
241 // Serialization performance
242 start_time = std::chrono::high_resolution_clock::now();
243
244 std::vector<std::string> serialized_data;
245 serialized_data.reserve(num_operations);
246
247 for (const auto& container : containers) {
248 serialized_data.push_back(container->serialize_string(value_container::serialization_format::binary).value());
249 }
250
251 end_time = std::chrono::high_resolution_clock::now();
252 duration = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
253
254 double serializations_per_second = (num_operations * 1000000.0) / duration.count();
255
256 std::cout << " Serialized " << num_operations << " containers in "
257 << duration.count() << " microseconds" << std::endl;
258 std::cout << " Rate: " << std::fixed << std::setprecision(2)
259 << serializations_per_second << " serializations/second" << std::endl;
260
261 // Calculate total data size
262 size_t total_size = 0;
263 for (const auto& data : serialized_data) {
264 total_size += data.size();
265 }
266
267 std::cout << " Total serialized data: " << total_size << " bytes" << std::endl;
268 std::cout << " Average per container: " << (total_size / num_operations) << " bytes" << std::endl;
269}
