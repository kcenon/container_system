213
214 const int num_operations = 1000;
215
216 // Container creation performance
217 auto start_time = std::chrono::high_resolution_clock::now();
218
219 std::vector<std::shared_ptr<value_container>> containers;

222 for (int i = 0; i < num_operations; ++i) {

224 container->set_source("perf_client", "session_" + std::to_string(i));
225 container->set_target("perf_server", "handler");
226 container->set_message_type("performance_test");

228 std::string index_key = "index";
229 container->add(std::make_shared<int_value>(index_key, i));
230 std::string data_key = "data";
231 std::string data_val = "test_data_" + std::to_string(i);
232 container->add(std::make_shared<string_value>(data_key, data_val));
233
234 containers.push_back(container);
235 }
236
237 auto end_time = std::chrono::high_resolution_clock::now();
238 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
239
240 double containers_per_second = (num_operations * 1000000.0) / duration.count();
241
242 std::cout << "Performance results:" << std::endl;
243 std::cout << " Created " << num_operations << " containers in "

245 std::cout << " Rate: " << std::fixed << std::setprecision(2)
