99 std::cout << "Payload: " << *v << "\n";

102
103 if (auto val = container->get_value("is_urgent")) {
104 if (auto* v = std::get_if<bool>(&val->data)) {
105 std::cout << "Is Urgent: " << (*v ? "yes" : "no") << "\n";

107 }
108

116void demonstrate_serialization_roundtrip() {
117 std::cout << "\n=== Serialization Round-trip ===\n";
118
119 // Create source container
120 auto source = std::make_shared<value_container>();
121 source->set_source("sender", "app_1");
122 source->set_target("receiver", "app_2");
123 source->set_message_type("roundtrip_test");
124
125 source->set("int_val", static_cast<int32_t>(42));
126 source->set("double_val", 3.14159);
127 source->set("string_val", std::string("Hello, World!"));
128 source->set("bool_val", true);
129 source->set("long_val", static_cast<int64_t>(9876543210L));
130
131 // Serialize
132 std::string serialized = source->serialize_string(value_container::serialization_format::binary).value();
133 std::cout << "Serialized " << source->size() << " values to " << serialized.size() << " bytes\n";
134
135 // Deserialize
136 auto restored = std::make_shared<value_container>(serialized);
137 std::cout << "Restored container: " << restored->message_type() << "\n";
138
139 // Verify values
140 bool all_match = true;
141
142 if (auto val = restored->get_value("int_val")) {
143 if (auto* v = std::get_if<int32_t>(&val->data)) {
144 if (*v != 42) all_match = false;
145 }
146 }
147
148 if (auto val = restored->get_value("string_val")) {
149 if (auto* v = std::get_if<std::string>(&val->data)) {
150 if (*v != "Hello, World!") all_match = false;
151 }
152 }
153
154 std::cout << "Round-trip verification: " << (all_match ? "PASSED" : "FAILED") << "\n";
155}

174void performance_comparison() {
175 std::cout << "\n=== Performance Comparison ===\n";
176
177 const int iterations = 1000;
178
179 // Test set_value performance
180 auto start = std::chrono::high_resolution_clock::now();
181 for (int i = 0; i < iterations; ++i) {
182 auto container = std::make_shared<value_container>();
183 container->set_message_type("perf_test");
184 container->set("index", static_cast<int32_t>(i));
185 container->set("data", std::string("test_data"));
186 std::string serialized = container->serialize_string(value_container::serialization_format::binary).value();
187 }
188 auto standard_time = std::chrono::high_resolution_clock::now() - start;
189
190 auto standard_ms = std::chrono::duration_cast<std::chrono::milliseconds>(standard_time).count();
191 std::cout << "set_value API: " << standard_ms << " ms for " << iterations << " operations\n";
192
193 if (standard_ms > 0) {
194 double rate = (iterations * 1000.0) / standard_ms;
195 std::cout << "Rate: " << std::fixed << std::setprecision(2) << rate << " containers/second\n";
