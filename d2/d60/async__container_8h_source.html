93 template<typename T>
94 struct async_state
95 {
96 std::function<T()> work_;
97 std::optional<T> result_;
98 std::exception_ptr exception_;
99 std::atomic<bool> ready_{false};
100
101 explicit async_state(std::function<T()> work)
102 : work_(std::move(work)) {}
103
104 // Non-copyable, non-movable
105 async_state(const async_state&) = delete;
106 async_state& operator=(const async_state&) = delete;
107 async_state(async_state&&) = delete;
108 async_state& operator=(async_state&&) = delete;
109 };

117 template<typename T>
118 struct async_awaitable
119 {
120 std::shared_ptr<async_state<T>> state_;
121
122 explicit async_awaitable(std::function<T()> work)
123 : state_(std::make_shared<async_state<T>>(std::move(work))) {}
124
125 async_awaitable(async_awaitable&&) noexcept = default;
126 async_awaitable& operator=(async_awaitable&&) noexcept = default;
127
128 async_awaitable(const async_awaitable&) = delete;
129 async_awaitable& operator=(const async_awaitable&) = delete;
130
131 [[nodiscard]] bool await_ready() const noexcept
132 {
133 return false;
134 }
135
136 void await_suspend(std::coroutine_handle<> handle)
137 {
138 // Copy shared_ptr for thread to ensure state lifetime
139 auto state = state_;
140 std::thread([state, handle]() mutable {
141 try {
142 state->result_.emplace(state->work_());
143 } catch (...) {
144 state->exception_ = std::current_exception();
145 }
146 // Release barrier ensures all writes above are visible
147 // to the thread that reads with acquire semantics
148 state->ready_.store(true, std::memory_order_release);
149 handle.resume();
150 }).detach();
151 }
152
153 T await_resume()
154 {
155 // Acquire barrier synchronizes with the release store above,
156 // ensuring all writes in the worker thread are visible here
157 while (!state_->ready_.load(std::memory_order_acquire)) {
158 // Spin until ready - in practice, handle.resume()
159 // ensures we only get here after ready_ is set
160 }
161 if (state_->exception_) {
162 std::rethrow_exception(state_->exception_);
163 }
164 return std::move(*state_->result_);
165 }
166 };

168 template<typename F>
169 auto make_async_awaitable(F&& func) -> async_awaitable<std::invoke_result_t<F>>
170 {
171 using result_type = std::invoke_result_t<F>;
172 return async_awaitable<result_type>(std::forward<F>(func));
173 }

622 inline task<kcenon::common::VoidResult>
623 async_container::load_async(std::string_view path, progress_callback callback)
624 {
625 using namespace container_module;
626 auto container = container_;
627 std::string path_str(path);
628 auto result = co_await detail::make_async_awaitable(
629 [container, path_str, callback]() -> kcenon::common::VoidResult {
630 std::ifstream file(path_str, std::ios::binary | std::ios::ate);
631 if (!file) {
632 return kcenon::common::VoidResult(
633 kcenon::common::error_info{
634 container_module::error_codes::file_not_found,
635 container_module::error_codes::make_message(
636 container_module::error_codes::file_not_found, path_str),
637 "container_system"});
638 }
639
640 auto size = file.tellg();
641 if (size < 0) {
642 return kcenon::common::VoidResult(
643 kcenon::common::error_info{
644 container_module::error_codes::file_read_error,
645 container_module::error_codes::make_message(
646 container_module::error_codes::file_read_error, path_str),
647 "container_system"});
648 }
649 file.seekg(0, std::ios::beg);
650
651 std::vector<uint8_t> buffer(static_cast<size_t>(size));
652 const size_t chunk_size = 64 * 1024;
653 size_t bytes_read = 0;
654
655 while (bytes_read < static_cast<size_t>(size)) {
656 size_t to_read = std::min(chunk_size,
657 static_cast<size_t>(size) - bytes_read);
658 file.read(reinterpret_cast<char*>(buffer.data() + bytes_read),
659 static_cast<std::streamsize>(to_read));
660 if (!file) {
661 return kcenon::common::VoidResult(
662 kcenon::common::error_info{
663 container_module::error_codes::file_read_error,
664 container_module::error_codes::make_message(
665 container_module::error_codes::file_read_error, path_str),
666 "container_system"});
667 }
668 bytes_read += to_read;
669 if (callback) {
670 callback(bytes_read, static_cast<size_t>(size));
671 }
672 }
673
674 auto deser_result = container->deserialize_result(buffer, false);
675 if (!deser_result.is_ok()) {
676 return deser_result;
677 }
678 return kcenon::common::ok();
679 });
680 co_return result;
681 }

683 inline task<kcenon::common::VoidResult>
684 async_container::save_async(std::string_view path, progress_callback callback)
685 {
686 auto container = container_;
687 std::string path_str(path);
688 auto result = co_await detail::make_async_awaitable(
689 [container, path_str, callback]() -> kcenon::common::VoidResult {
690 auto data_result = container->serialize(value_container::serialization_format::binary);
691 if (!data_result.is_ok()) {
692 return kcenon::common::VoidResult(data_result.error());
693 }
694 const auto& data = data_result.value();
695
696 std::ofstream file(path_str, std::ios::binary);
697 if (!file) {
698 return kcenon::common::VoidResult(
699 kcenon::common::error_info{
700 container_module::error_codes::file_write_error,
701 container_module::error_codes::make_message(
702 container_module::error_codes::file_write_error, path_str),
703 "container_system"});
704 }
705
706 const size_t chunk_size = 64 * 1024;
707 size_t bytes_written = 0;
708 const size_t total_size = data.size();
709
710 while (bytes_written < total_size) {
711 size_t to_write = std::min(chunk_size, total_size - bytes_written);
712 file.write(reinterpret_cast<const char*>(data.data() + bytes_written),
713 static_cast<std::streamsize>(to_write));
714 if (!file) {
715 return kcenon::common::VoidResult(
716 kcenon::common::error_info{
717 container_module::error_codes::file_write_error,
718 container_module::error_codes::make_message(
719 container_module::error_codes::file_write_error, path_str),
720 "container_system"});
721 }
722 bytes_written += to_write;
723 if (callback) {
724 callback(bytes_written, total_size);
725 }
726 }
727
728 return kcenon::common::ok();
729 });
730 co_return result;
731 }

838 inline generator<std::vector<uint8_t>>
839 async_container::serialize_chunked(size_t chunk_size) const
840 {
841 auto data_result = container_->serialize(value_container::serialization_format::binary);
842 if (!data_result.is_ok()) {
843 co_return;
844 }
845
846 const auto& full_data = data_result.value();
847 const size_t total_size = full_data.size();
848 size_t offset = 0;
849
850 while (offset < total_size) {
851 size_t current_chunk_size = std::min(chunk_size, total_size - offset);
852 std::vector<uint8_t> chunk(
853 full_data.begin() + static_cast<std::ptrdiff_t>(offset),
854 full_data.begin() + static_cast<std::ptrdiff_t>(offset + current_chunk_size));
855 offset += current_chunk_size;
856 co_yield chunk;
857 }
858 }
