57 template<typename T, size_t PoolSize = 128>
58 class value_pool
59 {
60 public:
64 static value_pool& instance()
65 {
66 static value_pool pool;
67 return pool;
68 }
69
76 template<typename... Args>
77 std::shared_ptr<T> allocate(Args&&... args)
78 {
79 bool reused = false;
80 auto unique = pool_.acquire(&reused, std::forward<Args>(args)...);
81
82 if (reused)
83 {
84 pool_hits_.fetch_add(1, std::memory_order_relaxed);
85 available_count_.fetch_sub(1, std::memory_order_release);
86 }
87 else
88 {
89 pool_misses_.fetch_add(1, std::memory_order_relaxed);
90 }
91
92 T* ptr = unique.release();
93 return std::shared_ptr<T>(ptr, [this](T* p) {
94 this->pool_.release(p);
95 this->available_count_.fetch_add(1, std::memory_order_release);
96 });
97 }
98
103 auto stats() const
104 {
105 return std::make_tuple(
106 pool_hits_.load(std::memory_order_relaxed),
107 pool_misses_.load(std::memory_order_relaxed),
108 available_count_.load(std::memory_order_relaxed)
109 );
110 }
111
116 double hit_rate() const
117 {
118 auto hits = pool_hits_.load(std::memory_order_relaxed);
119 auto misses = pool_misses_.load(std::memory_order_relaxed);
120 auto total = hits + misses;
121
122 return total > 0 ? static_cast<double>(hits) / total : 0.0;
123 }
124
128 void reserve(std::size_t count)
129 {
130 pool_.reserve(count);
131 available_count_.fetch_add(count, std::memory_order_release);
132 }

137 void clear()
138 {
139 pool_.clear();

151 clear();
152 }
