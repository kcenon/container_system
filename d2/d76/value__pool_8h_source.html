55 template<typename T, size_t PoolSize = 128>
56 class value_pool
57 {
58 public:
62 static value_pool& instance()
63 {
64 static value_pool pool;
65 return pool;
66 }

74 template<typename... Args>
75 std::shared_ptr<T> allocate(Args&&... args)
76 {
77 T* ptr = nullptr;
78
79 // Try to get from pool (lock-free fast path)
80 if (available_count_.load(std::memory_order_acquire) > 0)
81 {
82 std::lock_guard<std::mutex> lock(mutex_);
83 if (!free_list_.empty())
84 {
85 ptr = free_list_.back();
86 free_list_.pop_back();
87 available_count_.fetch_sub(1, std::memory_order_release);
88 pool_hits_.fetch_add(1, std::memory_order_relaxed);
89
90 // Placement new with forwarded arguments
91 new (ptr) T(std::forward<Args>(args)...);
92
93 return std::shared_ptr<T>(ptr, [this](T* p) {
94 this->deallocate(p);
95 });
96 }
97 }
98
99 // Pool miss - allocate new memory
100 pool_misses_.fetch_add(1, std::memory_order_relaxed);
101 ptr = new T(std::forward<Args>(args)...);
102
103 return std::shared_ptr<T>(ptr, [this](T* p) {
104 this->deallocate(p);
105 });
106 }
107
112 auto stats() const
113 {
114 return std::make_tuple(
115 pool_hits_.load(std::memory_order_relaxed),
116 pool_misses_.load(std::memory_order_relaxed),
117 available_count_.load(std::memory_order_relaxed)
118 );
119 }
120
125 double hit_rate() const
126 {
127 auto hits = pool_hits_.load(std::memory_order_relaxed);
128 auto misses = pool_misses_.load(std::memory_order_relaxed);
129 auto total = hits + misses;
130
131 return total > 0 ? static_cast<double>(hits) / total : 0.0;
132 }
133
137 void clear()
138 {
139 std::lock_guard<std::mutex> lock(mutex_);
140
141 for (auto ptr : free_list_)
142 {
143 delete ptr;
144 }
145 free_list_.clear();
146 available_count_.store(0, std::memory_order_release);
147 }
148
149 // Prevent copying and moving
150 value_pool(const value_pool&) = delete;
151 value_pool& operator=(const value_pool&) = delete;
152 value_pool(value_pool&&) = delete;
153 value_pool& operator=(value_pool&&) = delete;
154
155 ~value_pool()
156 {
157 clear();
158 }
159
160 private:
161 value_pool() = default;
162
166 void deallocate(T* ptr)
167 {
168 if (!ptr) return;
169
170 // Explicitly call destructor
171 ptr->~T();
172
173 // Try to return to pool if not full
174 std::lock_guard<std::mutex> lock(mutex_);
175 if (free_list_.size() < PoolSize)
176 {
177 free_list_.push_back(ptr);
178 available_count_.fetch_add(1, std::memory_order_release);
179 }
180 else
181 {
182 // Pool full - actually delete
183 delete ptr;
184 }
185 }
186
187 std::vector<T*> free_list_;
188 std::mutex mutex_;
189 std::atomic<size_t> available_count_{0};
190 std::atomic<size_t> pool_hits_{0};
191 std::atomic<size_t> pool_misses_{0};
192 };
