19 // Thread-local set for detecting circular references during serialization
20 thread_local std::set<const void*> serializing_containers;
21
22 struct circular_ref_guard {
23 const void* ptr;
24 bool inserted;
25
26 explicit circular_ref_guard(const thread_safe_container* container)
27 : ptr(container), inserted(false) {
28 if (ptr) {
29 auto result = serializing_containers.insert(ptr);
30 inserted = result.second;
31 }
32 }
33
34 ~circular_ref_guard() {
35 if (inserted && ptr) {
36 serializing_containers.erase(ptr);
37 }
38 }
39
40 bool is_circular() const { return ptr && !inserted; }
41 };

72 if (!values[i] && !other.values[i]) continue;
73 if (!values[i] || !other.values[i]) return false;
74 if (*values[i] != *other.values[i]) return false;

138 value_types value::type() const {
139 std::shared_lock lock(mutex_);
140 size_t idx = data_.index();
141
142 // Direct mapping: variant index == value_types enum value
143 // Handle platform-specific llong/ullong
144 if constexpr (!has_separate_llong) {
145 // On macOS/Linux, llong_value (8) and ullong_value (9) are monostate
146 // These should never be created directly, map to long_value instead
147 if (idx == 8) return value_types::long_value; // llong ‚Üí long
148 if (idx == 9) return value_types::ulong_value; // ullong ‚Üí ulong
149 }
150
151 return static_cast<value_types>(idx);
152 }

154 std::string value::to_string() const {
155 return visit([](auto&& value) -> std::string {
156 using T = std::decay_t<decltype(value)>;
157
158 if constexpr (std::is_same_v<T, std::monostate>) {
159 return "null";
160 }
161 else if constexpr (std::is_same_v<T, bool>) {
162 return value ? "true" : "false";
163 }
164 else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
165 std::ostringstream oss;
166 oss << std::hex << std::setfill('0');
167 for (auto byte : value) {
168 oss << std::setw(2) << static_cast<int>(byte);
169 }
170 return oss.str();
171 }
172 else if constexpr (std::is_arithmetic_v<T>) {
173 return std::to_string(value);
174 }
175 else if constexpr (std::is_same_v<T, std::string>) {
176 return value;
177 }
178 else if constexpr (std::is_same_v<T, std::shared_ptr<thread_safe_container>>) {
179 return value ? value->to_json() : "null";
180 }
181 else if constexpr (std::is_same_v<T, array_variant>) {
182 std::string result = "[";
183 for (size_t i = 0; i < value.values.size(); ++i) {
184 if (i > 0) result += ",";
185 result += value.values[i] ? value.values[i]->to_string() : "null";
186 }
187 result += "]";
188 return result;
189 }
190 return "";
191 });
192 }

194 std::string value::to_json() const {
195 auto var_name = name();
196 auto var_type = type();
197
198 return visit([var_name, var_type](auto&& value) -> std::string {
199 using T = std::decay_t<decltype(value)>;
200
201 // Manual JSON header formatting (no std::format dependency)
202 std::string result = "{\"name\":\"" + std::string(var_name) +
203 "\",\"type\":" + std::to_string(static_cast<int>(var_type)) +
204 ",\"value\":";
205
206 if constexpr (std::is_same_v<T, std::monostate>) {
207 result += "null";
208 }
209 else if constexpr (std::is_same_v<T, bool>) {
210 result += value ? "true" : "false";
211 }
212 else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
213 result += "\"";
214 std::ostringstream oss;
215 oss << std::hex << std::setfill('0');
216 for (auto byte : value) {
217 oss << std::setw(2) << static_cast<int>(byte);
218 }
219 result += oss.str();
220 result += "\"";
221 }
222 else if constexpr (std::is_arithmetic_v<T>) {
223 result += std::to_string(value);
224 }
225 else if constexpr (std::is_same_v<T, std::string>) {
226 // Escape string for JSON
227 result += "\"";
228 for (char c : value) {
229 switch (c) {
230 case '"': result += "\\\""; break;
231 case '\\': result += "\\\\"; break;
232 case '\b': result += "\\b"; break;
233 case '\f': result += "\\f"; break;
234 case '\n': result += "\\n"; break;
235 case '\r': result += "\\r"; break;
236 case '\t': result += "\\t"; break;
237 default:
238 if (c >= 0x20 && c <= 0x7E) {
239 result += c;
240 } else {
241 // Manual Unicode escape formatting (no std::format dependency)
242 char buf[7]; // "\uXXXX" + null terminator
243 std::snprintf(buf, sizeof(buf), "\\u%04x", static_cast<unsigned char>(c));
244 result += buf;
245 }
246 }
247 }
248 result += "\"";
249 }
250 else if constexpr (std::is_same_v<T, std::shared_ptr<thread_safe_container>>) {
251 result += value ? value->to_json() : "null";
252 }
253 else if constexpr (std::is_same_v<T, array_variant>) {
254 result += "[";
255 for (size_t i = 0; i < value.values.size(); ++i) {
256 if (i > 0) result += ",";
257 result += value.values[i] ? value.values[i]->to_json() : "null";
258 }
259 result += "]";
260 }
261
262 result += "}";
263 return result;
264 });

267 void value::serialize_data(std::vector<uint8_t>& result) const {
268 visit([&result](auto&& val) {
269 using T = std::decay_t<decltype(val)>;
270
271 if constexpr (std::is_same_v<T, std::monostate>) {
272 // Null val: no data
273 }
274 else if constexpr (std::is_same_v<T, bool>) {
275 result.push_back(val ? 1 : 0);
276 }
277 else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
278 // Bytes: [length:4][data:length]
279 uint32_t size = static_cast<uint32_t>(val.size());
280 result.insert(result.end(),
281 reinterpret_cast<const uint8_t*>(&size),
282 reinterpret_cast<const uint8_t*>(&size) + sizeof(size));
283 result.insert(result.end(), val.begin(), val.end());
284 }
285 else if constexpr (std::is_arithmetic_v<T> && !std::is_same_v<T, bool>) {
286 // Numeric: raw bytes
287 result.insert(result.end(),
288 reinterpret_cast<const uint8_t*>(&val),
289 reinterpret_cast<const uint8_t*>(&val) + sizeof(T));
290 }
291 else if constexpr (std::is_same_v<T, std::string>) {
292 // String: [length:4][UTF-8 data:length]
293 uint32_t size = static_cast<uint32_t>(val.size());
294 result.insert(result.end(),
295 reinterpret_cast<const uint8_t*>(&size),
296 reinterpret_cast<const uint8_t*>(&size) + sizeof(size));
297 result.insert(result.end(), val.begin(), val.end());
298 }
299 else if constexpr (std::is_same_v<T, std::shared_ptr<thread_safe_container>>) {
300 // Container: [serialized_size:4][serialized_data:size]
301 if (val) {
302 // Check for circular reference
303 detail::circular_ref_guard guard(val.get());
304 if (guard.is_circular()) {
305 // Circular reference detected - serialize as null container
306 uint32_t size = 0;
307 result.insert(result.end(),
308 reinterpret_cast<const uint8_t*>(&size),
309 reinterpret_cast<const uint8_t*>(&size) + sizeof(size));
310 } else {
311 auto container_data = val->serialize();
312 uint32_t size = static_cast<uint32_t>(container_data.size());
313 result.insert(result.end(),
314 reinterpret_cast<const uint8_t*>(&size),
315 reinterpret_cast<const uint8_t*>(&size) + sizeof(size));
316 result.insert(result.end(), container_data.begin(), container_data.end());
317 }
318 } else {
319 uint32_t size = 0;
320 result.insert(result.end(),
321 reinterpret_cast<const uint8_t*>(&size),
322 reinterpret_cast<const uint8_t*>(&size) + sizeof(size));
323 }
324 }
325 else if constexpr (std::is_same_v<T, array_variant>) {
326 // Array: [count:4][value1_serialized][value2_serialized]...
327 uint32_t count = static_cast<uint32_t>(val.values.size());
328 result.insert(result.end(),
329 reinterpret_cast<const uint8_t*>(&count),
330 reinterpret_cast<const uint8_t*>(&count) + sizeof(count));
331
332 for (const auto& elem : val.values) {
333 if (elem) {
334 auto elem_data = elem->serialize();
335 result.insert(result.end(), elem_data.begin(), elem_data.end());
336 } else {
337 // Null element: serialize as null_value with empty name
338 container_module::value null_elem("");
339 auto null_data = null_elem.serialize();
340 result.insert(result.end(), null_data.begin(), null_data.end());
341 }
342 }
343 }
344 });
345 }

347 std::vector<uint8_t> value::serialize() const {
348 std::vector<uint8_t> result;
349
350 // Header: [name_length:4][name:UTF-8][type:1]
351 uint32_t name_len = static_cast<uint32_t>(name_.size());
352 result.insert(result.end(),
353 reinterpret_cast<const uint8_t*>(&name_len),
354 reinterpret_cast<const uint8_t*>(&name_len) + sizeof(name_len));
355 result.insert(result.end(), name_.begin(), name_.end());
356
357 // Type byte: use value_types enum value (0-15)
358 uint8_t type_byte = static_cast<uint8_t>(type());
359 result.push_back(type_byte);
360
361 // Data: type-specific serialization
362 serialize_data(result);
363
364 return result;
365 }

367 bool value::deserialize_data(value& result,
368 value_types type,
369 const std::vector<uint8_t>& data,
370 size_t& offset) {
371 switch (type) {
372 case value_types::null_value:
373 result.data_.emplace<0>(); // null is at index 0
374 return true;
375
376 case value_types::bool_value:
377 if (offset >= data.size()) return false;
378 result.data_ = (data[offset++] != 0);
379 return true;
380
381 case value_types::short_value: {
382 if (offset + sizeof(int16_t) > data.size()) return false;
383 int16_t value;
384 std::memcpy(&value, data.data() + offset, sizeof(value));
385 offset += sizeof(value);
386 result.data_ = value;
387 return true;
388 }
389
390 case value_types::ushort_value: {
391 if (offset + sizeof(uint16_t) > data.size()) return false;
392 uint16_t value;
393 std::memcpy(&value, data.data() + offset, sizeof(value));
394 offset += sizeof(value);
395 result.data_ = value;
396 return true;
397 }
398
399 case value_types::int_value: {
400 if (offset + sizeof(int32_t) > data.size()) return false;
401 int32_t value;
402 std::memcpy(&value, data.data() + offset, sizeof(value));
403 offset += sizeof(value);
404 result.data_ = value;
405 return true;
406 }
407
408 case value_types::uint_value: {
409 if (offset + sizeof(uint32_t) > data.size()) return false;
410 uint32_t value;
411 std::memcpy(&value, data.data() + offset, sizeof(value));
412 offset += sizeof(value);
413 result.data_ = value;
414 return true;
415 }
416
417 case value_types::long_value: {
418 if (offset + sizeof(int64_t) > data.size()) return false;
419 int64_t value;
420 std::memcpy(&value, data.data() + offset, sizeof(value));
421 offset += sizeof(value);
422 result.data_ = value;
423 return true;
424 }
425
426 case value_types::llong_value: {
427 if (offset + sizeof(int64_t) > data.size()) return false;
428#if defined(_MSC_VER) && _MSC_VER < 1900
429 // Old MSVC (2013-): int64_t is __int64, long long is separate
430 long long value;
431 std::memcpy(&value, data.data() + offset, sizeof(value));
432 offset += sizeof(value);
433 result.data_ = value; // Stored at index 8 (llong_value slot)
434#else
435 // Modern platforms (macOS, Linux, MSVC 2015+): int64_t == long long
436 // llong_value uses same storage as long_value (int64_t)
437 int64_t value;
438 std::memcpy(&value, data.data() + offset, sizeof(value));
439 offset += sizeof(value);
440 result.data_ = value; // Stored at index 6 (long_value slot)
441#endif
442 return true;
443 }
444
445 case value_types::ulong_value: {
446 if (offset + sizeof(uint64_t) > data.size()) return false;
447 uint64_t value;
448 std::memcpy(&value, data.data() + offset, sizeof(value));
449 offset += sizeof(value);
450 result.data_ = value;
451 return true;
452 }
453
454 case value_types::ullong_value: {
455 if (offset + sizeof(uint64_t) > data.size()) return false;
456#if defined(_MSC_VER) && _MSC_VER < 1900
457 // Old MSVC (2013-): uint64_t is unsigned __int64, unsigned long long is separate
458 unsigned long long value;
459 std::memcpy(&value, data.data() + offset, sizeof(value));
460 offset += sizeof(value);
461 result.data_ = value; // Stored at index 9 (ullong_value slot)
462#else
463 // Modern platforms (macOS, Linux, MSVC 2015+): uint64_t == unsigned long long
464 // ullong_value uses same storage as ulong_value (uint64_t)
465 uint64_t value;
466 std::memcpy(&value, data.data() + offset, sizeof(value));
467 offset += sizeof(value);
468 result.data_ = value; // Stored at index 7 (ulong_value slot)
469#endif
470 return true;
471 }
472
473 case value_types::float_value: {
474 if (offset + sizeof(float) > data.size()) return false;
475 float value;
476 std::memcpy(&value, data.data() + offset, sizeof(value));
477 offset += sizeof(value);
478 result.data_ = value;
479 return true;
480 }
481
482 case value_types::double_value: {
483 if (offset + sizeof(double) > data.size()) return false;
484 double value;
485 std::memcpy(&value, data.data() + offset, sizeof(value));
486 offset += sizeof(value);
487 result.data_ = value;
488 return true;
489 }
490
491 case value_types::bytes_value: {
492 if (offset + sizeof(uint32_t) > data.size()) return false;
493 uint32_t size;
494 std::memcpy(&size, data.data() + offset, sizeof(size));
495 offset += sizeof(size);
496 if (offset + size > data.size()) return false;
497 std::vector<uint8_t> bytes(data.begin() + offset,
498 data.begin() + offset + size);
499 offset += size;
500 result.data_ = std::move(bytes);
501 return true;
502 }
503
504 case value_types::string_value: {
505 if (offset + sizeof(uint32_t) > data.size()) return false;
506 uint32_t size;
507 std::memcpy(&size, data.data() + offset, sizeof(size));
508 offset += sizeof(size);
509 if (offset + size > data.size()) return false;
510 std::string str(data.begin() + offset, data.begin() + offset + size);
511 offset += size;
512 result.data_ = std::move(str);
513 return true;
514 }
515
516 case value_types::container_value: {
517 if (offset + sizeof(uint32_t) > data.size()) return false;
518 uint32_t size;
519 std::memcpy(&size, data.data() + offset, sizeof(size));
520 offset += sizeof(size);
521 if (size == 0) {
522 result.data_ = std::shared_ptr<thread_safe_container>(nullptr);
523 return true;
524 }
525 if (offset + size > data.size()) return false;
526 std::vector<uint8_t> container_data(data.begin() + offset,
527 data.begin() + offset + size);
528 offset += size;
529 auto container = thread_safe_container::deserialize(container_data);
530 result.data_ = container;
531 return true;
532 }
533
534 case value_types::array_value: {
535 if (offset + sizeof(uint32_t) > data.size()) return false;
536 uint32_t count;
537 std::memcpy(&count, data.data() + offset, sizeof(count));
538 offset += sizeof(count);
539
540 array_variant arr;
541 arr.values.reserve(count);
542
543 for (uint32_t i = 0; i < count; ++i) {
544 auto elem = deserialize(std::vector<uint8_t>(data.begin() + offset, data.end()));
545 if (!elem) return false;
546
547 // Update offset based on elem's serialized size
548 auto elem_size = elem->serialize().size();
549 offset += elem_size;
550
551 arr.values.push_back(std::make_shared<value>(std::move(*elem)));
552 }
553
554 result.data_ = std::move(arr);
555 return true;
556 }
557
558 default:
559 return false;
560 }

563 std::optional<value> value::deserialize(const std::vector<uint8_t>& data) {
564 if (data.size() < sizeof(uint32_t) + 1) {
565 return std::nullopt; // Too small: need at least name_len + type
566 }
567
568 size_t offset = 0;
569
570 // Read name length
571 uint32_t name_len;
572 std::memcpy(&name_len, data.data() + offset, sizeof(name_len));
573 offset += sizeof(name_len);
574
575 if (offset + name_len + 1 > data.size()) {
576 return std::nullopt; // Invalid name length
577 }
578
579 // Read name
580 std::string name(data.begin() + offset, data.begin() + offset + name_len);
581 offset += name_len;
582
583 // Read type byte
584 uint8_t type_byte = data[offset++];
585 value_types type = static_cast<value_types>(type_byte);
586
587 // Validate type range
588 if (type_byte > 15) {
589 return std::nullopt; // Invalid type
590 }
591
592 // Create result with name
593 container_module::value result(name);
594
595 // Deserialize data based on type
596 if (!deserialize_data(result, type, data, offset)) {
597 return std::nullopt;
598 }
599
600 return result;
601 }
