183 constexpr std::string_view get_message(int code) noexcept
184 {
185 switch (code)
186 {
187 // Value operations (1xx)
188 case key_not_found: return "Key not found";
189 case type_mismatch: return "Type mismatch";
190 case value_out_of_range: return "Value out of range";
191 case invalid_value: return "Invalid value";
192 case key_already_exists: return "Key already exists";
193 case empty_key: return "Empty key";
194
195 // Serialization (2xx)
196 case serialization_failed: return "Serialization failed";
197 case deserialization_failed: return "Deserialization failed";
198 case invalid_format: return "Invalid format";
199 case version_mismatch: return "Version mismatch";
200 case corrupted_data: return "Corrupted data";
201 case header_parse_failed: return "Header parse failed";
202 case value_parse_failed: return "Value parse failed";
203 case encoding_error: return "Encoding error";
204
205 // Validation (3xx)
206 case schema_validation_failed: return "Schema validation failed";
207 case missing_required_field: return "Missing required field";
208 case constraint_violated: return "Constraint violated";
209 case type_constraint_violated: return "Type constraint violated";
210 case max_size_exceeded: return "Maximum size exceeded";
211
212 // Resource (4xx)
213 case memory_allocation_failed: return "Memory allocation failed";
214 case file_not_found: return "File not found";
215 case file_read_error: return "File read error";
216 case file_write_error: return "File write error";
217 case permission_denied: return "Permission denied";
218 case resource_exhausted: return "Resource exhausted";

221 // Thread safety (5xx)
222 case lock_acquisition_failed: return "Lock acquisition failed";
223 case concurrent_modification: return "Concurrent modification";
224 case lock_timeout: return "Lock timeout";

235 constexpr std::string_view get_category(int code) noexcept
236 {
237 if (code >= 100 && code < 200)
238 {
239 return "value_operation";
240 }
241 if (code >= 200 && code < 300)
242 {
243 return "serialization";
244 }
245 if (code >= 300 && code < 400)
246 {
247 return "validation";
248 }
249 if (code >= 400 && code < 500)
250 {
251 return "resource";
252 }
253 if (code >= 500 && code < 600)
254 {
255 return "thread_safety";
256 }
257 return "unknown";
258 }
