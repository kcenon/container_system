26 void* allocate() {
27 std::lock_guard<std::mutex> lock(mutex_);
28 if (!free_list_) {
29 try {
30 allocate_chunk_unlocked();
31 } catch (const std::bad_alloc&) {
32 // Chunk allocation failed, propagate the exception
33 throw;
34 }
35 // Verify that allocate_chunk_unlocked() actually created free blocks
36 if (!free_list_) {
37 throw std::runtime_error("Memory pool chunk allocation failed to create free list");
38 }
39 }
40 void* p = free_list_;
41 free_list_ = *reinterpret_cast<void**>(free_list_);
42 ++allocated_count_;
43 return p;
44 }

46 void deallocate(void* p) {
47 if (!p) return;
48 std::lock_guard<std::mutex> lock(mutex_);
49
50#ifndef NDEBUG
51 // Debug-mode validation: ensure pointer appears to be from our pool
52 // This is a simple sanity check, not a complete validation
53 bool found = false;
54 for (const auto& chunk : chunks_) {
55 std::uint8_t* chunk_start = chunk.get();
56 std::uint8_t* chunk_end = chunk_start + (block_size_ * blocks_per_chunk_);
57 if (reinterpret_cast<std::uint8_t*>(p) >= chunk_start &&
58 reinterpret_cast<std::uint8_t*>(p) < chunk_end) {
59 found = true;
60 break;
61 }
62 }
63 if (!found) {
64 // Pointer does not belong to any of our chunks
65 // In debug mode, we could log a warning or assert
66 // For now, we'll just return to avoid corruption
67 return;
68 }
69#endif
70
71 *reinterpret_cast<void**>(p) = free_list_;
72 free_list_ = p;
73 --allocated_count_;
74 }

129 std::uint8_t* base = chunk.get();
130 chunks_.push_back(std::move(chunk));
131 // Build free list
132 for (std::size_t i = 0; i < blocks_per_chunk_; ++i) {
133 void* p = base + i * block_size_;

135 free_list_ = p;
