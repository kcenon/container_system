32/*
33 * Simple fixed-block memory pool (prototype)
34 * Not wired into container_system by default; intended for experimentation.
35 */

42#include <stdexcept>

44namespace kcenon::container::internal {
45
46class fixed_block_pool {
47public:
48 explicit fixed_block_pool(std::size_t block_size,
49 std::size_t blocks_per_chunk = 1024)
50 : block_size_(block_size < sizeof(void*) ? sizeof(void*) : block_size)
51 , blocks_per_chunk_(blocks_per_chunk) {}
52
53 ~fixed_block_pool() = default;
54
55 void* allocate() {
56 std::lock_guard<std::mutex> lock(mutex_);
57 if (!free_list_) {
58 try {
59 allocate_chunk_unlocked();
60 } catch (const std::bad_alloc&) {
61 // Chunk allocation failed, propagate the exception
62 throw;
63 }
64 // Verify that allocate_chunk_unlocked() actually created free blocks
65 if (!free_list_) {
66 throw std::runtime_error("Memory pool chunk allocation failed to create free list");
67 }
68 }
69 void* p = free_list_;
70 free_list_ = *reinterpret_cast<void**>(free_list_);
71 ++allocated_count_;
72 return p;
73 }
74
75 void deallocate(void* p) {
76 if (!p) return;
77 std::lock_guard<std::mutex> lock(mutex_);
78
79 // Validate pointer belongs to one of our pool chunks.
80 // This lightweight range check runs in all build modes to
81 // prevent free-list corruption from invalid deallocations.
82 bool found = false;
83 for (const auto& chunk : chunks_) {
84 std::uint8_t* chunk_start = chunk.get();
85 std::uint8_t* chunk_end = chunk_start + (block_size_ * blocks_per_chunk_);
86 if (reinterpret_cast<std::uint8_t*>(p) >= chunk_start &&
87 reinterpret_cast<std::uint8_t*>(p) < chunk_end) {
88 found = true;
89 break;
90 }
91 }
92 if (!found) {
93 return;
94 }
95
96 *reinterpret_cast<void**>(p) = free_list_;
97 free_list_ = p;
98 --allocated_count_;
99 }
100
101 std::size_t block_size() const noexcept { return block_size_; }
102
106 struct statistics {
107 std::size_t total_chunks; // Number of chunks allocated
108 std::size_t allocated_blocks; // Number of blocks currently in use
109 std::size_t total_capacity; // Total number of blocks across all chunks
110 std::size_t free_blocks; // Number of blocks in free list
111
112 // Calculated metrics
113 double utilization_ratio() const {
114 return total_capacity > 0
115 ? static_cast<double>(allocated_blocks) / total_capacity
116 : 0.0;
117 }
118
119 std::size_t memory_bytes() const {
120 // This would need block_size, which we don't have in this struct
121 // For now, just return 0; caller can calculate if needed
122 return 0;
123 }
124 };
125
130 statistics get_statistics() const {
131 std::lock_guard<std::mutex> lock(mutex_);
132
133 // Count free blocks in the free list
134 std::size_t free_count = 0;
135 void* current = free_list_;
136 while (current) {
137 ++free_count;
138 current = *reinterpret_cast<void**>(current);
139 }
140
141 std::size_t total_capacity = chunks_.size() * blocks_per_chunk_;
142
143 return statistics{
144 chunks_.size(),
145 allocated_count_,
146 total_capacity,
147 free_count
148 };
149 }
150
151private:
152 void allocate_chunk_unlocked() {
153 std::unique_ptr<std::uint8_t[]> chunk(new std::uint8_t[block_size_ * blocks_per_chunk_]);
154 std::uint8_t* base = chunk.get();
155 chunks_.push_back(std::move(chunk));
156 // Build free list
157 for (std::size_t i = 0; i < blocks_per_chunk_; ++i) {
158 void* p = base + i * block_size_;
159 *reinterpret_cast<void**>(p) = free_list_;
160 free_list_ = p;
161 }
162 }
163
164private:
165 std::size_t block_size_;
166 std::size_t blocks_per_chunk_;
167 std::vector<std::unique_ptr<std::uint8_t[]>> chunks_;
168 void* free_list_{nullptr};
169 mutable std::mutex mutex_; // Mutable to allow locking in const methods
170 std::size_t allocated_count_{0}; // Track number of currently allocated blocks
171};
